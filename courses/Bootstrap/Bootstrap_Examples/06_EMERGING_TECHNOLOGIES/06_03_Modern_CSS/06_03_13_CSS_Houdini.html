3. Paint Worklet Code

// worklet.js - Paint Worklet file
class DottedBorderPainter {
    // Get custom properties used by this worklet
    static get inputProperties() {
        return ['--dot-color', '--dot-size'];
    }

    // Paint method: draws on a canvas-like context
    paint(ctx, size, properties) {
        const dotColor = properties.get('--dot-color').toString().trim();
        const dotSize = parseFloat(properties.get('--dot-size').toString());

        ctx.fillStyle = dotColor || '#6c5ce7';
        const step = dotSize * 3;

        // Draw dots around the border
        for (let x = 0; x < size.width; x += step) {
            ctx.beginPath();
            ctx.arc(x, dotSize, dotSize / 2, 0, 2 * Math.PI);
            ctx.arc(x, size.height - dotSize, dotSize / 2, 0, 2 * Math.PI);
            ctx.fill();
        }
    }
}

// Register the worklet
registerPaint('dotted-border', DottedBorderPainter);

// CSS usage:
// .element {
//   --dot-color: #6c5ce7;
//   --dot-size: 4;
//   border-image: paint(dotted-border) 30;
// }
