// lib/theme/bootstrap_theme.dart
import 'package:flutter/material.dart';

// Bootstrap 5 color palette mapped to Flutter
class BootstrapColors {
  static const Color primary = Color(0xFF0d6efd);
  static const Color secondary = Color(0xFF6c757d);
  static const Color success = Color(0xFF198754);
  static const Color danger = Color(0xFFdc3545);
  static const Color warning = Color(0xFFffc107);
  static const Color info = Color(0xFF0dcaf0);
  static const Color light = Color(0xFFf8f9fa);
  static const Color dark = Color(0xFF212529);
  static const Color white = Color(0xFFffffff);
}

// Bootstrap spacing scale (4px base unit)
class BootstrapSpacing {
  static const double xs = 4.0;    // Bootstrap spacer * 0.25
  static const double sm = 8.0;    // Bootstrap spacer * 0.5
  static const double md = 16.0;   // Bootstrap spacer * 1
  static const double lg = 24.0;   // Bootstrap spacer * 1.5
  static const double xl = 32.0;   // Bootstrap spacer * 2
  static const double xxl = 48.0;  // Bootstrap spacer * 3
}

// ThemeData replicating Bootstrap's visual language
ThemeData bootstrapTheme() {
  return ThemeData(
    useMaterial3: true,
    colorScheme: ColorScheme.fromSeed(
      seedColor: BootstrapColors.primary,
      primary: BootstrapColors.primary,
      secondary: BootstrapColors.secondary,
      error: BootstrapColors.danger,
      surface: BootstrapColors.white,
    ),
    // Typography matching Bootstrap's font sizes
    textTheme: const TextTheme(
      displayLarge: TextStyle(fontSize: 57, fontWeight: FontWeight.w300),   // display-1
      displayMedium: TextStyle(fontSize: 45, fontWeight: FontWeight.w300),  // display-2
      headlineLarge: TextStyle(fontSize: 32, fontWeight: FontWeight.bold),  // h1
      headlineMedium: TextStyle(fontSize: 28, fontWeight: FontWeight.bold), // h2
      titleLarge: TextStyle(fontSize: 22, fontWeight: FontWeight.w600),     // h3
      bodyLarge: TextStyle(fontSize: 16),                                    // body text
      bodyMedium: TextStyle(fontSize: 14),                                   // small
      labelLarge: TextStyle(fontSize: 16, fontWeight: FontWeight.w600),     // btn text
    ),
    // Button theme matching Bootstrap buttons
    elevatedButtonTheme: ElevatedButtonThemeData(
      style: ElevatedButton.styleFrom(
        backgroundColor: BootstrapColors.primary,
        foregroundColor: BootstrapColors.white,
        padding: const EdgeInsets.symmetric(horizontal: 20, vertical: 12),
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(6),  // Bootstrap border-radius
        ),
      ),
    ),
    // Card theme matching Bootstrap cards
    cardTheme: CardTheme(
      elevation: 2,                               // Bootstrap shadow-sm
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(8),
        side: BorderSide(color: Colors.grey.shade300),
      ),
      margin: const EdgeInsets.only(bottom: 16),
    ),
    // Input decoration matching Bootstrap form controls
    inputDecorationTheme: InputDecorationTheme(
      border: OutlineInputBorder(
        borderRadius: BorderRadius.circular(6),
        borderSide: const BorderSide(color: Color(0xFFdee2e6)),
      ),
      contentPadding: const EdgeInsets.symmetric(horizontal: 12, vertical: 10),
    ),
    // AppBar matching Bootstrap navbar-dark
    appBarTheme: const AppBarTheme(
      backgroundColor: BootstrapColors.dark,
      foregroundColor: BootstrapColors.white,
      elevation: 0,
    ),
  );
}
Flutter Bootstrap-Style Widgets
// lib/widgets/bootstrap_card.dart
class BootstrapCard extends StatelessWidget {
  final String? title;
  final Widget child;
  final Color? headerColor;
  final Color? headerTextColor;

  const BootstrapCard({
    this.title,
    required this.child,
    this.headerColor,
    this.headerTextColor = Colors.white,
  });

  @override
  Widget build(BuildContext context) {
    return Card(
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          // Card header (like Bootstrap .card-header)
          if (title != null)
            Container(
              width: double.infinity,
              padding: const EdgeInsets.all(12),
              decoration: BoxDecoration(
                color: headerColor ?? Theme.of(context).primaryColor,
                borderRadius: const BorderRadius.only(
                  topLeft: Radius.circular(8),
                  topRight: Radius.circular(8),
                ),
              ),
              child: Text(
                title!,
                style: TextStyle(
                  color: headerTextColor,
                  fontWeight: FontWeight.bold,
                ),
              ),
            ),
          // Card body (like Bootstrap .card-body)
          Padding(padding: const EdgeInsets.all(16), child: child),
        ],
      ),
    );
  }
}

// lib/widgets/bootstrap_alert.dart
class BootstrapAlert extends StatelessWidget {
  final String message;
  final String type; // 'success', 'danger', 'warning', 'info'
  final VoidCallback? onDismiss;

  const BootstrapAlert({
    required this.message,
    this.type = 'info',
    this.onDismiss,
  });

  Color get backgroundColor {
    switch (type) {
      case 'success': return const Color(0xFFd1e7dd);
      case 'danger': return const Color(0xFFf8d7da);
      case 'warning': return const Color(0xFFfff3cd);
      default: return const Color(0xFFcff4fc);
    }
  }

  @override
  Widget build(BuildContext context) {
    return Container(
      padding: const EdgeInsets.all(12),
      margin: const EdgeInsets.only(bottom: 16),
      decoration: BoxDecoration(
        color: backgroundColor,
        borderRadius: BorderRadius.circular(6),
      ),
      child: Row(
        children: [
          Expanded(child: Text(message)),
          if (onDismiss != null)
            IconButton(
              icon: const Icon(Icons.close, size: 16),
              onPressed: onDismiss,
            ),
        ],
      ),
    );
  }
}

// lib/widgets/bootstrap_list_group.dart
class BootstrapListGroup extends StatelessWidget {
  final List<String> items;
  final int? selectedIndex;
  final ValueChanged<int>? onTap;

  const BootstrapListGroup({required this.items, this.selectedIndex, this.onTap});

  @override
  Widget build(BuildContext context) {
    return Container(
      decoration: BoxDecoration(
        border: Border.all(color: const Color(0xFFdee2e6)),
        borderRadius: BorderRadius.circular(8),
      ),
      child: ListView.separated(
        shrinkWrap: true,
        physics: const NeverScrollableScrollPhysics(),
        itemCount: items.length,
        separatorBuilder: (_, __) => const Divider(height: 1),
        itemBuilder: (context, index) {
          return ListTile(
            title: Text(items[index]),
            selected: index == selectedIndex,
            selectedTileColor: BootstrapColors.primary.withOpacity(0.1),
            onTap: () => onTap?.call(index),
          );
        },
      ),
    );
  }
}
Flutter Responsive Grid (Row/Col)
// lib/widgets/bootstrap_grid.dart
class BootstrapRow extends StatelessWidget {
  final List<BootstrapCol> children;
  final MainAxisAlignment mainAxisAlignment;

  const BootstrapRow({
    required this.children,
    this.mainAxisAlignment = MainAxisAlignment.start,
  });

  @override
  Widget build(BuildContext context) {
    return Row(
      crossAxisAlignment: CrossAxisAlignment.start,
      mainAxisAlignment: mainAxisAlignment,
      children: children,
    );
  }
}

class BootstrapCol extends StatelessWidget {
  final int? sm; // col-sm-{n}
  final int? md; // col-md-{n}
  final int? lg; // col-lg-{n}
  final int? xl; // col-xl-{n}
  final Widget child;

  const BootstrapCol({
    this.sm, this.md, this.lg, this.xl,
    required this.child,
  });

  // Calculate flex based on screen width and breakpoint
  int _getFlex(BoxConstraints constraints) {
    final width = constraints.maxWidth;
    if (xl != null && width >= 1200) return xl!;
    if (lg != null && width >= 992) return lg!;
    if (md != null && width >= 768) return md!;
    if (sm != null && width >= 576) return sm!;
    return 12; // col-12 (full width) on smallest screens
  }

  @override
  Widget build(BuildContext context) {
    return LayoutBuilder(
      builder: (context, constraints) {
        final flex = _getFlex(constraints);
        // Bootstrap uses 12-column grid: col-6 = 50% = flex 6
        return Expanded(
          flex: flex,
          child: Padding(
            padding: const EdgeInsets.symmetric(horizontal: 8), // Gutter
            child: child,
          ),
        );
      },
    );
  }
}

Complete Flutter Screen Example
// lib/screens/home_screen.dart
class HomeScreen extends StatefulWidget {
  @override
  _HomeScreenState createState() => _HomeScreenState();
}

class _HomeScreenState extends State<HomeScreen> {
  int _counter = 0;
  final List<String> _todos = ['Learn Flutter', 'Build Bootstrap theme'];
  final TextEditingController _todoController = TextEditingController();

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      // Bootstrap navbar equivalent
      appBar: AppBar(
        title: const Text('Flutter Bootstrap App'),
        actions: [
          IconButton(icon: const Icon(Icons.settings), onPressed: () {}),
        ],
      ),
      body: SingleChildScrollView(
        // Bootstrap container equivalent
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [
            // Bootstrap alert
            const BootstrapAlert(
              type: 'success',
              message: 'Welcome to Flutter with Bootstrap patterns!',
            ),

            // Bootstrap card with counter
            BootstrapCard(
              title: 'Counter',
              headerColor: BootstrapColors.primary,
              child: Column(
                children: [
                  Text('Count: $_counter', style: Theme.of(context).textTheme.headlineLarge),
                  const SizedBox(height: 16),
                  // Bootstrap button group
                  Row(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      ElevatedButton(
                        style: ElevatedButton.styleFrom(backgroundColor: BootstrapColors.danger),
                        onPressed: () => setState(() => _counter--),
                        child: const Text('- Decrement'),
                      ),
                      const SizedBox(width: 8),
                      OutlinedButton(
                        onPressed: () => setState(() => _counter = 0),
                        child: const Text('Reset'),
                      ),
                      const SizedBox(width: 8),
                      ElevatedButton(
                        style: ElevatedButton.styleFrom(backgroundColor: BootstrapColors.success),
                        onPressed: () => setState(() => _counter++),
                        child: const Text('+ Increment'),
                      ),
                    ],
                  ),
                ],
              ),
            ),

            // Bootstrap card with todo list
            BootstrapCard(
              title: 'Todo List',
              headerColor: BootstrapColors.success,
              child: Column(
                children: [
                  // Bootstrap form input
                  Row(
                    children: [
                      Expanded(
                        child: TextFormField(
                          controller: _todoController,
                          decoration: const InputDecoration(hintText: 'Add a todo...'),
                        ),
                      ),
                      const SizedBox(width: 8),
                      ElevatedButton(
                        onPressed: () {
                          if (_todoController.text.isNotEmpty) {
                            setState(() {
                              _todos.add(_todoController.text);
                              _todoController.clear();
                            });
                          }
                        },
                        child: const Text('Add'),
                      ),
                    ],
                  ),
                  const SizedBox(height: 16),
                  // Bootstrap list group
                  ..._todos.asMap().entries.map((entry) =>
                    ListTile(
                      title: Text(entry.value),
                      trailing: IconButton(
                        icon: const Icon(Icons.delete, color: BootstrapColors.danger),
                        onPressed: () => setState(() => _todos.removeAt(entry.key)),
                      ),
                    ),
                  ),
                ],
              ),
            ),
          ],
        ),
      ),
      // Bootstrap floating action button
      floatingActionButton: FloatingActionButton(
        backgroundColor: BootstrapColors.primary,
        onPressed: () {},
        child: const Icon(Icons.add),
      ),
    );
  }
}
