Astro + Bootstrap 5

Islands Architecture, Partial Hydration, Content Collections, Multi-Framework

Astro Pattern: Zero JS by default. Use client:* directives to hydrate interactive components. Static HTML ships to the browser instantly.

Islands Architecture (client:* directives)
---
// pages/index.astro - Mixing static and interactive components
import Layout from '../layouts/Layout.astro';
import Navbar from '../components/Navbar.astro';       // Static - no JS
import Counter from '../components/Counter.tsx';        // React island
import TodoList from '../components/TodoList.vue';      // Vue island
import Slider from '../components/Slider.svelte';       // Svelte island
---

<Layout title="Islands Demo">
  <!-- STATIC: No JS shipped, renders at build time -->
  <div class="card mb-4">
    <div class="card-body">
      <h5>Static Component (0 KB JS)</h5>
      <p>This content is pure HTML, no JavaScript needed.</p>
    </div>
  </div>

  <!-- ISLAND: Hydrated immediately on page load -->
  <!-- client:load - hydrate eagerly, needed for above-fold interactivity -->
  <Counter client:load />

  <!-- ISLAND: Hydrated when browser is idle -->
  <!-- client:idle - good for below-fold interactive content -->
  <TodoList client:idle />

  <!-- ISLAND: Hydrated when scrolled into viewport -->
  <!-- client:visible - best for heavy components far down the page -->
  <Slider client:visible />

  <!-- ISLAND: Hydrated when media query matches -->
  <!-- client:media - responsive hydration -->
  <MobileNav client:media="(max-width: 768px)" />

  <!-- ISLAND: Hydrated only on specific framework's load -->
  <!-- client:only - skip server rendering, hydrate on client only -->
  <Chart client:only="react" />
</Layout>
Multi-Framework Islands
// astro.config.mjs - Enable framework integrations
import { defineConfig } from 'astro/config';
import react from '@astrojs/react';
import vue from '@astrojs/vue';
import svelte from '@astrojs/svelte';

export default defineConfig({
  integrations: [react(), vue(), svelte()],
});

// components/Counter.tsx - React Island
import { useState } from 'react';
export default function Counter() {
  const [count, setCount] = useState(0);
  return (
    <div className="card shadow-sm mb-4">
      <div className="card-header bg-primary text-white">
        React Island (client:load)
      </div>
      <div className="card-body text-center">
        <p className="display-4">Count: {count}</p>
        <div className="btn-group">
          <button className="btn btn-outline-danger" onClick={() => setCount(c => c - 1)}>-</button>
          <button className="btn btn-outline-success" onClick={() => setCount(c => c + 1)}>+</button>
        </div>
      </div>
    </div>
  );
}

// components/TodoList.vue - Vue Island

Content Collections
// src/content/config.ts - Define content schema
import { defineCollection, z } from 'astro:content';

const blog = defineCollection({
  schema: z.object({
    title: z.string(),
    date: z.date(),
    author: z.string(),
    tags: z.array(z.string()),
    image: z.string().optional(),
    excerpt: z.string(),
  }),
});

export const collections = { blog };

// src/pages/blog/index.astro - List blog posts
---
import { getCollection } from 'astro:content';
import Layout from '../../layouts/Layout.astro';

// getCollection fetches and validates all posts against schema
const posts = await getCollection('blog');
const sortedPosts = posts.sort((a, b) => b.data.date - a.data.date);

// src/pages/blog/[...slug].astro - Individual post
---
import { getCollection } from 'astro:content';
import Layout from '../../layouts/Layout.astro';

export async function getStaticPaths() {
  const posts = await getCollection('blog');
  return posts.map((post) => ({
    params: { slug: post.slug },
    props: { post },
  }));
}

const { post } = Astro.props;
const { Content } = await post.render();
---

<Layout title={post.data.title}>
