Svelte Pattern: Components compile to vanilla JS - no runtime overhead. Reactive state uses $: labels.
Svelte Reactive Declarations ($:)
<!-- Svelte Counter Component -->
<script>
  let count = 0;

  // Reactive declaration - auto-updates when count changes
  $: badgeClass = count > 0 ? 'bg-success'
                 : count < 0 ? 'bg-danger'
                 : 'bg-secondary';

  // Reactive statement block
  $: if (count > 10) {
    alert('Count is getting high!');
    count = 0;
  }

  // Reactive derived value
  $: doubled = count * 2;
</script>

<!-- Bootstrap-styled Svelte template -->
<div class="card shadow-sm">
  <div class="card-body text-center">
    <p class="display-4">
      Count: <span class="badge {badgeClass}">{count}</span>
    </p>
    <p>Doubled: {doubled}</p>
    <div class="btn-group">
      <button class="btn btn-outline-danger" on:click={() => count--}>-</button>
      <button class="btn btn-outline-secondary" on:click={() => count = 0}>Reset</button>
      <button class="btn btn-outline-success" on:click={() => count++}>+</button>
    </div>
  </div>
</div>

Svelte Stores (writable, readable, derived)
<!-- stores.js - Shared reactive state -->
import { writable, derived } from 'svelte/store';

// Writable store - can be updated from any component
export const todos = writable([
  { id: 1, text: 'Learn Svelte stores', completed: false },
  { id: 2, text: 'Integrate Bootstrap', completed: true },
]);

// Derived store - computed from other stores
export const completedCount = derived(
  todos,
  $todos => $todos.filter(t => t.completed).length
);

// Readable store - external data source
export const time = readable(new Date(), function start(set) {
  const interval = setInterval(() => set(new Date()), 1000);
  return function stop() { clearInterval(interval); };
});

<!-- TodoList.svelte - Consuming the store -->
<script>
  import { todos, completedCount } from './stores.js';

  let newTodo = '';

  function addTodo() {
    if (!newTodo.trim()) return;
    // $todos auto-subscribes, .update() modifies store value
    todos.update(t => [...t, { id: Date.now(), text: newTodo, completed: false }]);
    newTodo = '';
  }

  function removeTodo(id) {
    todos.update(t => t.filter(todo => todo.id !== id));
  }

  function toggleTodo(id) {
    todos.update(t => t.map(todo =>
      todo.id === id ? { ...todo, completed: !todo.completed } : todo
    ));
  }
</script>

<!-- $todos auto-subscribes in template -->
<ul class="list-group">
  {#each $todos as todo (todo.id)}
    <li class="list-group-item" class:list-group-item-success={todo.completed}>
      <span on:click={() => toggleTodo(todo.id)}>{todo.text}</span>
      <button on:click={() => removeTodo(todo.id)}>Remove</button>
    </li>
  {/each}
</ul>
<p>{$completedCount} of {$todos.length} completed</p>
Svelte Transitions & Animations
<script>
  import { fade, fly, slide, scale } from 'svelte/transition';
  import { flip } from 'svelte/animate';
  import { quintOut } from 'svelte/easing';

  let visible = true;
  let items = [1, 2, 3, 4, 5];
</script>

<!-- fade transition for alerts -->
{#if visible}
  <div class="alert alert-success"
       transition:fade={{ duration: 300 }}>
    This alert fades in and out smoothly
  </div>
{/if}

<!-- fly transition with Bootstrap cards -->
{#if visible}
  <div class="card"
       transition:fly={{ y: 200, duration: 500 }}>
    <div class="card-body">Fly transition</div>
  </div>
{/if}

<!-- slide transition for accordion-like content -->
{#if visible}
  <div transition:slide={{ duration: 300 }}>
    <div class="card card-body">Slide content</div>
  </div>
{/if}

<!-- animate:flip for list reordering -->
{#each items as item (item)}
  <div class="list-group-item"
       animate:flip={{ duration: 250, easing: quintOut }}>
    Item {item}
  </div>
{/each}
This simulates Svelte's transition:slide using Bootstrap's collapse component. Svelte transitions compile to CSS animations at build time.
Svelte Actions (use:action)
<!-- actions.js - Reusable DOM behavior -->
export function tooltip(node, text) {
  // Initialize Bootstrap tooltip on the node
  node.setAttribute('data-bs-toggle', 'tooltip');
  node.setAttribute('data-bs-placement', 'top');
  node.setAttribute('title', text);
  const tip = new bootstrap.Tooltip(node);

  return {
    // Called when parameters change
    update(newText) {
      node.setAttribute('title', newText);
      tip.setContent({ '.tooltip-inner': newText });
    },
    // Called when element is destroyed
    destroy() {
      tip.dispose();
    }
  };
}

export function clickOutside(node, callback) {
  function handleClick(event) {
    if (!node.contains(event.target)) {
      callback(event);
    }
  }
  document.addEventListener('click', handleClick);
  return {
    destroy() {
      document.removeEventListener('click', handleClick);
    }
  };
}

<!-- Using actions in a component -->
<script>
  import { tooltip, clickOutside } from './actions.js';
  let dropdownOpen = false;
</script>

<!-- use:tooltip applies Bootstrap tooltip via action -->
<button class="btn btn-primary" use:tooltip={"Save your changes"}>
  Save
</button>

<!-- use:clickOutside closes dropdown on outside click -->
<div class="dropdown" use:clickOutside={() => dropdownOpen = false}>
  <button class="btn btn-secondary dropdown-toggle"
          on:click={() => dropdownOpen = !dropdownOpen}>
    Menu
  </button>
  <ul class="dropdown-menu" class:show={dropdownOpen}>
    <li><a class="dropdown-item" href="#">Action</a></li>
    <li><a class="dropdown-item" href="#">Another</a></li>
  </ul>
</div>
Svelte Template Syntax ({#if}, {#each}, {#await})
<!-- Conditional rendering with Bootstrap alerts -->
{#if status === 'success'}
  <div class="alert alert-success">Operation successful!</div>
{:else if status === 'error'}
  <div class="alert alert-danger">Something went wrong</div>
{:else}
  <div class="alert alert-info">Ready</div>
{/if}

<!-- List rendering with Bootstrap list-group -->
{#each users as user, index (user.id)}
  <div class="list-group-item">
    {index + 1}. {user.name}
  </div>
{:empty}
  <div class="list-group-item text-muted">No users found</div>
{/each}

<!-- Async data with Bootstrap loading states -->
{#await fetchUsers()}
  <div class="text-center">
    <div class="spinner-border text-primary"></div>
  </div>
{:then users}
  {#each users as user}
    <div class="card mb-2">
      <div class="card-body">{user.name}</div>
    </div>
  {/each}
{:catch error}
  <div class="alert alert-danger">{error.message}</div>
{/await}
