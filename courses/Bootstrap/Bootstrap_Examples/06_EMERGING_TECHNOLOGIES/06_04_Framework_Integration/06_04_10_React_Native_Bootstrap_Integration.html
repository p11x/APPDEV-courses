React Native + Bootstrap

StyleSheet API, Responsive Design, Component Mapping, Cross-Platform

React Native Pattern: No CSS - uses StyleSheet.create() for styles. Bootstrap patterns translate to View/Text styling via flexbox and utility functions.
StyleSheet vs Bootstrap CSS Mapping
// React Native uses StyleSheet.create() instead of CSS
import { StyleSheet, View, Text } from 'react-native';

// Bootstrap: <div class="container p-3">
// React Native equivalent:
const styles = StyleSheet.create({
  container: {
    flex: 1,                    // Full height
    paddingHorizontal: 16,     // px-3 (Bootstrap spacing)
    paddingVertical: 16,       // py-3
  },
  // Bootstrap: <div class="row">
  row: {
    flexDirection: 'row',       // flex-row
    flexWrap: 'wrap',           // flex-wrap
    marginHorizontal: -8,       // Negative margin for gutters
  },
  // Bootstrap: <div class="col-md-6">
  col6: {
    flex: 0,                    // Don't grow
    maxWidth: '50%',            // col-6 width
    paddingHorizontal: 8,      // Gutter padding
  },
  // Bootstrap: <div class="card shadow">
  card: {
    backgroundColor: '#fff',
    borderRadius: 8,
    shadowColor: '#000',        // iOS shadow
    shadowOffset: { width: 0, height: 2 },
    shadowOpacity: 0.1,
    shadowRadius: 4,
    elevation: 3,               // Android shadow
    marginBottom: 16,
  },
  // Bootstrap: <button class="btn btn-primary">
  btnPrimary: {
    backgroundColor: '#0d6efd',
    paddingVertical: 10,
    paddingHorizontal: 20,
    borderRadius: 6,
    alignItems: 'center',
  },
  btnPrimaryText: {
    color: '#fff',
    fontWeight: '600',
    fontSize: 16,
  },
});

// utils/bootstrap.js - Bootstrap utility mapping for React Native

// Spacing scale matches Bootstrap's $spacer (0.25rem increments)
const spacing = {
  0: 0, 1: 4, 2: 8, 3: 12, 4: 16, 5: 24,
  auto: 'auto',
};

// Color palette matching Bootstrap 5
const colors = {
  primary: '#0d6efd',
  secondary: '#6c757d',
  success: '#198754',
  danger: '#dc3545',
  warning: '#ffc107',
  info: '#0dcaf0',
  light: '#f8f9fa',
  dark: '#212529',
  white: '#ffffff',
};

// Generate spacing utility functions (p-3, mx-2, etc.)
const p = (n) => ({ padding: spacing[n] });
const px = (n) => ({ paddingHorizontal: spacing[n] });
const py = (n) => ({ paddingVertical: spacing[n] });
const pt = (n) => ({ paddingTop: spacing[n] });
const m = (n) => ({ margin: spacing[n] });
const mx = (n) => ({ marginHorizontal: spacing[n] });
const my = (n) => ({ marginVertical: spacing[n] });
const mt = (n) => ({ marginTop: spacing[n] });
const mb = (n) => ({ marginBottom: spacing[n] });

// Flexbox utilities (matches Bootstrap flex classes)
const flexRow = { flexDirection: 'row' };
const flexCol = { flexDirection: 'column' };
const justifyCenter = { justifyContent: 'center' };
const justifyBetween = { justifyContent: 'space-between' };

React Native Component Patterns
import { View, Text, TouchableOpacity, ScrollView, FlatList } from 'react-native';
import { p, px, py, m, mb, flexRow, justifyBetween, alignCenter } from '../utils/bootstrap';

// === BOOTSTRAP CARD ===
// Bootstrap: <div class="card"><div class="card-body">...</div></div>
const Card = ({ children, style }) => (
  <View style={[styles.card, p(3), mb(3), style]}>
    {children}
  </View>
);

// === BOOTSTRAP BUTTON ===
// Bootstrap: <button class="btn btn-primary">
const Button = ({ title, variant = 'primary', onPress, size = 'md' }) => {
  const buttonStyles = {
    primary: { backgroundColor: '#0d6efd' },
    success: { backgroundColor: '#198754' },
    danger: { backgroundColor: '#dc3545' },
    outline: { backgroundColor: 'transparent', borderWidth: 1, borderColor: '#0d6efd' },
  };
  const textColors = {
    primary: '#fff', success: '#fff', danger: '#fff', outline: '#0d6efd',
  };

  return (
    <TouchableOpacity
      style={[styles.btnBase, buttonStyles[variant], py(size === 'sm' ? 1 : 2)]}
      onPress={onPress}
    >
      <Text style={[styles.btnText, { color: textColors[variant] }]}>{title}</Text>
    </TouchableOpacity>
  );
};

// === BOOTSTRAP LIST GROUP ===
// Bootstrap: <ul class="list-group"><li class="list-group-item">
const ListGroup = ({ data, renderItem }) => (
  <FlatList
    data={data}
    renderItem={({ item, index }) => (
      <View style={[
        styles.listGroupItem,
        index === 0 && { borderTopLeftRadius: 8, borderTopRightRadius: 8 },
        index === data.length - 1 && { borderBottomLeftRadius: 8, borderBottomRightRadius: 8 },
      ]}>
        {renderItem(item)}
      </View>
    )}
    keyExtractor={(item) => item.id.toString()}
  />
);

// === BOOTSTRAP ALERT ===
// Bootstrap: <div class="alert alert-success">
const Alert = ({ type = 'info', message, onClose }) => {
  const alertColors = {
    success: { bg: '#d1e7dd', text: '#0a3622', border: '#a3cfbb' },
    danger: { bg: '#f8d7da', text: '#58151c', border: '#f1aeb5' },
    warning: { bg: '#fff3cd', text: '#664d03', border: '#ffe69c' },
    info: { bg: '#cff4fc', text: '#055160', border: '#9eeaf9' },
  };
  const colors = alertColors[type];

  return (
    <View style={[styles.alert, { backgroundColor: colors.bg, borderColor: colors.border }]}>
      <Text style={{ color: colors.text }}>{message}</Text>
      {onClose && (
        <TouchableOpacity onPress={onClose} style={styles.closeBtn}>
          <Text style={{ color: colors.text }}>X</Text>
        </TouchableOpacity>
      )}
    </View>
  );
};

// === BOOTSTRAP BADGE ===
// Bootstrap: <span class="badge bg-primary">
const Badge = ({ text, variant = 'primary' }) => (
  <View style={[styles.badge, { backgroundColor: colors[variant] }]}>
    <Text style={styles.badgeText}>{text}</Text>
  </View>
);
Responsive Design (useWindowDimensions)
import { useWindowDimensions, View } from 'react-native';

// Responsive grid that mimics Bootstrap's breakpoint system
function ResponsiveGrid({ children }) {
  const { width } = useWindowDimensions();

  // Bootstrap breakpoints: sm=576, md=768, lg=992, xl=1200
  const getColumns = () => {
    if (width >= 1200) return 4;    // col-xl-3 (4 columns)
    if (width >= 992) return 3;     // col-lg-4 (3 columns)
    if (width >= 768) return 2;     // col-md-6 (2 columns)
    return 1;                       // col-12 (1 column)
  };

  const columns = getColumns();
  const itemWidth = (width - (columns + 1) * 16) / columns;

  return (
    <View style={[flexRow, flexWrap, { justifyContent: 'space-between' }]}>
      {React.Children.map(children, (child) => (
        <View style={{ width: itemWidth, marginBottom: 16 }}>
          {child}
        </View>
      ))}
    </View>
  );
}

// Usage with Bootstrap-like components
function UserGrid({ users }) {
  return (
    <ResponsiveGrid>
      {users.map(user => (
        <Card key={user.id}>
          <Text style={textBold}>{user.name}</Text>
          <Text style={textMuted}>{user.email}</Text>
        </Card>
      ))}
    </ResponsiveGrid>
  );
}

Navigation (React Navigation + Bootstrap Styling)
// Navigation mimics Bootstrap navbar behavior
import { NavigationContainer } from '@react-navigation/native';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import { createNativeStackNavigator } from '@react-navigation/native-stack';

const Stack = createNativeStackNavigator();
const Tab = createBottomTabNavigator();

// Bottom tab navigator (Bootstrap-style bottom navigation)
function MainTabs() {
  return (
    <Tab.Navigator
      screenOptions={{
        tabBarStyle: {
          backgroundColor: '#212529',    // Bootstrap dark
          borderTopColor: '#343a40',
        },
        tabBarActiveTintColor: '#0d6efd', // Bootstrap primary
        tabBarInactiveTintColor: '#6c757d', // Bootstrap secondary
        headerStyle: { backgroundColor: '#212529' },
        headerTintColor: '#fff',
      }}
    >
      <Tab.Screen name="Home" component={HomeScreen} />
      <Tab.Screen name="Dashboard" component={DashboardScreen} />
      <Tab.Screen name="Settings" component={SettingsScreen} />
    </Tab.Navigator>
  );
}

// Stack navigator for drill-down navigation
export default function App() {
  return (
    <NavigationContainer>
      <Stack.Navigator>
        <Stack.Screen name="Main" component={MainTabs} options={{ headerShown: false }} />
        <Stack.Screen
          name="UserDetail"
          component={UserDetailScreen}
          options={{ title: 'User Profile' }}
        />
      </Stack.Navigator>
    </NavigationContainer>
  );
}
