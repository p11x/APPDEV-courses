GraphQL Data Layer with Bootstrap
import { graphql, useStaticQuery } from 'gatsby';

// useStaticQuery - can be used anywhere, no page query needed
// Runs at build time, data is inlined in the bundle
function BlogList() {
  const data = useStaticQuery(graphql`
    query BlogIndexQuery {
      allMarkdownRemark(sort: { frontmatter: { date: DESC } }) {
        nodes {
          id
          frontmatter {
            title
            date(formatString: "MMMM DD, YYYY")
            slug
            excerpt
          }
          excerpt(pruneLength: 160)
        }
      }
    }
  `);

  return (
    <div className="row">
      {data.allMarkdownRemark.nodes.map(post => (
        <div key={post.id} className="col-md-6 col-lg-4 mb-4">
          <div className="card h-100 shadow-sm">
            <div className="card-body">
              <h5 className="card-title">{post.frontmatter.title}</h5>
              <p className="text-muted small">
                {post.frontmatter.date}
              </p>
              <p className="card-text">{post.excerpt}</p>
            </div>
            <div className="card-footer bg-transparent">
              <Link to={`/blog/${post.frontmatter.slug}`}
                    className="btn btn-outline-primary btn-sm">
                Read More
              </Link>
            </div>
          </div>
        </div>
      ))}
    </div>
  );
}

// PAGE QUERY - only works in page components
// Provides data as props to the page component
export const query = graphql`
  query BlogPostBySlug($slug: String!) {
    markdownRemark(frontmatter: { slug: { eq: $slug } }) {
      html
      frontmatter {
        title
        date(formatString: "MMMM DD, YYYY")
        author
      }
    }
  }
`;

export default function BlogPost({ data }) {
  const post = data.markdownRemark;
  return (
    <article className="container py-4">
      <h1 className="mb-3">{post.frontmatter.title}</h1>
      <p className="text-muted">
        By {post.frontmatter.author} | {post.frontmatter.date}
      </p>
      <hr />
      {/* dangerouslySetInnerHTML renders HTML from markdown */}
      <div dangerouslySetInnerHTML={{ __html: post.html }} />
    </article>
  );
}

Programmatic Page Creation (gatsby-node.js)
// gatsby-node.js - Create pages programmatically
const path = require('path');

exports.createPages = async ({ graphql, actions }) => {
  const { createPage } = actions;

  // Query all blog posts
  const result = await graphql(`
    query {
      allMarkdownRemark {
        nodes {
          frontmatter {
            slug
            category
          }
        }
      }
    }
  `);

  // Create individual blog post pages
  const blogPostTemplate = path.resolve('./src/templates/blog-post.js');
  result.data.allMarkdownRemark.nodes.forEach(node => {
    createPage({
      path: `/blog/${node.frontmatter.slug}`,
      component: blogPostTemplate,
      context: {
        slug: node.frontmatter.slug,
      },
    });
  });

  // Create category pages
  const categoryTemplate = path.resolve('./src/templates/category.js');
  const categories = [...new Set(
    result.data.allMarkdownRemark.nodes.map(n => n.frontmatter.category)
  )];
  categories.forEach(category => {
    createPage({
      path: `/category/${category}`,
      component: categoryTemplate,
      context: { category },
    });
  });
};
Gatsby Image + Bootstrap Grid
import { GatsbyImage, getImage } from 'gatsby-plugin-image';

// Query for optimized images
export const query = graphql`
  query {
    allFile(filter: { relativeDirectory: { eq: "gallery" } }) {
      nodes {
        childImageSharp {
          gatsbyImageData(
            width: 400
            placeholder: BLURRED
            formats: [AUTO, WEBP]
          )
        }
        name
      }
    }
  }
`;

// Component with Bootstrap grid and Gatsby images
export default function Gallery({ data }) {
  return (
    <div className="row">
      {data.allFile.nodes.map((image, index) => (
        <div key={index} className="col-md-6 col-lg-4 mb-4">
          <div className="card shadow-sm">
            {/* GatsbyImage replaces img - handles lazy loading, WebP, blur-up */}
            <GatsbyImage
              image={getImage(image)}
              alt={image.name}
              className="card-img-top"
            />
            <div className="card-body">
              <p className="card-text">{image.name}</p>
            </div>
          </div>
        </div>
      ))}
    </div>
  );
}
