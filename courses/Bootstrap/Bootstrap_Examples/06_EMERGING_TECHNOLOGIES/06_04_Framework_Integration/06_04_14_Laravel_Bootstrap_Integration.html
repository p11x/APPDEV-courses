Blade Templates, Livewire, Alpine.js, Vite, Components

Laravel Pattern: Blade templates use @directives for control flow. Livewire provides server-driven reactivity. Alpine.js adds lightweight client interactivity.
Laravel Bootstrap Setup (Vite)
# Install Bootstrap via npm
npm install bootstrap @popperjs/core

# vite.config.js - Configure Vite with Laravel plugin
import { defineConfig } from 'vite';
import laravel from 'laravel-vite-plugin';

export default defineConfig({
    plugins: [
        laravel({
            input: [
                'resources/css/app.css',      // Or app.scss for Sass
                'resources/js/app.js',
            ],
            refresh: true,
        }),
    ],
});

// resources/css/app.scss - Bootstrap import with customization
// Override Bootstrap variables BEFORE the import
$primary: #ff2d20;             // Laravel red
$enable-rounded: true;
$enable-shadows: true;

@import 'bootstrap/scss/bootstrap';

// resources/js/app.js - Bootstrap JS import
import 'bootstrap';

// Alpine.js for lightweight interactivity (ships with Laravel Breeze)
import Alpine from 'alpinejs';
window.Alpine = Alpine;
Alpine.start();

Blade Components (x- syntax)
<!-- resources/views/components/alert.blade.php -->
@props(['type' => 'info', 'dismissible' => false])

<div {{ $attributes->merge(['class' => "alert alert-{$type} {$dismissibleClass}"]) }}
     role="alert">
    {{ $slot }}
    @if($dismissible)
        <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
    @endif
</div>

<!-- resources/views/components/card.blade.php -->
@props(['title' => null, 'headerColor' => 'primary'])

            <h5 class="mb-0">{{ $title }}</h5>
        </div>
    @endif
    <div class="card-body">
        {{ $slot }}
    </div>
    @isset($footer)
        <div class="card-footer bg-transparent">
            {{ $footer }}
        </div>
    @endisset
</div>

Livewire Components + Bootstrap
// app/Http/Livewire/Counter.php
namespace App\Http\Livewire;

use Livewire\Component;

class Counter extends Component
{
    public $count = 0;

    public function increment() { $this->count++; }
    public function decrement() { $this->count--; }
    public function resetCount() { $this->count = 0; }

    public function render()
    {
        return view('livewire.counter');
    }
}

<!-- resources/views/livewire/counter.blade.php -->
<div class="card shadow-sm mb-4">
    <div class="card-header bg-primary text-white">
        <h5 class="mb-0">Livewire Counter (Server-Driven Reactivity)</h5>
    </div>
    <div class="card-body text-center">
        <p class="display-4">
            Count: <span class="badge bg-{{ $count > 0 ? 'success' : ($count < 0 ? 'danger' : 'secondary') }}">
                {{ $count }}
            </span>
        </p>
        <!-- wire:click calls PHP methods without writing JavaScript -->
        <div class="btn-group">
            <button wire:click="decrement" class="btn btn-outline-danger">- Decrement</button>
            <button wire:click="resetCount" class="btn btn-outline-secondary">Reset</button>
            <button wire:click="increment" class="btn btn-outline-success">+ Increment</button>
        </div>
    </div>
</div>
