Composables (Auto-Imported)
// composables/useAuth.ts - Authentication composable
export function useAuth() {
  const user = useState<User | null>('auth_user', () => null);
  const isAuthenticated = computed(() => !!user.value);

  async function login(email: string, password: string) {
    // $fetch is Nuxt's built-in fetch (works on server and client)
    const data = await $fetch<{ user: User; token: string }>('/api/auth/login', {
      method: 'POST',
      body: { email, password },
    });
    user.value = data.user;
    // useCookie is auto-imported
    const token = useCookie('auth_token');
    token.value = data.token;
  }

  async function logout() {
    user.value = null;
    const token = useCookie('auth_token');
    token.value = null;
    navigateTo('/login'); // navigateTo is auto-imported
  }

  return { user, isAuthenticated, login, logout };
}

// composables/useBootstrap.ts - Bootstrap utilities
export function useBootstrap() {
  // useState is SSR-safe reactive state (shared across components)
  const toasts = useState<Toast[]>('toasts', () => []);

  function showToast(message: string, type: string = 'success') {
    const toast = { id: Date.now(), message, type };
    toasts.value.push(toast);
    setTimeout(() => {
      toasts.value = toasts.value.filter(t => t.id !== toast.id);
    }, 5000);
  }

  function showModal(modalId: string) {
    if (import.meta.client) {
      const modal = new (window as any).bootstrap.Modal(
        document.getElementById(modalId)
      );
      modal.show();
    }
  }

  return { toasts, showToast, showModal };
}

// server/api/users/index.ts - REST API endpoint
export default defineEventHandler(async (event) => {
  // getQuery auto-imported for query parameters
  const query = getQuery(event);

  // Direct database access in server route
  const users = await db.user.findMany({
    where: query.search ? { name: { contains: query.search } } : {},
  });

  return users;
});

// server/api/users/[id].ts - Dynamic API route
export default defineEventHandler(async (event) => {
  // getRouterParam auto-imported
  const id = getRouterParam(event, 'id');

  // Method-based handling
  if (event.method === 'GET') {
    return await db.user.findUnique({ where: { id: Number(id) } });
  }

  if (event.method === 'PUT') {
    const body = await readBody(event); // Parse request body
    return await db.user.update({ where: { id: Number(id) }, data: body });
  }

  if (event.method === 'DELETE') {
    return await db.user.delete({ where: { id: Number(id) } });
  }
});

// middleware/auth.ts - Route guard middleware
export default defineNuxtRouteMiddleware((to, from) => {
  const { isAuthenticated } = useAuth();

  if (!isAuthenticated.value && to.path.startsWith('/admin')) {
    return navigateTo('/login', { redirectCode: 302 });
  }
});
