Next.js + Bootstrap 5

App Router, Server Components, SSR, and SSG

Next.js Pattern: App Router uses file-based routing with app/ directory. Server Components render on the client by default.
App Router File Structure
app/
├── layout.tsx          // Root layout (wraps all pages, imports Bootstrap CSS)
├── page.tsx            // Homepage (Server Component by default)
├── globals.css         // Global styles, @import Bootstrap
├── loading.tsx         // Loading UI (Suspense fallback)
├── error.tsx           // Error boundary
├── not-found.tsx       // 404 page
├── dashboard/
│   ├── page.tsx        // /dashboard route
│   ├── loading.tsx     // Dashboard loading state
│   └── settings/
│       └── page.tsx    // /dashboard/settings route
├── components/
│   ├── Navbar.tsx      // Client Component ("use client")
│   ├── CardGrid.tsx    // Server Component (default)
│   └── Modal.tsx       // Client Component
└── api/
    └── users/
        └── route.ts    // API route handler
Root Layout (layout.tsx) - Bootstrap Import
// app/layout.tsx - Root layout wraps ALL pages
// Bootstrap CSS imported once here, available everywhere

import 'bootstrap/dist/css/bootstrap.min.css';
import { Navbar } from '@/components/Navbar';

export const metadata = {
  title: 'My Next.js App',
  description: 'Next.js with Bootstrap 5',

        <footer className="bg-dark text-white text-center py-3 mt-5">
          <p className="mb-0">© 2024 Next.js Bootstrap App</p>

  );
}
Server Components (Default in App Router)
// app/dashboard/page.tsx - Server Component (no "use client" directive)
// Can fetch data directly - no useEffect needed!

async function getUsers() {
  // Direct database or API call - runs on server
  const res = await fetch('https://jsonplaceholder.typicode.com/users', {
    // Next.js extended fetch with caching options
    next: { revalidate: 60 } // ISR: revalidate every 60 seconds
  });
  return res.json();
}

export default async function DashboardPage() {
  // Data fetched at request time (SSR) or build time (SSG)
  const users = await getUsers();

  return (
    <div>
      <h1 className="mb-4">Dashboard</h1>

      {/* Bootstrap grid for user cards */}
      <div className="row">
        {users.map((user: any) => (
          <div key={user.id} className="col-md-4 mb-3">
            <div className="card h-100 shadow-sm">
              <div className="card-body">
                <h5 className="card-title">{user.name}</h5>
                <p className="card-text text-muted">
                  {user.email}<br />
                  {user.company.name}
                </p>
                <button className="btn btn-primary btn-sm">View Profile</button>
              </div>
            </div>
          </div>
        ))}
      </div>

      {/* Server Component can import and render Client Components */}
      <AddUserButton />
    </div>
  );
}

'use client';
// components/Navbar.tsx - Client Component
// Required for: useState, useEffect, event handlers, browser APIs

import { useState } from 'react';
import Link from 'next/link'; // Next.js client-side navigation

export function Navbar() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <nav className="navbar navbar-expand-lg navbar-dark bg-dark">
      <div className="container">
        <Link href="/" className="navbar-brand">
          Next.js App
        </Link>

        {/* Bootstrap responsive toggle */}
        <button
          className="navbar-toggler"
          type="button"
          onClick={() => setIsOpen(!isOpen)}
        >
          <span className="navbar-toggler-icon"></span>
        </button>

        <div className={`collapse navbar-collapse ${isOpen ? 'show' : ''}`}>
          <ul className="navbar-nav me-auto">
            <li className="nav-item">
              <Link href="/" className="nav-link">Home</Link>

}

import { useState, useTransition } from 'react';

export function AddUserButton() {
  const [isPending, startTransition] = useTransition();

  const handleAdd = () => {
    startTransition(async () => {

    });
  };

  return (
    <button
      className="btn btn-success mt-3"
      onClick={handleAdd}
      disabled={isPending}

      ) : 'Add User'}

  );
}
Data Fetching: SSR, SSG, ISR
// STATIC SITE GENERATION (SSG) - builds at build time
// app/blog/page.tsx
export default async function BlogPage() {
  const posts = await fetch('https://api.example.com/posts').then(r => r.json());
  // Default behavior: static generation with caching
  return (
    <div className="row">
      {posts.map(post => (
        <div key={post.id} className="col-md-6 mb-3">
          <div className="card">
            <div className="card-body">
              <h5>{post.title}</h5>
              <p>{post.excerpt}</p>
            </div>
          </div>
        </div>
      ))}
    </div>
  );
}

Server Actions & Forms
// app/actions.ts - Server Actions
'use server';

export async function createUser(formData: FormData) {
  const name = formData.get('name') as string;
  const email = formData.get('email') as string;

  // Direct database access - runs on server only
  await db.user.create({ data: { name, email } });

  // Revalidate cached data
  revalidatePath('/users');
}

// components/UserForm.tsx - Form with Server Action
'use client';
import { useFormStatus } from 'react-dom';
import { createUser } from '@/app/actions';

function SubmitButton() {
  const { pending } = useFormStatus();
  return (
    <button type="submit" className="btn btn-primary" disabled={pending}>
      {pending ? (
        <><span className="spinner-border spinner-border-sm me-2"></span>Creating...</>
      ) : 'Create User'}
    </button>
  );
}

export function UserForm() {
  return (
    <form action={createUser}>
      <div className="mb-3">
        <label className="form-label">Name</label>
        <input name="name" className="form-control" required />
      </div>
      <div className="mb-3">
        <label className="form-label">Email</label>
        <input name="email" type="email" className="form-control" required />
      </div>
      <SubmitButton />
    </form>
  );
}
