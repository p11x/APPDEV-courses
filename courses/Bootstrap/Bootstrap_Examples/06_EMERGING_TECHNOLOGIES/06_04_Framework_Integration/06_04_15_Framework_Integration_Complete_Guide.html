Universal Integration Patterns

<!-- Works with EVERY framework - zero build configuration -->
<!-- CSS in <head> -->
<link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">

<!-- JS before closing </body> -->
<script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js"></script>

// npm install alternative (for build tools):
npm install bootstrap@5.3.3 @popperjs/core

// Import in JS entry point:
import 'bootstrap/dist/css/bootstrap.min.css';
import 'bootstrap';  // JS components

// REACT: className with template literals
<div className={`alert alert-${type} ${isVisible ? 'show' : 'd-none'}`}>

// VUE: :class object/array binding
<div :class="['alert', `alert-${type}`, { 'd-none': !isVisible }]">

// ANGULAR: [ngClass] with expressions
<div [ngClass]="{'alert': true, [`alert-${type}`]: true, 'd-none': !isVisible}">

// SVELTE: class: directive
<div class="alert" class:alert-{type}={true} class:d-none={!isVisible}>

// SOLID: classList helper
<div class="alert" classList={{ [`alert-${type}`]: true, 'd-none': !isVisible() }}>

// BLADE (Laravel): @class directive
<div @class(['alert', "alert-{$type}", 'd-none' => !$isVisible])>

// ERB (Rails): ternary in class string
<div class="alert alert-<%= type %> <%= 'd-none' unless is_visible %>">

// PROGRAMMATIC BOOTSTRAP JS - Works in any JS framework

// Initialize a modal programmatically
const modal = new bootstrap.Modal(document.getElementById('myModal'));
modal.show();
modal.hide();

// Initialize tooltips
document.querySelectorAll('[data-bs-toggle="tooltip"]')
  .forEach(el => new bootstrap.Tooltip(el));

// Initialize popovers
document.querySelectorAll('[data-bs-toggle="popover"]')
  .forEach(el => new bootstrap.Popover(el));

// Listen for Bootstrap events
document.getElementById('myModal').addEventListener('hidden.bs.modal', () => {
  console.log('Modal was hidden');
});

// REACT: useRef for DOM access
const modalRef = React.useRef(null);
React.useEffect(() => {
  const modal = new bootstrap.Modal(modalRef.current);
  return () => modal.dispose(); // Cleanup on unmount
}, []);

// VUE: ref + onMounted
const modalRef = ref(null);
onMounted(() => {
  const modal = new bootstrap.Modal(modalRef.value);
});

// SVELTE: use:action
function bootstrapModal(node) {
  const modal = new bootstrap.Modal(node);
  return { destroy: () => modal.dispose() };
}
// <div use:bootstrapModal>...</div>

<!-- Bootstrap validation classes: is-valid, is-invalid -->
<!-- Works the same across all frameworks -->

<div class="mb-3">
  <label class="form-label">Email</label>
  <input type="email"
         class="form-control {hasError ? 'is-invalid' : 'is-valid'}"
         value="{email}">
  <div class="invalid-feedback">Please enter a valid email.</div>
  <div class="valid-feedback">Looks good!</div>
</div>

// Framework validation libraries that integrate with Bootstrap:
// React: react-hook-form, formik, zod
// Vue: vee-validate, vuelidate, vorms
// Angular: Reactive Forms (built-in), ngx-formly
// Svelte: svelte-forms-lib, superforms
// Laravel: built-in validation with @error directive
// Django: built-in form validation with is-invalid class
// Rails: built-in validations with Bootstrap error classes

/* Override Bootstrap CSS variables (works everywhere) */
:root {
  --bs-primary: #ff6b6b;
  --bs-primary-rgb: 255, 107, 107;
  --bs-font-sans-serif: 'Inter', system-ui, sans-serif;
  --bs-border-radius: 0.5rem;
  --bs-border-radius-sm: 0.25rem;
  --bs-border-radius-lg: 0.75rem;
}

/* Dark mode support */
[data-bs-theme="dark"] {
  --bs-body-bg: #1a1a2e;
  --bs-body-color: #eee;
}

// Toggle theme programmatically (any framework)
document.documentElement.setAttribute('data-bs-theme', 'dark');
document.documentElement.setAttribute('data-bs-theme', 'light');

// SCSS customization (requires build step)
$primary: #ff6b6b;
$theme-colors: (
  "primary": $primary,
  "brand": #6c5ce7,
);
@import 'bootstrap/scss/bootstrap';
