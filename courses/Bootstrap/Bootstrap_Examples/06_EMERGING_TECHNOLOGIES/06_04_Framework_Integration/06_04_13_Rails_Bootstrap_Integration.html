Ruby on Rails + Bootstrap 5

Asset Pipeline, Hotwire, Turbo, Stimulus, Action View Helpers

Rails Pattern: ERB templates with Action View helpers. Hotwire/Turbo provides SPA-like navigation without JavaScript. Stimulus adds interactivity.

Stimulus Controllers + Bootstrap
// app/javascript/controllers/toggle_controller.js
// Stimulus controller for toggling Bootstrap elements
import { Controller } from "@hotwired/stimulus"

export default class extends Controller {
  static targets = ["content"]
  static values = { visible: { type: Boolean, default: true } }

  toggle() {
    this.visibleValue = !this.visibleValue
    this.contentTarget.classList.toggle("d-none")
  }
}

// app/javascript/controllers/modal_controller.js
// Stimulus controller for Bootstrap modals
export default class extends Controller {
  static values = { url: String }

  open() {
    // Fetch content via Turbo and display in Bootstrap modal
    fetch(this.urlValue)
      .then(response => response.text())
      .then(html => {
        document.getElementById("modal-body").innerHTML = html
        const modal = new bootstrap.Modal(document.getElementById("appModal"))
        modal.show()
      })
  }
}

// app/javascript/controllers/toast_controller.js
// Stimulus controller for Bootstrap toasts
export default class extends Controller {
  static values = { message: String, type: { type: String, default: "success" } }

  show() {
    const toastHtml = `
      <div class="toast align-items-center text-bg-${this.typeValue} border-0" role="alert">
        <div class="d-flex">
          <div class="toast-body">${this.messageValue}</div>
          <button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast"></button>
        </div>
      </div>
    `
    const container = document.getElementById("toast-container")
    container.insertAdjacentHTML("beforeend", toastHtml)
    const toastEl = container.lastElementChild
    const toast = new bootstrap.Toast(toastEl)
    toast.show()
  }
}
