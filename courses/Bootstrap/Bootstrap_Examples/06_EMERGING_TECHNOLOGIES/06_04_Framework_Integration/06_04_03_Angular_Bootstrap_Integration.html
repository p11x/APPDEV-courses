Angular Two-Way Binding ([(ngModel)])
<!-- Angular two-way binding with Bootstrap form controls -->
<div class="mb-3">
  <label class="form-label">Username</label>
  <input type="text" class="form-control"
         [(ngModel)]="username"
         [class.is-invalid]="username.length < 3">

Angular Custom Directives for Bootstrap
// Custom directive: tooltip.directive.ts
@Directive({
  selector: '[appTooltip]'
})
export class TooltipDirective implements OnInit {
  @Input() appTooltip = '';  // Tooltip text
  @Input() placement = 'top';

  constructor(private el: ElementRef) {}

  ngOnInit() {
    // Apply Bootstrap tooltip via data attributes
    this.el.nativeElement.setAttribute('data-bs-toggle', 'tooltip');
    this.el.nativeElement.setAttribute('data-bs-placement', this.placement);
    this.el.nativeElement.setAttribute('title', this.appTooltip);
    // Initialize Bootstrap tooltip
    new bootstrap.Tooltip(this.el.nativeElement);
  }
}

// Usage in template:
// <button class="btn btn-primary" appTooltip="Save changes" placement="bottom">
//   Save
// </button>
Angular Services & HTTP Client
// user.service.ts - Injectable service for data management
@Injectable({ providedIn: 'root' })
export class UserService {
  private apiUrl = 'https://jsonplaceholder.typicode.com/users';

  constructor(private http: HttpClient) {}

  // Returns Observable - Angular's async data pattern
  getUsers(): Observable<User[]> {
    return this.http.get<User[]>(this.apiUrl).pipe(
      catchError(this.handleError)
    );
  }
}

// user-list.component.ts - Component consuming the service
@Component({
  template: `
    <div *ngIf="loading" class="text-center">
      <div class="spinner-border text-primary"></div>
    </div>
    <div class="row" *ngIf="!loading">
      <div *ngFor="let user of users$ | async" class="col-md-4 mb-3">
        <div class="card h-100">
          <div class="card-body">
            <h6>{{ user.name }}</h6>
            <p class="text-muted">{{ user.email }}</p>
          </div>
        </div>
      </div>
    </div>
  `
})
export class UserListComponent implements OnInit {
  users$!: Observable<User[]>;
  loading = true;

  constructor(private userService: UserService) {}

  ngOnInit() {
    this.users$ = this.userService.getUsers();
  }
}
