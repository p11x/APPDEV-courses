Signals, Fine-Grained Reactivity, and JSX

SolidJS Pattern: Uses signals for reactive state. Components run once - reactivity is in the template via signals, not re-renders.
SolidJS createSignal
import { createSignal, createMemo } from 'solid-js';

function Counter() {
  // createSignal returns [getter, setter] - not [value, setter]
  // IMPORTANT: call count() to read, not just count
  const [count, setCount] = createSignal(0);

  // createMemo caches derived values (like computed)
  const badgeClass = createMemo(() => {
    if (count() > 0) return 'bg-success';
    if (count() < 0) return 'bg-danger';
    return 'bg-secondary';
  });

  return (
    <div class="card shadow-sm">
      <div class="card-body text-center">
        {/* count() is a function call - triggers reactivity */}
        <p class="display-4">
          Count: <span class={`badge ${badgeClass()}`}>{count()}</span>
        </p>
        <div class="btn-group">
          {/* setCount with function for prev value */}
          <button class="btn btn-outline-danger"
                  onClick={() => setCount(c => c - 1)}>- Decrement</button>
          <button class="btn btn-outline-secondary"
                  onClick={() => setCount(0)}>Reset</button>
          <button class="btn btn-outline-success"
                  onClick={() => setCount(c => c + 1)}>+ Increment</button>
        </div>
      </div>
    </div>
  );
}

SolidJS createEffect & createResource
import { createSignal, createEffect, createResource } from 'solid-js';

// createEffect - runs when any signal inside it changes
function DataLogger() {
  const [query, setQuery] = createSignal('');

  // Effect automatically tracks query() dependency
  createEffect(() => {
    console.log('Query changed:', query());
    // No dependency array needed - Solid tracks automatically!
  });

  // createResource - handles async data fetching
  // Re-fetches when source signal (query) changes
  const [users] = createResource(query, async (q) => {
    if (!q) return [];
    const res = await fetch(
      `https://jsonplaceholder.typicode.com/users`
    );
    return res.json();
  });

  return (
    <div>
      <input
        class="form-control"
        value={query()}
        onInput={(e) => setQuery(e.target.value)}
        placeholder="Search users..."
      />

      {/* Show loading spinner while fetching */}
      {users.loading && (
        <div class="text-center mt-3">
          <div class="spinner-border text-primary"></div>
        </div>
      )}

      {/* Show error if fetch fails */}
      {users.error && (
        <div class="alert alert-danger mt-3">{users.error.message}</div>
      )}

      {/* Render user cards when data loads */}
      <div class="row mt-3">
        {/* Show wraps content in a Suspense boundary */}
        <Show when={!users.loading}>
          <For each={users()}>
            {(user) => (
              <div class="col-md-4 mb-3">
                <div class="card h-100">
                  <div class="card-body">
                    <h6>{user.name}</h6>
                    <p class="text-muted">{user.email}</p>
                  </div>
                </div>
              </div>
            )}
          </For>
        </Show>
      </div>
    </div>
  );
}

SolidJS Context & Stores
import { createContext, useContext } from 'solid-js';
import { createStore, produce } from 'solid-js/store';

// Create context for theme sharing
const ThemeContext = createContext();

// Provider component wraps children with context value
function ThemeProvider(props) {
  const [theme, setTheme] = createStore({
    primary: 'primary',
    dark: false,
  });

  return (
    <ThemeContext.Provider value={[theme, setTheme]}>
      {props.children}
    </ThemeContext.Provider>
  );
}

// Consumer component reads from context
function ThemedCard() {
  const [theme] = useContext(ThemeContext);

  return (
    <div class={`card border-${theme.primary}`}>
      <div class={`card-header bg-${theme.primary} text-white`}>
        Themed Header
      </div>
      <div class="card-body">
        <p>This card uses context-provided theme</p>
      </div>
    </div>
  );
}

// createStore for complex nested state (path-based updates)
function ShoppingCart() {
  const [store, setStore] = createStore({
    items: [
      { id: 1, name: 'Bootstrap Book', price: 29.99, qty: 1 },
      { id: 2, name: 'SolidJS Course', price: 49.99, qty: 2 },
    ],
    discount: 0,
  });

  // produce() enables mutable-style updates on immutable store
  const updateQty = (id, delta) => {
    setStore(
      'items',
      item => item.id === id,
      'qty',
      qty => Math.max(0, qty + delta)
    );
  };

  // Path-based store access for fine-grained reactivity
  return (
    <ul class="list-group">
      <For each={store.items}>
        {(item) => (
          <li class="list-group-item d-flex justify-content-between">
            <span>{item.name} (${item.price})</span>
            <div class="btn-group btn-group-sm">
              <button class="btn btn-outline-secondary"
                      onClick={() => updateQty(item.id, -1)}>-</button>
              <span class="btn btn-outline-secondary disabled">{item.qty}</span>
              <button class="btn btn-outline-secondary"
                      onClick={() => updateQty(item.id, 1)}>+</button>
            </div>
          </li>
        )}
      </For>
    </ul>
  );
}

SolidJS + Bootstrap Component Patterns
// SolidJS Modal component with Bootstrap
function Modal(props) {
  return (
    <Show when={props.show}>
      <div class="modal fade show d-block" tabindex="-1">
        <div class="modal-dialog">
          <div class="modal-content">
            <div class="modal-header">
              <h5 class="modal-title">{props.title}</h5>
              <button class="btn-close" onClick={() => props.onClose()}></button>
            </div>
            <div class="modal-body">
              {props.children}
            </div>
            <div class="modal-footer">
              <button class="btn btn-secondary" onClick={() => props.onClose()}>Close</button>
              <button class="btn btn-primary" onClick={() => props.onConfirm()}>Confirm</button>
            </div>
          </div>
        </div>
      </div>
      <div class="modal-backdrop fade show"></div>
    </Show>
  );
}
