🔍 Lighthouse PWA Audit

Running Lighthouse
Chrome DevTools:
  1. Open Chrome DevTools (F12)
  2. Go to Lighthouse tab
  3. Select Progressive Web App category
  4. Click Analyze page load
CLI:
# Install Lighthouse globally
npm install -g lighthouse

# Run PWA audit
lighthouse https://example.com --only-categories=pwa --output=json

# Run full audit
lighthouse https://example.com --output=html

🔌 Offline Testing

Automated Offline Test Suite
// PWA Offline Testing with Puppeteer
// Tests that the app works correctly when offline

const puppeteer = require('puppeteer');

describe('PWA Offline Tests', () => {
    let browser, page;

    beforeAll(async () => {
        browser = await puppeteer.launch();
        page = await browser.newPage();
    });

    // Test 1: Page loads when online
    test('loads successfully when online', async () => {
        await page.goto('https://example.com');
        const title = await page.title();
        expect(title).toBeTruthy();
    });

    // Test 2: Service worker registers
    test('service worker registers', async () => {
        await page.goto('https://example.com');
        const swRegistered = await page.evaluate(() => {
            return navigator.serviceWorker.getRegistration();
        });
        expect(swRegistered).toBeTruthy();
    });

    // Test 3: Page works offline
    test('works offline', async () => {
        // First visit while online to cache
        await page.goto('https://example.com');
        await page.waitForTimeout(2000); // Wait for caching

        // Go offline
        await page.setOfflineMode(true);

        // Reload and check it still works
        await page.reload();
        const title = await page.title();
        expect(title).toBeTruthy();

        // Reset
        await page.setOfflineMode(false);
    });

    // Test 4: Offline fallback page
    test('shows offline page for uncached routes', async () => {
        await page.setOfflineMode(true);
        await page.goto('https://example.com/uncached-page');
        const content = await page.content();
        expect(content).toContain('offline');
        await page.setOfflineMode(false);
    });

    afterAll(async () => {
        await browser.close();
    });
});

📦 Cache Testing

// Cache testing utilities
class CacheTester {
    // List all caches
    static async listCaches() {
        const names = await caches.keys();
        return names;
    }

    // Count entries in a cache
    static async countEntries(cacheName) {
        const cache = await caches.open(cacheName);
        const keys = await cache.keys();
        return keys.length;
    }

    // Check if a URL is cached
    static async isCached(url) {
        const response = await caches.match(url);
        return response !== undefined;
    }

    // Get cache size estimation
    static async getCacheSize(cacheName) {
        const cache = await caches.open(cacheName);
        const keys = await cache.keys();
        let totalSize = 0;

        for (const request of keys) {
            const response = await cache.match(request);
            const blob = await response.clone().blob();
            totalSize += blob.size;
        }

        return totalSize;
    }

    // Clear a specific cache
    static async clearCache(cacheName) {
        return await caches.delete(cacheName);
    }

    // Clear all caches
    static async clearAllCaches() {
        const names = await caches.keys();
        return Promise.all(names.map(name => caches.delete(name)));
    }
}

// Playwright integration tests
test('PWA installs correctly', async ({ page }) => {
    await page.goto('/');
    const swScope = await page.evaluate(
        () => navigator.serviceWorker
            .getRegistration()
            .then(r => r?.scope)
    );
    expect(swScope).toBeDefined();
});
