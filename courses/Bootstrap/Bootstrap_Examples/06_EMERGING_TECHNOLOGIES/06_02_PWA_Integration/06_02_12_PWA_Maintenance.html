🔄 Update Strategies

Immediate Update

New SW activates immediately. User sees changes instantly.

// sw.js
self.addEventListener('install', e => {
    self.skipWaiting();
});
self.addEventListener('activate', e => {
    e.waitUntil(self.clients.claim());
});
Best for: Critical fixes
Prompt to Update

Notify user of update, let them choose when to refresh.

// Page script
navigator.serviceWorker
    .addEventListener(
        'controllerchange', () => {
            // SW changed, show prompt
            showUpdateToast();
    });
Best for: Most apps
Background Update

Download update silently, apply on next visit.

// sw.js - Don't skip waiting
self.addEventListener('install', e => {
    // New SW waits until old one
    // is no longer controlling pages
});

// Update on next navigation
navigator.serviceWorker
    .register('/sw.js', {
        updateViaCache: 'none'
    });
Best for: Content sites

📦 Cache Invalidation

Cache Versioning & Invalidation
// Cache versioning strategy
// Change CACHE_VERSION to force cache refresh

const CACHE_VERSION = 'v2.3.1';
const CACHE_NAME = `app-${CACHE_VERSION}`;

// Precache with version
const PRECACHE_URLS = [
    { url: '/', revision: CACHE_VERSION },
    { url: '/css/app.css', revision: 'abc123' },
    { url: '/js/app.js', revision: 'def456' }
];

// Install new cache version
self.addEventListener('install', event => {
    event.waitUntil(
        caches.open(CACHE_NAME).then(cache => {
            return cache.addAll(PRECACHE_URLS.map(item => item.url));
        })
    );
});

// Delete old cache versions on activate
self.addEventListener('activate', event => {
    event.waitUntil(
        caches.keys().then(cacheNames => {
            return Promise.all(
                cacheNames
                    .filter(name => name.startsWith('app-') && name !== CACHE_NAME)
                    .map(name => {
                        console.log('Deleting old cache:', name);
                        return caches.delete(name);
                    })
            );
        })
    );
});

// URL-based cache busting
// Add version query parameter to force refresh
function versionedUrl(url) {
    return `${url}?v=${CACHE_VERSION}`;
}
💡 Best Practice: Use content hashes in filenames (e.g., app.abc123.js) for automatic cache busting. When the file content changes, the filename changes, naturally invalidating the cache.

// Version management with update checking
class VersionManager {
    constructor(versionUrl) {
        this.versionUrl = versionUrl;
        this.currentVersion = null;
    }

    // Check for new version from server
    async checkForUpdate() {
        try {
            const response = await fetch(this.versionUrl, {
                cache: 'no-cache' // Always get fresh version
            });
            const { version } = await response.json();

            if (this.currentVersion && version !== this.currentVersion) {
                this.onUpdateAvailable(version);
            }

            this.currentVersion = version;
            return version;
        } catch (error) {
            console.log('Version check failed:', error);
        }
    }

    // Handle update available
    onUpdateAvailable(newVersion) {
        // Show Bootstrap alert
        const alert = document.createElement('div');
        alert.className = 'alert alert-info alert-dismissible fade show position-fixed top-0 start-50 translate-middle-x mt-3';
        alert.style.zIndex = '1060';
        alert.innerHTML = `
            <strong>Update Available!</strong> Version ${newVersion} is ready.
            <button class="btn btn-sm btn-primary ms-2" onclick="location.reload()">Update</button>
            <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
        `;
        document.body.appendChild(alert);
    }

    // Periodic version check
    startPolling(intervalMs = 60000) {
        this.checkForUpdate();
        setInterval(() => this.checkForUpdate(), intervalMs);
    }
}

// Complete lifecycle management
class SWLifecycleManager {
    async init() {
        if (!('serviceWorker' in navigator)) return;

        const registration = await navigator.serviceWorker.register('/sw.js');

        // Track SW state changes
        if (registration.installing) {
            this.trackState(registration.installing, 'installing');
        }
        if (registration.waiting) {
            this.trackState(registration.waiting, 'waiting');
        }
        if (registration.active) {
            this.trackState(registration.active, 'active');
        }

        // Listen for updates
        registration.addEventListener('updatefound', () => {
            const newWorker = registration.installing;
            this.trackState(newWorker, 'update-found');
        });
    }

    trackState(worker, label) {
        worker.addEventListener('statechange', () => {
            console.log(`[SW ${label}] State: ${worker.state}`);
            this.updateUI(worker.state);
        });
    }

    updateUI(state) {
        const el = document.getElementById('swState');
        if (el) el.textContent = state;
    }
}
