Service Workers

The backbone of PWAs. Service workers intercept network requests, manage caches, and enable offline functionality.

Install Phase
// The install event fires when the SW is first registered
// Use it to cache essential assets (app shell)

self.addEventListener('install', (event) => {
    console.log('[SW] Installing...');

    // waitUntil ensures the SW stays in installing state
    // until the cache is fully populated
    event.waitUntil(
        caches.open('app-shell-v1').then(cache => {
            // Cache all essential files
            return cache.addAll([
                '/',
                '/index.html',
                '/css/bootstrap.min.css',
                '/js/app.js',
                '/offline.html'
            ]);
        })
    );

    // skipWaiting forces the new SW to activate immediately
    // instead of waiting for old SW to be terminated
    self.skipWaiting();
});
Activate Phase
// The activate event fires after install completes
// Use it to clean up old caches and claim clients

self.addEventListener('activate', (event) => {
    console.log('[SW] Activating...');

    event.waitUntil(
        // List all cache names
        caches.keys().then(cacheNames => {
            return Promise.all(
                cacheNames.map(cacheName => {
                    // Delete outdated caches
                    if (cacheName !== 'app-shell-v1') {
                        console.log('[SW] Deleting old cache:', cacheName);
                        return caches.delete(cacheName);
                    }
                })
            );
        })
    );

    // Claim all open clients immediately
    // so the new SW controls all pages without reload
    event.waitUntil(self.clients.claim());
});

📝 Service Worker Registration

Complete Registration Script
// Service Worker Registration with full lifecycle handling
// Place this script in your main HTML file

class ServiceWorkerManager {
    constructor(swPath = '/sw.js') {
        this.swPath = swPath;
        this.registration = null;
    }

    // Check if SW is supported in this browser
    isSupported() {
        return 'serviceWorker' in navigator;
    }

    // Register the service worker
    async register() {
        if (!this.isSupported()) {
            console.log('Service Workers not supported');
            return null;
        }

        try {
            this.registration = await navigator.serviceWorker.register(this.swPath, {
                scope: '/' // Control all pages on the domain
            });

            console.log('[SW] Registered:', this.registration.scope);

            // Set up lifecycle event listeners
            this.setupListeners();
            return this.registration;

        } catch (error) {
            console.error('[SW] Registration failed:', error);
            return null;
        }
    }

    // Listen for SW state changes and updates
    setupListeners() {
        // Check for updates periodically (every hour)
        setInterval(() => {
            this.registration.update();
        }, 60 * 60 * 1000);

        // Handle new SW found
        this.registration.addEventListener('updatefound', () => {
            const newWorker = this.registration.installing;
            console.log('[SW] New version found');

            newWorker.addEventListener('statechange', () => {
                if (newWorker.state === 'installed' && navigator.serviceWorker.controller) {
                    this.showUpdateUI();
                }
            });
        });
    }

    // Show Bootstrap update notification
    showUpdateUI() {
        const container = document.createElement('div');
        container.className = 'position-fixed bottom-0 end-0 p-3';
        container.style.zIndex = '1060';
        container.innerHTML = `
            <div class="toast show">
                <div class="toast-header">
                    <strong class="me-auto">Update Available</strong>
                    <button type="button" class="btn-close" data-bs-dismiss="toast"></button>
                </div>
                <div class="toast-body">
                    A new version is ready.
                    <button class="btn btn-primary btn-sm" onclick="location.reload()">Refresh</button>
                </div>
            </div>`;
        document.body.appendChild(container);
    }
}

// Initialize on page load
const swManager = new ServiceWorkerManager('/sw.js');
swManager.register();

📦 Caching in Service Workers

Cache-First

Serve from cache, fallback to network. Best for static assets.

event.respondWith(
    caches.match(request).then(cached => {
        return cached || fetch(request)
            .then(response => {
                const clone = response.clone();
                caches.open('v1').then(c => c.put(request, clone));
                return response;
            });
    })
);
Network-First

Try network, fallback to cache. Best for dynamic content.

event.respondWith(
    fetch(request).then(response => {
        const clone = response.clone();
        caches.open('v1').then(c => c.put(request, clone));
        return response;
    }).catch(() => {
        return caches.match(request);
    })
);
Stale-While-Revalidate

Serve cached, update in background. Best balance.

event.respondWith(
    caches.match(request).then(cached => {
        const fetchPromise = fetch(request)
            .then(response => {
                caches.open('v1').then(c =>
                    c.put(request, response));
                return response.clone();
            });
        return cached || fetchPromise;
    })
);

🔍 Fetch Event Interception

Complete Service Worker (sw.js)
// Complete service worker with install, activate, and fetch handling

const CACHE_NAME = 'pwa-cache-v1';
const OFFLINE_URL = '/offline.html';

// Files to cache on install
const PRECACHE_URLS = [
    '/',
    '/index.html',
    '/offline.html',
    'https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css',
    'https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js'
];

// INSTALL: Cache essential files
self.addEventListener('install', event => {
    event.waitUntil(
        caches.open(CACHE_NAME)
            .then(cache => cache.addAll(PRECACHE_URLS))
            .then(() => self.skipWaiting())
    );
});

// ACTIVATE: Clean up old caches
self.addEventListener('activate', event => {
    event.waitUntil(
        caches.keys().then(keys =>
            Promise.all(keys.filter(k => k !== CACHE_NAME).map(k => caches.delete(k)))
        ).then(() => self.clients.claim())
    );
});

// FETCH: Intercept requests with strategy
self.addEventListener('fetch', event => {
    // Only handle GET requests
    if (event.request.method !== 'GET') return;

    // Navigation requests: network-first with offline fallback
    if (event.request.mode === 'navigate') {
        event.respondWith(
            fetch(event.request).catch(() => caches.match(OFFLINE_URL))
        );
        return;
    }

    // Static assets: cache-first
    if (event.request.url.includes('bootstrap') || event.request.url.match(/\.(css|js|png|jpg|woff2)$/)) {
        event.respondWith(
            caches.match(event.request).then(cached =>
                cached || fetch(event.request).then(response => {
                    const clone = response.clone();
                    caches.open(CACHE_NAME).then(c => c.put(event.request, clone));
                    return response;
                })
            )
        );
        return;
    }

    // API calls: network-first
    if (event.request.url.includes('/api/')) {
        event.respondWith(
            fetch(event.request)
                .then(response => {
                    const clone = response.clone();
                    caches.open(CACHE_NAME).then(c => c.put(event.request, clone));
                    return response;
                })
                .catch(() => caches.match(event.request))
        );
        return;
    }
});
