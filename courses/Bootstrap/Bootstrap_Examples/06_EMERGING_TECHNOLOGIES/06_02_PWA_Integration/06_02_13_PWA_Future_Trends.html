💻 Emerging Web APIs for PWAs

File System Access API
// Open file picker and read file contents
async function openFile() {
    const [fileHandle] = await window.showOpenFilePicker({
        types: [{
            description: 'Text Files',
            accept: { 'text/plain': ['.txt'] }
        }]
    });
    const file = await fileHandle.getFile();
    const contents = await file.text();
    return contents;
}

// Save file to disk
async function saveFile(contents) {
    const handle = await window.showSaveFilePicker({
        suggestedName: 'document.txt'
    });
    const writable = await handle.createWritable();
    await writable.write(contents);
    await writable.close();
}
Web Share Target API
// manifest.json - Register as share target
{
    "share_target": {
        "action": "/share-target",
        "method": "POST",
        "enctype": "multipart/form-data",
        "params": {
            "title": "title",
            "text": "text",
            "url": "url",
            "files": [{
                "name": "files",
                "accept": ["image/*"]
            }]
        }
    }
}

// Handle shared content on server
app.post('/share-target', (req, res) => {
    const { title, text, url } = req.body;
    // Process shared content
    res.redirect('/?shared=true');
});
