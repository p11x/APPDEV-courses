🔍 Intelligent Prefetching

Smart Prefetch with Intersection Observer
// Intelligent link prefetching based on visibility and interaction
class SmartPrefetcher {
    constructor() {
        this.prefetchedUrls = new Set();
        this.observer = null;
    }

    // Prefetch links visible in viewport
    observeLinks() {
        this.observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const link = entry.target;
                    // Prefetch after 200ms of visibility
                    setTimeout(() => {
                        if (!this.prefetchedUrls.has(link.href)) {
                            this.prefetch(link.href);
                        }
                    }, 200);
                }
            });
        }, { rootMargin: '200px' }); // Start 200px before visible

        // Observe all internal links
        document.querySelectorAll('a[href^="/"]').forEach(link => {
            this.observer.observe(link);
        });
    }

    // Prefetch on hover (highest intent signal)
    setupHoverPrefetch() {
        document.addEventListener('mouseover', (e) => {
            const link = e.target.closest('a');
            if (link && link.href.startsWith(location.origin)) {
                this.prefetch(link.href);
            }
        });
    }

    // Create prefetch link element
    prefetch(url) {
        if (this.prefetchedUrls.has(url)) return;
        this.prefetchedUrls.add(url);

        const link = document.createElement('link');
        link.rel = 'prefetch';
        link.href = url;
        document.head.appendChild(link);
    }
}

// Initialize prefetching
const prefetcher = new SmartPrefetcher();
prefetcher.observeLinks();
prefetcher.setupHoverPrefetch();

🎥 Streaming & Progressive Rendering

ReadableStream API
// Stream large responses progressively
async function streamResponse(url) {
    const response = await fetch(url);
    const reader = response.body.getReader();
    const decoder = new TextDecoder();

    let result = '';
    while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        result += decoder.decode(value, { stream: true });
        // Update UI with partial content
        document.getElementById('content').innerHTML = result;
    }
    return result;
}
Progressive Image Loading
// Load images progressively: blur-up technique
function progressiveImage(container, lowRes, highRes) {
    // Load low-res placeholder first
    const img = new Image();
    img.src = lowRes;
    img.className = 'blur-up';
    img.style.filter = 'blur(10px)';
    img.style.transition = 'filter 0.5s';
    container.appendChild(img);

    // Load high-res in background
    const highResImg = new Image();
    highResImg.onload = () => {
        img.src = highRes;
        img.style.filter = 'none';
    };
    highResImg.src = highRes;
}

Web Vitals Measurement
// Measure Core Web Vitals
const perfObserver = new PerformanceObserver((list) => {
    list.getEntries().forEach(entry => {
        console.log(`${entry.name}: ${entry.startTime.toFixed(0)}ms`);
    });
});

// Observe paint timing
perfObserver.observe({ entryTypes: ['paint', 'largest-contentful-paint', 'layout-shift'] });

// First Contentful Paint
new PerformanceObserver((list) => {
    const entries = list.getEntries();
    const fcp = entries.find(e => e.name === 'first-contentful-paint');
    if (fcp) console.log('FCP:', fcp.startTime.toFixed(0), 'ms');
}).observe({ entryTypes: ['paint'] });

// Largest Contentful Paint
new PerformanceObserver((list) => {
    const entries = list.getEntries();
    const lastEntry = entries[entries.length - 1];
    console.log('LCP:', lastEntry.startTime.toFixed(0), 'ms');
}).observe({ entryTypes: ['largest-contentful-paint'] });

// Cumulative Layout Shift
let clsValue = 0;
new PerformanceObserver((list) => {
    list.getEntries().forEach(entry => {
        if (!entry.hadRecentInput) clsValue += entry.value;
    });
    console.log('CLS:', clsValue.toFixed(3));
}).observe({ entryTypes: ['layout-shift'] });
