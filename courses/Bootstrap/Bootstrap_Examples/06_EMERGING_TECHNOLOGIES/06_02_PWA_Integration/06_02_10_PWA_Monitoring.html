PWA Analytics Tracker
// Comprehensive PWA analytics tracking
class PWAAnalytics {
    constructor(measurementId) {
        this.measurementId = measurementId;
        this.sessionStart = Date.now();
    }

    // Track PWA-specific events
    trackInstallStatus() {
        // Detect if running as installed PWA
        const isStandalone = window.matchMedia('(display-mode: standalone)').matches;
        const isIOSStandalone = navigator.standalone === true;

        this.sendEvent('pwa_mode', {
            display_mode: isStandalone || isIOSStandalone ? 'standalone' : 'browser',
            platform: navigator.platform,
            is_installed: isStandalone || isIOSStandalone
        });
    }

    // Track service worker lifecycle events
    trackServiceWorker() {
        if ('serviceWorker' in navigator) {
            navigator.serviceWorker.ready.then(reg => {
                this.sendEvent('sw_ready', {
                    scope: reg.scope,
                    update_via_cache: reg.updateViaCache
                });
            });
        }
    }

    // Track offline/online transitions
    trackNetworkStatus() {
        window.addEventListener('online', () => {
            this.sendEvent('network_status', { status: 'online' });
        });
        window.addEventListener('offline', () => {
            this.sendEvent('network_status', { status: 'offline' });
        });
    }

    // Track cache performance
    trackCacheHit(url, wasHit) {
        this.sendEvent('cache_performance', {
            url: url,
            cache_hit: wasHit,
            timestamp: Date.now()
        });
    }

    // Track installation
    trackInstall() {
        window.addEventListener('appinstalled', () => {
            this.sendEvent('pwa_installed', {
                install_time: Date.now() - this.sessionStart
            });
        });
    }

    // Send event to analytics
    sendEvent(eventName, params) {
        // Google Analytics 4
        if (typeof gtag === 'function') {
            gtag('event', eventName, params);
        }
        console.log('[Analytics]', eventName, params);
    }

    // Initialize all tracking
    init() {
        this.trackInstallStatus();
        this.trackServiceWorker();
        this.trackNetworkStatus();
        this.trackInstall();
    }
}

// Monitor service worker errors
class SWErrorTracker {
    constructor(errorEndpoint) {
        this.endpoint = errorEndpoint;
    }

    init() {
        // Catch unhandled SW errors
        navigator.serviceWorker?.addEventListener('error', (e) => {
            this.reportError({
                type: 'sw_error',
                message: 'Service worker error',
                timestamp: Date.now()
            });
        });

        // Catch fetch failures
        const originalFetch = window.fetch;
        window.fetch = async (...args) => {
            try {
                return await originalFetch(...args);
            } catch (error) {
                this.reportError({
                    type: 'fetch_error',
                    url: args[0],
                    message: error.message,
                    online: navigator.onLine
                });
                throw error;
            }
        };
    }

    async reportError(errorData) {
        try {
            await fetch(this.endpoint, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(errorData)
            });
        } catch {
            // Store locally if offline
            const errors = JSON.parse(localStorage.getItem('pending_errors') || '[]');
            errors.push(errorData);
            localStorage.setItem('pending_errors', JSON.stringify(errors));
        }
    }
}
