Offline-Aware Alert
⚠
You are offline. Changes will sync when reconnected.
✅
Connected. All changes synced.
// Network-aware Bootstrap alert
function updateNetworkAlert() {
    const offline = document.getElementById('offlineAlert');
    const online = document.getElementById('onlineAlert');
    if (navigator.onLine) {
        offline.style.display = 'none';
        online.style.display = 'flex';
    } else {
        offline.style.display = 'flex';
        online.style.display = 'none';
    }
}
window.addEventListener('online', updateNetworkAlert);
window.addEventListener('offline', updateNetworkAlert);

📝 Offline-Ready Forms

Offline Form Submission
// Form that works offline and syncs when back online
class OfflineForm {
    constructor(formElement, submitUrl) {
        this.form = formElement;
        this.submitUrl = submitUrl;
        this.queue = [];
    }

    async handleSubmit(event) {
        event.preventDefault();
        const formData = new FormData(this.form);

        if (navigator.onLine) {
            // Online: submit directly
            await fetch(this.submitUrl, {
                method: 'POST',
                body: formData
            });
            this.showStatus('Submitted successfully!', 'success');
        } else {
            // Offline: queue for sync
            const data = Object.fromEntries(formData);
            data.timestamp = Date.now();
            this.queue.push(data);
            localStorage.setItem('formQueue', JSON.stringify(this.queue));

            // Register background sync if available
            if ('sync' in (await navigator.serviceWorker.ready)) {
                await (await navigator.serviceWorker.ready).sync.register('sync-forms');
            }

            this.showStatus('Saved! Will sync when online.', 'warning');
            this.form.reset();
        }
    }

    showStatus(message, type) {
        const status = document.getElementById('formStatus');
        status.innerHTML = `<span class="badge bg-${type}">${message}</span>`;
    }
}
