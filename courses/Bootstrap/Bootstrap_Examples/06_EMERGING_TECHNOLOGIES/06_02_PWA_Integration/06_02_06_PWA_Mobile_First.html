beforeinstallprompt Event Handler
// Handle PWA install prompt
let deferredPrompt;

// Capture the beforeinstallprompt event
window.addEventListener('beforeinstallprompt', (event) => {
    // Prevent the default mini-infobar
    event.preventDefault();
    // Store the event for later use
    deferredPrompt = event;
    // Show your custom install UI
    showInstallBanner();
});

// Show custom install banner
function showInstallBanner() {
    const banner = document.getElementById('installBanner');
    banner.style.display = 'block';
}

// Trigger install when user clicks button
document.getElementById('installBtn').addEventListener('click', async () => {
    if (!deferredPrompt) return;

    // Show the browser install prompt
    deferredPrompt.prompt();

    // Wait for user response
    const { outcome } = await deferredPrompt.userChoice;
    console.log('Install outcome:', outcome); // 'accepted' or 'dismissed'

    // Clear the deferred prompt
    deferredPrompt = null;
    hideInstallBanner();
});

// Listen for successful installation
window.addEventListener('appinstalled', () => {
    console.log('PWA was installed');
    deferredPrompt = null;
});

👉 Touch Gestures

Touch and swipe here

// Touch gesture detection for PWA
class GestureHandler {
    constructor(element) {
        this.element = element;
        this.startX = 0;
        this.startY = 0;
        this.threshold = 50; // minimum swipe distance
    }

    init() {
        this.element.addEventListener('touchstart', e => this.onStart(e), { passive: true });
        this.element.addEventListener('touchend', e => this.onEnd(e), { passive: true });
        this.element.addEventListener('touchmove', e => this.onMove(e), { passive: true });
    }

    onStart(e) {
        this.startX = e.touches[0].clientX;
        this.startY = e.touches[0].clientY;
    }

    onEnd(e) {
        const endX = e.changedTouches[0].clientX;
        const endY = e.changedTouches[0].clientY;
        const diffX = endX - this.startX;
        const diffY = endY - this.startY;

        if (Math.abs(diffX) > Math.abs(diffY)) {
            if (Math.abs(diffX) > this.threshold) {
                this.onSwipe(diffX > 0 ? 'right' : 'left');
            }
        } else {
            if (Math.abs(diffY) > this.threshold) {
                this.onSwipe(diffY > 0 ? 'down' : 'up');
            }
        }
    }

    onSwipe(direction) {
        console.log('Swipe:', direction);
    }
}

👉 Touch Optimizations

Touch Target Guidelines
  • Minimum touch target: 44x44 CSS pixels
  • Recommended spacing: 8px between targets
  • Use touch-action: manipulation to disable double-tap zoom
  • Add -webkit-tap-highlight-color: transparent for cleaner taps
CSS Touch Optimizations
/* Touch-optimized styles for PWA */

/* Remove tap highlight on mobile */
* {
    -webkit-tap-highlight-color: transparent;
}

/* Ensure minimum touch targets */
.btn, a, button {
    min-height: 44px;
    min-width: 44px;
}

/* Prevent text selection on interactive elements */
button, .btn {
    user-select: none;
    -webkit-user-select: none;
}

/* Smooth scrolling with momentum */
.scroll-container {
    -webkit-overflow-scrolling: touch;
    overflow-y: auto;
}

/* Disable double-tap zoom on buttons */
.no-zoom {
    touch-action: manipulation;
}
