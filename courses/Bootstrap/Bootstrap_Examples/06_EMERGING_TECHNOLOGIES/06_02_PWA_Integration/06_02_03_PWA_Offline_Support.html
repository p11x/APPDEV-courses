Offline Support

Build web applications that work seamlessly without an internet connection. Offline-first patterns, fallback pages, and background synchronization.

🔌 Offline-First Patterns

Cache-Only

Never goes to network. All content must be precached.

event.respondWith(
    caches.match(request)
        .then(r => r || new Response(
            'Not cached',
            { status: 404 }
        ))
);
Network-Only

Always tries network. Fails completely when offline.

event.respondWith(
    fetch(request)
        .catch(() => new Response(
            'Offline',
            { status: 503 }
        ))
);
Offline-First

Cache-first with network fallback and offline page.

event.respondWith(
    caches.match(request)
        .then(cached => cached ||
            fetch(request).catch(() =>
                caches.match('/offline.html')
            )
        )
);

🗃 Offline Fallback Page

offline.html - Fallback Page
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>You're Offline</title>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body>
    <div class="container text-center py-5">
        <div class="display-1 mb-4">📡</div>
        <h1>You're Offline</h1>
        <p class="text-muted">Please check your internet connection and try again.</p>
        <button class="btn btn-primary" onclick="location.reload()">
            Try Again
        </button>
        <hr>
        <h5>Available Offline:</h5>
        <div class="list-group" id="cachedPages">
            <!-- Dynamically populated from cache -->
        </div>
    </div>
    <script>
        // Show cached pages available offline
        caches.keys().then(names => {
            const list = document.getElementById('cachedPages');
            names.forEach(name => {
                caches.open(name).then(cache => {
                    cache.keys().then(keys => {
                        keys.forEach(key => {
                            if (key.url.endsWith('.html')) {
                                const a = document.createElement('a');
                                a.href = key.url;
                                a.className = 'list-group-item list-group-item-action';
                                a.textContent = key.url;
                                list.appendChild(a);
                            }
                        });
                    });
                });
            });
        });
    </script>
</body>
</html>

🔄 Background Sync

Queue failed requests and retry them when the connection is restored.

Service Worker Sync Handler
// Listen for sync events
self.addEventListener('sync', event => {
    if (event.tag === 'sync-forms') {
        event.waitUntil(syncFormData());
    }
});

// Retry failed form submissions
async function syncFormData() {
    const db = await openDB();
    const pending = await db.getAll('pending-sync');

    for (const item of pending) {
        try {
            await fetch(item.url, {
                method: item.method,
                headers: item.headers,
                body: item.body
            });
            // Remove from queue on success
            await db.delete('pending-sync', item.id);
        } catch (error) {
            // Will retry on next sync event
            console.log('Sync failed, will retry');
        }
    }
}
Client-Side Queue
// Queue form data for background sync
async function submitWithSync(formData, url) {
    try {
        // Try to submit immediately
        const response = await fetch(url, {
            method: 'POST',
            body: formData
        });
        return response;
    } catch (error) {
        // If offline, store for later sync
        const db = await openDB();
        await db.add('pending-sync', {
            url: url,
            method: 'POST',
            body: formData,
            timestamp: Date.now()
        });

        // Register background sync
        const reg = await navigator.serviceWorker.ready;
        await reg.sync.register('sync-forms');

        // Show user feedback
        showToast('Saved! Will sync when online.');
    }
}
