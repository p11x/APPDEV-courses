Code Reference — Real-Time API Patterns
WebSocket
// Establish connection
const ws = new WebSocket('wss://example.com/ws');

ws.onopen = () => {
  console.log('Connected');
  ws.send('Hello Server');
};

ws.onmessage = (event) => {
  console.log('Received:', event.data);
};

ws.onclose = () => {
  console.log('Disconnected');
};

ws.onerror = (err) => {
  console.error('Error:', err);
};
Server-Sent Events
// Open event stream
const source = new EventSource('/api/events');

source.addEventListener('message', (e) => {
  console.log('Data:', e.data);
});

source.addEventListener('stock', (e) => {
  const data = JSON.parse(e.data);
  console.log('Stock:', data);
});

// Built-in reconnection
source.onerror = () => {
  // Browser auto-reconnects
  console.log('Reconnecting...');
};
Polling
// Simple polling
let pollTimer = null;

function startPolling(interval) {
  pollTimer = setInterval(async () => {
    try {
      const res = await fetch('/api/data');
      const data = await res.json();
      handleUpdate(data);
    } catch (err) {
      console.error('Poll failed:', err);
    }
  }, interval);
}

function stopPolling() {
  clearInterval(pollTimer);
}
When to Use Each Approach
Use Case Recommended Why
Chat / messaging WebSocket Bidirectional, low latency, persistent connection
Live stock prices SSE One-way stream, auto-reconnect, HTTP-friendly
Notifications SSE Simple, works through proxies, built-in retry
Status dashboards Polling Data doesn't change fast, simple to implement
Collaborative editing WebSocket Requires real-time bidirectional sync
Progress updates SSE / Polling Server pushes updates; polling works too if infrequent
Game / real-time multiplayer WebSocket Lowest latency, binary data support
