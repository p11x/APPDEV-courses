TaskFlow API

v2.1.0

A RESTful API for managing tasks, projects, and team collaboration. Supports CRUD operations, authentication via JWT, and real-time webhooks.

Base URL
https://api.taskflow.io/v2
API Status
Operational

99.9% uptime (30 days)

Authentication

API Key

Pass your API key in the X-API-Key header. Generate keys from your dashboard.

GET /tasks Header: X-API-Key: your_api_key_here
Bearer Token

Obtain a JWT via /auth/login, then include it in the Authorization header.

Authorization: Bearer eyJhbGciOi...
OAuth 2.0

Use the authorization code flow for third-party integrations. Supports Google and GitHub providers.

POST /auth/token grant_type=authorization_code

Code Examples 

# List tasks with pagination
curl -X GET "https://api.taskflow.io/v2/tasks?page=1&limit=10" \
  -H "Authorization: Bearer YOUR_TOKEN_HERE" \
  -H "Content-Type: application/json"

# Create a new task
curl -X POST "https://api.taskflow.io/v2/tasks" \
  -H "Authorization: Bearer YOUR_TOKEN_HERE" \
  -H "Content-Type: application/json" \
  -d '{"title": "Fix bug", "status": "open", "priority": "high"}'
// Using fetch API (browser / Node 18+)
const response = await fetch('https://api.taskflow.io/v2/tasks?page=1&limit=10', {
    method: 'GET',
    headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
    }
});

const data = await response.json();
console.log(data);

// Create a new task
const newTask = await fetch('https://api.taskflow.io/v2/tasks', {
    method: 'POST',
    headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
    },
    body: JSON.stringify({
        title: 'Fix bug',
        status: 'open',
        priority: 'high'
    })
});
# Using requests library
import requests

headers = {
    "Authorization": f"Bearer {token}",
    "Content-Type": "application/json"
}

# List tasks
response = requests.get(
    "https://api.taskflow.io/v2/tasks",
    headers=headers,
    params={"page": 1, "limit": 10}
)
data = response.json()
print(data)

# Create a new task
response = requests.post(
    "https://api.taskflow.io/v2/tasks",
    headers=headers,
    json={
        "title": "Fix bug",
        "status": "open",
        "priority": "high"
    }
)
// Using Java 11+ HttpClient
import java.net.http.*;
import java.net.URI;

HttpClient client = HttpClient.newHttpClient();

// List tasks
HttpRequest request = HttpRequest.newBuilder()
    .uri(URI.create("https://api.taskflow.io/v2/tasks?page=1&limit=10"))
    .header("Authorization", "Bearer " + token)
    .header("Content-Type", "application/json")
    .GET()
    .build();

HttpResponse<String> response = client.send(
    request, HttpResponse.BodyHandlers.ofString()
);
System.out.println(response.body());
