// AI-powered browser detection and capability analysis
// Parses user agent and tests feature support

class BrowserDetector {
    constructor() {
        this.userAgent = navigator.userAgent;
        this.capabilities = {};
    }

    // Parse user agent string to identify browser
    detectBrowser() {
        const ua = this.userAgent;
        if (ua.includes('Firefox')) return { name: 'Firefox', engine: 'Gecko' };
        if (ua.includes('Edg')) return { name: 'Edge', engine: 'Blink' };
        if (ua.includes('Chrome')) return { name: 'Chrome', engine: 'Blink' };
        if (ua.includes('Safari')) return { name: 'Safari', engine: 'WebKit' };
        return { name: 'Unknown', engine: 'Unknown' };
    }

    // Test feature support using feature detection
    testCapabilities() {
        return {
            cssVariables: CSS.supports('color', 'var(--test)'),
            cssGrid: CSS.supports('display', 'grid'),
            intersectionObserver: 'IntersectionObserver' in window,
            webAnimations: 'animate' in Element.prototype,
            esModules: 'noModule' in HTMLScriptElement.prototype,
            serviceWorker: 'serviceWorker' in navigator,
            webp: this.supportsImageFormat('webp'),
            avif: this.supportsImageFormat('avif')
        };
    }

    // Determine required polyfills based on missing features
    getRequiredPolyfills(capabilities) {
        const polyfills = [];
        if (!capabilities.intersectionObserver) polyfills.push('intersection-observer');
        if (!capabilities.cssVariables) polyfills.push('css-vars-ponyfill');
        if (!capabilities.webAnimations) polyfills.push('web-animations-js');
        return polyfills;
    }

    async supportsImageFormat(format) {
        const img = new Image();
        return new Promise(resolve => {
            img.onload = () => resolve(true);
            img.onerror = () => resolve(false);
            img.src = `data:image/${format};base64,AAAA`;
        });
    }
}

// Conditional polyfill loading based on browser capabilities
// Only loads what the current browser actually needs

class SmartPolyfillLoader {
    constructor() {
        // Map of features to their polyfill CDN URLs
        this.polyfillMap = {
            'intersection-observer': 'https://polyfill.io/v3/polyfill.min.js?features=IntersectionObserver',
            'resize-observer': 'https://polyfill.io/v3/polyfill.min.js?features=ResizeObserver',
            'fetch': 'https://polyfill.io/v3/polyfill.min.js?features=fetch',
            'css-vars': 'https://cdn.jsdelivr.net/npm/css-vars-ponyfill@2',
            'web-animations': 'https://cdn.jsdelivr.net/npm/web-animations-js@2',
            'structured-clone': 'https://cdn.jsdelivr.net/npm/@ungap/structured-clone'
        };
    }

    // Detect which features are missing
    detectMissingFeatures() {
        const missing = [];
        if (!('IntersectionObserver' in window)) missing.push('intersection-observer');
        if (!('ResizeObserver' in window)) missing.push('resize-observer');
        if (!window.CSS || !CSS.supports('color', 'var(--test)')) missing.push('css-vars');
        if (!('fetch' in window)) missing.push('fetch');
        if (!('structuredClone' in window)) missing.push('structured-clone');
        return missing;
    }

    // Load only the required polyfills dynamically
    async loadRequired() {
        const missing = this.detectMissingFeatures();
        const loadPromises = missing.map(feature => {
            return new Promise((resolve, reject) => {
                const script = document.createElement('script');
                script.src = this.polyfillMap[feature];
                script.onload = resolve;
                script.onerror = reject;
                document.head.appendChild(script);
            });
        });
        await Promise.all(loadPromises);
        return missing;
    }
}

// Auto-initialize on page load
const polyfillLoader = new SmartPolyfillLoader();
polyfillLoader.loadRequired().then(loaded => {
    if (loaded.length > 0) {
        console.log('Polyfills loaded:', loaded);
    }
});

// AI-generated feature fallback
// If modern API unavailable, use older approach

function smoothScroll(target) {
    const element = document.querySelector(target);

    // Modern: native smooth scroll
    if ('scrollBehavior' in document.documentElement.style) {
        element.scrollIntoView({ behavior: 'smooth' });
    }
    // Fallback: manual animation
    else {
        const start = window.scrollY;
        const end = element.offsetTop;
        const duration = 500;
        let startTime = null;

        function animate(time) {
            if (!startTime) startTime = time;
            const progress = Math.min((time - startTime) / duration, 1);
            window.scrollTo(0, start + (end - start) * progress);
            if (progress < 1) requestAnimationFrame(animate);
        }
        requestAnimationFrame(animate);
    }
}
