🔧 API Setup & Configuration

AI API Client Setup
// Complete AI API client configuration
// Handles authentication, retries, caching, and error handling

class AIApiClient {
    constructor(config) {
        this.baseURL = config.baseURL;
        this.apiKey = config.apiKey;
        this.timeout = config.timeout || 30000;
        this.retryAttempts = config.retryAttempts || 3;
        this.cache = new Map();
    }

    // Add authentication and standard headers
    getHeaders() {
        return {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${this.apiKey}`,
            'X-Request-ID': crypto.randomUUID(),
            'X-Client-Version': '1.0.0'
        };
    }

    // Generic request with retry logic
    async request(endpoint, options = {}) {
        const cacheKey = `${endpoint}:${JSON.stringify(options.body)}`;

        // Check cache first
        if (this.cache.has(cacheKey)) {
            return this.cache.get(cacheKey);
        }

        // Retry loop with exponential backoff
        for (let attempt = 1; attempt <= this.retryAttempts; attempt++) {
            try {
                const controller = new AbortController();
                const timeoutId = setTimeout(() => controller.abort(), this.timeout);

                const response = await fetch(`${this.baseURL}${endpoint}`, {
                    method: options.method || 'POST',
                    headers: this.getHeaders(),
                    body: JSON.stringify(options.body),
                    signal: controller.signal
                });

                clearTimeout(timeoutId);

                if (!response.ok) throw new Error(`API Error: ${response.status}`);

                const data = await response.json();
                this.cache.set(cacheKey, data);
                return data;

            } catch (error) {
                if (attempt === this.retryAttempts) throw error;
                // Exponential backoff: 1s, 2s, 4s
                await new Promise(r => setTimeout(r, Math.pow(2, attempt) * 1000));
            }
        }
    }

    // Specific AI endpoint methods
    async analyzeText(text) {
        return this.request('/analyze/text', { body: { text } });
    }

    async analyzeImage(imageUrl) {
        return this.request('/analyze/image', { body: { image_url: imageUrl } });
    }

    async getRecommendations(userId, context) {
        return this.request('/recommend', { body: { user_id: userId, context } });
    }
}

// Initialize client
const aiClient = new AIApiClient({
    baseURL: 'https://api.example.com/v1/ai',
    apiKey: 'your-api-key-here',
    timeout: 15000,
    retryAttempts: 3
});

⚠ Error Handling & Fallbacks

// Robust error handling for AI features
// Provides graceful degradation when AI services are unavailable

class AIErrorHandler {
    // Error types and their fallback strategies
    static strategies = {
        'timeout': { fallback: 'cache', message: 'Taking longer than expected...' },
        'rate_limit': { fallback: 'queue', message: 'Please wait a moment...' },
        'auth_error': { fallback: 'static', message: 'Service temporarily unavailable' },
        'server_error': { fallback: 'cache', message: 'Using cached results...' },
        'network_error': { fallback: 'offline', message: 'Check your connection...' }
    };

    // Handle AI API errors with appropriate fallback
    static async handle(error, context) {
        const errorType = this.classifyError(error);
        const strategy = this.strategies[errorType];

        // Show user-friendly message
        this.showNotification(strategy.message, 'warning');

        // Apply fallback strategy
        switch (strategy.fallback) {
            case 'cache':
                return this.useCachedResult(context);
            case 'queue':
                return this.queueForLater(context);
            case 'static':
                return this.useStaticFallback(context);
            case 'offline':
                return this.useOfflineMode(context);
        }
    }

    // Classify error into known types
    static classifyError(error) {
        if (error.name === 'AbortError') return 'timeout';
        if (error.status === 429) return 'rate_limit';
        if (error.status === 401 || error.status === 403) return 'auth_error';
        if (error.status >= 500) return 'server_error';
        return 'network_error';
    }

    // Show Bootstrap toast notification
    static showNotification(message, type) {
        const toast = document.createElement('div');
        toast.className = `toast align-items-center text-bg-${type} border-0 show`;
        toast.setAttribute('role', 'alert');
        toast.innerHTML = `
            <div class="d-flex">
                <div class="toast-body">${message}</div>
                <button type="button" class="btn-close me-2 m-auto" data-bs-dismiss="toast"></button>
            </div>`;
        document.body.appendChild(toast);
        setTimeout(() => toast.remove(), 5000);
    }
}

🔒 Security Considerations

🔐 API Key Security
  • Never expose API keys in client code
  • Use environment variables
  • Implement server-side proxy
  • Rotate keys regularly
📝 Input Validation
  • Sanitize all user inputs
  • Limit input length
  • Validate file uploads
  • Rate limit requests
👤 Data Privacy
  • Encrypt data in transit (TLS)
  • Minimize data collection
  • Implement data retention policies
  • GDPR/CCPA compliance
