Chatbot Integration Pattern
// AI Chatbot integration pattern with Bootstrap 5
// Connects to AI API and renders conversation in Bootstrap UI

class AIChatbot {
    constructor(containerId, apiEndpoint) {
        this.container = document.getElementById(containerId);
        this.apiEndpoint = apiEndpoint;
        this.conversationHistory = [];
        this.isTyping = false;
    }

    // Add a message to the chat window
    addMessage(content, sender = 'bot') {
        const bubble = document.createElement('div');
        bubble.className = `chat-bubble chat-${sender}`;
        bubble.textContent = content;
        bubble.setAttribute('role', 'article');
        this.container.appendChild(bubble);
        this.container.scrollTop = this.container.scrollHeight;

        // Store in conversation history
        this.conversationHistory.push({ role: sender, content });
    }

    // Show typing indicator while AI processes
    showTyping() {
        this.isTyping = true;
        const indicator = document.createElement('div');
        indicator.className = 'chat-bubble chat-bot typing-indicator';
        indicator.id = 'typingIndicator';
        indicator.innerHTML = '';
        this.container.appendChild(indicator);
        this.container.scrollTop = this.container.scrollHeight;
    }

    // Remove typing indicator
    hideTyping() {
        this.isTyping = false;
        const indicator = document.getElementById('typingIndicator');
        if (indicator) indicator.remove();
    }

    // Send user message to AI API
    async sendMessage(userMessage) {
        this.addMessage(userMessage, 'user');
        this.showTyping();

        try {
            const response = await fetch(this.apiEndpoint, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    messages: this.conversationHistory,
                    max_tokens: 150
                })
            });
            const data = await response.json();
            this.hideTyping();
            this.addMessage(data.reply, 'bot');
        } catch (error) {
            this.hideTyping();
            this.addMessage('Sorry, I encountered an error. Please try again.', 'bot');
        }
    }
}

Recommendation Engine Pattern
// AI Recommendation Engine with Bootstrap card rendering
// Fetches personalized recommendations and renders as cards

class RecommendationEngine {
    constructor(apiEndpoint) {
        this.apiEndpoint = apiEndpoint;
        // User behavior tracking for personalization
        this.userProfile = {
            viewedProducts: [],
            purchasedItems: [],
            searchHistory: [],
            categoryPreferences: {}
        };
    }

    // Track user product views
    trackView(productId, category) {
        this.userProfile.viewedProducts.push({
            id: productId,
            category,
            timestamp: Date.now()
        });
        // Update category preference score
        this.userProfile.categoryPreferences[category] =
            (this.userProfile.categoryPreferences[category] || 0) + 1;
    }

    // Fetch AI-powered recommendations
    async getRecommendations(type = 'personalized', limit = 4) {
        const response = await fetch(this.apiEndpoint, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                user_profile: this.userProfile,
                recommendation_type: type,
                limit: limit
            })
        });
        return response.json();
    }

    // Render recommendations as Bootstrap cards
    renderCards(containerId, recommendations) {
        const container = document.getElementById(containerId);
        container.innerHTML = '';

        recommendations.forEach(item => {
            const col = document.createElement('div');
            col.className = 'col-md-3';
            col.innerHTML = `
                <div class="card h-100">
                    <img src="${item.image}" class="card-img-top" alt="${item.name}">
                    <div class="card-body">
                        <span class="badge bg-info">${item.matchScore}% Match</span>
                        <h6>${item.name}</h6>
                        <p class="small text-muted">${item.reason}</p>
                        <span class="fw-bold">$${item.price}</span>
                    </div>
                </div>`;
            container.appendChild(col);
        });
    }
}
