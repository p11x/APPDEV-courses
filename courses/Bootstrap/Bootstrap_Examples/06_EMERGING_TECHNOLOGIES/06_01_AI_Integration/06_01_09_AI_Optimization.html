AI Smart Lazy Loading Implementation 
// AI-powered lazy loading with Intersection Observer
// This script uses machine learning predictions to optimize image loading

// Step 1: Detect user's connection speed using Navigator API
const connection = navigator.connection || navigator.mozConnection || navigator.webkitConnection;
const effectiveType = connection ? connection.effectiveType : '4g'; // 2g, 3g, 4g

// Step 2: Calculate optimal root margin based on scroll speed
// Faster scrolling = larger margin = earlier preloading
let scrollSpeed = 0;
let lastScrollY = window.scrollY;

window.addEventListener('scroll', () => {
    scrollSpeed = Math.abs(window.scrollY - lastScrollY);
    lastScrollY = window.scrollY;
});

// Step 3: Create Intersection Observer with AI-calculated threshold
const lazyObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            const img = entry.target;
            // Load the image from data attribute
            img.src = img.dataset.lazySrc;
            img.classList.remove('lazy-placeholder');
            lazyObserver.unobserve(img);
        }
    });
}, {
    // Dynamic root margin based on connection speed
    rootMargin: effectiveType === '4g' ? '200px' : '50px',
    threshold: 0.1
});

// Step 4: Observe all lazy-load elements
document.querySelectorAll('[data-lazy-src]').forEach(el => {
    lazyObserver.observe(el);
});

// Predictive prefetching based on user behavior analysis
// Tracks hover time, click patterns, and navigation sequences

class PredictivePrefetcher {
    constructor() {
        // Store navigation history for pattern detection
        this.navigationHistory = [];
        // Map of page transitions with frequency counts
        this.transitionMap = new Map();
        // Minimum hover time (ms) before triggering prefetch
        this.hoverThreshold = 100;
    }

    // Track which page the user navigated from/to
    recordNavigation(fromPage, toPage) {
        const key = `${fromPage} -> ${toPage}`;
        this.transitionMap.set(key, (this.transitionMap.get(key) || 0) + 1);
    }

    // Predict next page based on current page and history
    predictNextPage(currentPage) {
        let maxCount = 0;
        let predicted = null;
        // Find the most common transition from current page
        this.transitionMap.forEach((count, key) => {
            if (key.startsWith(currentPage) && count > maxCount) {
                maxCount = count;
                predicted = key.split(' -> ')[1];
            }
        });
        return predicted;
    }

    // Prefetch a page by creating a hidden link element
    prefetch(url) {
        const link = document.createElement('link');
        link.rel = 'prefetch';
        link.href = url;
        document.head.appendChild(link);
    }
}

// Initialize the prefetcher
const prefetcher = new PredictivePrefetcher();
