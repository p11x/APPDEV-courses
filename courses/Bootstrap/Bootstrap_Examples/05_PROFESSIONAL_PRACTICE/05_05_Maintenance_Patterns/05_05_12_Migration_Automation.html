Automate your Bootstrap migration with scripts, CI/CD pipelines, and tools.

Automation Scripts

#!/bin/bash
# migrate-classes.sh - Replace deprecated Bootstrap classes

echo "Starting class migration..."

# Badge classes
find ./src -name "*.html" -exec sed -i 's/badge-pill/rounded-pill/g' {} \;
find ./src -name "*.html" -exec sed -i 's/badge-primary/bg-primary/g' {} \;
find ./src -name "*.html" -exec sed -i 's/badge-secondary/bg-secondary/g' {} \;
find ./src -name "*.html" -exec sed -i 's/badge-success/bg-success/g' {} \;
find ./src -name "*.html" -exec sed -i 's/badge-danger/bg-danger/g' {} \;
find ./src -name "*.html" -exec sed -i 's/badge-warning/bg-warning/g' {} \;
find ./src -name "*.html" -exec sed -i 's/badge-info/bg-info/g' {} \;

# Form classes
find ./src -name "*.html" -exec sed -i 's/form-group/mb-3/g' {} \;
find ./src -name "*.html" -exec sed -i 's/form-row row/row g-3/g' {} \;

# Margin/padding classes
find ./src -name "*.html" -exec sed -i 's/ml-/ms-/g' {} \;
find ./src -name "*.html" -exec sed -i 's/mr-/me-/g' {} \;
find ./src -name "*.html" -exec sed -i 's/pl-/ps-/g' {} \;
find ./src -name "*.html" -exec sed -i 's/pr-/pe-/g' {} \;

# Data attributes
find ./src -name "*.html" -exec sed -i 's/data-toggle/data-bs-toggle/g' {} \;
find ./src -name "*.html" -exec sed -i 's/data-dismiss/data-bs-dismiss/g' {} \;
find ./src -name "*.html" -exec sed -i 's/data-target/data-bs-target/g' {} \;
find ./src -name "*.html" -exec sed -i 's/data-content/data-bs-content/g' {} \;

echo "Migration complete! Review changes with git diff."

#!/bin/bash
# migrate-css-vars.sh - Update CSS custom properties

echo "Migrating CSS variables..."

# Replace old color variables
find ./src -name "*.scss" -exec sed -i 's/\$theme-colors/-bs-/g' {} \;
find ./src -name "*.css" -exec sed -i 's/--primary/--bs-primary/g' {} \;
find ./src -name "*.css" -exec sed -i 's/--secondary/--bs-secondary/g' {} \;

# Update spacing scale
find ./src -name "*.scss" -exec sed -i 's/\$spacer/1rem/g' {} \;

echo "CSS variable migration complete!"

#!/bin/bash
# migrate-js.sh - Update JavaScript API calls

echo "Migrating JavaScript..."

# Update modal triggers
find ./src -name "*.js" -exec sed -i "s/\$('#modal').modal('show')/new bootstrap.Modal(document.getElementById('modal')).show()/g" {} \;
find ./src -name "*.js" -exec sed -i "s/\$('#modal').modal('hide')/new bootstrap.Modal(document.getElementById('modal')).hide()/g" {} \;

# Update tooltip initialization
find ./src -name "*.js" -exec sed -i "s/\$('[data-toggle=\"tooltip\"]').tooltip()/new bootstrap.Tooltip(document.querySelector('[data-bs-toggle=\"tooltip\"]'))/g" {} \;

echo "JavaScript migration complete!"

CI/CD Pipeline Integration

GitHub Actions
name: Bootstrap Migration Check
on: [push, pull_request]

jobs:
  check:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
      - run: npm ci
      - run: npm run lint:bootstrap
      - run: npm run test:migration
GitLab CI
stages:
  - lint
  - test

bootstrap-check:
  stage: lint
  image: node:20
  script:
    - npm ci
    - npm run lint:bootstrap
    - npm run test:migration
  only:
    - merge_requests

Migration Configuration

bootstrap-migrate.config.js
module.exports = {
  from: '4.6.2',
  to: '5.3.3',
  paths: {
    html: './src/**/*.html',
    css: './src/**/*.css',
    scss: './src/**/*.scss',
    js: './src/**/*.js'
  },
  rules: {
    classes: true,
    attributes: true,
    variables: true,
    jsApi: true
  },
  ignore: [
    'node_modules',
    'dist',
    'vendor'
  ],
  dryRun: false,
  verbose: true
};

Package.json Scripts

{
  "scripts": {
    "migrate": "bootstrap-migrate --config bootstrap-migrate.config.js",
    "migrate:dry": "bootstrap-migrate --dry-run",
    "migrate:scan": "bootstrap-migrate --scan",
    "lint:bootstrap": "eslint --rule 'bootstrap/deprecated: error' src/",
    "test:migration": "jest --config jest.migration.config.js"
  }
}
