Tree Shaking Bootstrap JS

Import only the Bootstrap JavaScript components you actually use.

Before: Full Import (46 KB gzipped)
// Imports ALL Bootstrap JS components
import 'bootstrap';

// Or importing all from bundle
import * as bootstrap from 'bootstrap';
After: Selective Import (~12 KB gzipped)
// Import only needed components
import { Modal } from 'bootstrap';
import { Dropdown } from 'bootstrap';
import { Tooltip } from 'bootstrap';
import { Collapse } from 'bootstrap';

// Or from individual modules
import Modal from
  'bootstrap/js/dist/modal';
import Dropdown from
  'bootstrap/js/dist/dropdown';

Bundler Configuration

Vite Configuration
// vite.config.js
import { defineConfig } from 'vite';

export default defineConfig({
  build: {
    minify: 'terser',
    cssMinify: true,
    rollupOptions: {
      output: {
        manualChunks: {
          'bootstrap': [
            'bootstrap'
          ],
          'vendor': [
            '@popperjs/core'
          ]
        }
      }
    },
    target: 'es2020',
    reportCompressedSize: true
  }
});
Webpack Configuration
// webpack.config.js
const CssMinimizerPlugin =
  require('css-minimizer-webpack-plugin');
const TerserPlugin =
  require('terser-webpack-plugin');

module.exports = {
  mode: 'production',
  optimization: {
    minimize: true,
    minimizer: [
      new TerserPlugin({
        terserOptions: {
          compress: {
            drop_console: true
          }
        }
      }),
      new CssMinimizerPlugin()
    ],
    splitChunks: {
      chunks: 'all'
    }
  }
};
