Report Formats & Configuration

Jest HTML Reporter
// jest.config.js - HTML report config
module.exports = {
  reporters: [
    "default",
    ["jest-html-reporters", {
      publicPath: "./reports",
      filename: "test-report.html",
      expand: true,
      pageTitle: "Test Report",
      logoImgPath: "./logo.png",
      hideIcon: false,
      customInfos: [
        { title: "Environment",
          value: "CI" }
      ]
    }]
  ]
};
Coverage Report Config
// Coverage reporters configuration
module.exports = {
  coverageReporters: [
    "text",        // Terminal output
    "text-summary", // Summary in terminal
    "html",        // Interactive HTML report
    "json",        // JSON for tooling
    "json-summary",// Lightweight JSON
    "lcov",        // For SonarQube/Codecov
    "cobertura"    // Azure DevOps format
  ],
  coverageDirectory: "coverage",
  coverageThreshold: {
    global: {
      branches: 80,
      functions: 80,
      lines: 80,
      statements: 80
    }
  }
};
Format Use Case Tool Integration Output
HTML Human-readable visual reports Browser, CI artifacts report.html
JSON Programmatic consumption Custom dashboards, APIs results.json
JUnit XML CI/CD integration Jenkins, GitHub Actions junit.xml
LCOV Coverage visualization SonarQube, Codecov lcov.info
Markdown Documentation, PR comments GitHub, GitLab report.md

Quality Gates Configuration
// sonar-project.properties
sonar.projectKey=my-bootstrap-app
sonar.sources=src/
sonar.tests=tests/
sonar.javascript.lcov.reportPaths=coverage/lcov.info
# Quality gate conditions
sonar.qualitygate.wait=true
sonar.coverage.exclusions=**/*.test.js,**/*.spec.js
# Duplication thresholds
sonar.cpd.exclusions=**/*.spec.js
sonar.coverage.minimum=80
