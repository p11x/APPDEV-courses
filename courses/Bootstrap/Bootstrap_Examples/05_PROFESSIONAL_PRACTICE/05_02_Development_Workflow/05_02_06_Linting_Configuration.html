1. ESLint Configuration

Installation
# Install ESLint (flat config - v9+) npm install --save-dev eslint @eslint/js # For Bootstrap-specific rules npm install --save-dev eslint-plugin-html # Initialize configuration npx eslint --init
eslint.config.js (Flat Config)
// eslint.config.js - ESLint v9+ flat config import js from '@eslint/js'; export default [ js.configs.recommended, { files: ['**/*.js'], languageOptions: { ecmaVersion: 'latest', sourceType: 'module', globals: { // Bootstrap's global dependencies document: 'readonly', window: 'readonly', console: 'readonly', bootstrap: 'readonly', }, }, rules: { // Error prevention 'no-unused-vars': ['warn', { argsIgnorePattern: '^_' }], 'no-console': 'warn', 'no-undef': 'error', // Code quality 'eqeqeq': ['error', 'always'], 'no-var': 'error', 'prefer-const': 'error', 'prefer-arrow-callback': 'error', // Style (let Prettier handle most formatting) 'semi': ['error', 'always'], 'quotes': ['error', 'single'], }, }, { // Ignore patterns ignores: ['dist/**', 'node_modules/**', '**/*.min.js'], }, ];
Legacy .eslintrc Format
// .eslintrc.json - Legacy format (ESLint v8) { "env": { "browser": true, "es2021": true }, "extends": "eslint:recommended", "parserOptions": { "ecmaVersion": "latest", "sourceType": "module" }, "rules": { "no-unused-vars": "warn", "no-console": "warn", "eqeqeq": "error" }, "ignorePatterns": ["dist/", "node_modules/"] }
Run ESLint
# Check for errors npx eslint src/ # Auto-fix problems npx eslint src/ --fix # Check specific file types npx eslint "src/**/*.js" # Add to package.json { "scripts": { "lint": "eslint src/", "lint:fix": "eslint src/ --fix" } }
Common ESLint Rules
  • no-unused-vars - Catch dead code
  • eqeqeq - Require === instead of ==
  • no-var - Enforce let/const
  • prefer-const - Use const when possible
  • no-console - Flag console.log in prod

2. Stylelint Configuration

Installation
# Install Stylelint with SCSS support npm install --save-dev stylelint stylelint-config-standard-scss
.stylelintrc.json
{ "extends": "stylelint-config-standard-scss", "rules": { // Allow Bootstrap's class naming "selector-class-pattern": null, // Allow Bootstrap's @import usage "scss/at-import-partial-extension": null, // Disable rules that conflict with Bootstrap "scss/no-global-function-name": null, // Enforce consistent formatting "indentation": 2, "max-empty-lines": 1, // Color rules "color-hex-length": "short", "color-no-invalid-hex": true, // Property rules "declaration-block-no-duplicate-properties": true, "shorthand-property-no-redundant-values": true, // SCSS specific "scss/dollar-variable-pattern": "^[a-z][a-z0-9-]*$", "scss/percent-placeholder-pattern": "^[a-z][a-z0-9-]*$" } }
Run Stylelint
# Lint SCSS files npx stylelint "src/**/*.scss" # Auto-fix npx stylelint "src/**/*.scss" --fix # package.json scripts { "scripts": { "lint:css": "stylelint 'src/**/*.scss'", "lint:css:fix": "stylelint 'src/**/*.scss' --fix" } }
Rules for Bootstrap Projects
// Stylelint rules that work well with Bootstrap { "rules": { // Enforce Bootstrap variable naming "scss/dollar-variable-pattern": "^(_)?[a-z][a-z0-9-]*$", // Allow @extend with Bootstrap classes "scss/selector-no-redundant-nesting-selector": true, // Require @use over @import "scss/load-no-partial-leading-underscore": "always", // Consistent property order "order/properties-alphabetical-order": null, // OR custom order: "order/properties-order": [ ["position", "top", "right", "display", "flex"], { "unspecified": "bottomAlphabetical" } ] } }

3. Prettier Configuration

Installation
npm install --save-dev prettier
.prettierrc
{ "semi": true, "singleQuote": true, "tabWidth": 2, "useTabs": false, "trailingComma": "es5", "printWidth": 80, "bracketSpacing": true, "arrowParens": "always", "endOfLine": "lf", // HTML-specific "htmlWhitespaceSensitivity": "ignore", "wrapAttributes": "auto", // SCSS-specific "singleQuote": true, // Override for specific files "overrides": [ { "files": "*.html", "options": { "printWidth": 120 } } ] }
.prettierignore
# Ignore build output dist/ build/ # Ignore dependencies node_modules/ # Ignore minified files *.min.js *.min.css # Ignore Bootstrap source (if vendored) vendor/bootstrap/
Run Prettier
# Format all files npx prettier --write . # Check (without modifying) npx prettier --check . # Format specific file types npx prettier --write "src/**/*.{js,scss,html}" # package.json { "scripts": { "format": "prettier --write .", "format:check": "prettier --check ." } }
VS Code Integration
// .vscode/settings.json { "editor.formatOnSave": true, "editor.defaultFormatter": "esbenp.prettier-vscode", "editor.codeActionsOnSave": { "source.fixAll.eslint": "explicit", "source.fixAll.stylelint": "explicit" }, "scss.validate": false, // Use Stylelint for SCSS validation instead "[scss]": { "editor.defaultFormatter": "esbenp.prettier-vscode" } }
Recommended Extensions
  • Prettier - esbenp.prettier-vscode
  • ESLint - dbaeumer.vscode-eslint
  • Stylelint - stylelint.vscode-stylelint
  • SCSS IntelliSense - mrmlnc.vscode-scss

4. Tool Integration

Resolving Conflicts
// Disable formatting rules in ESLint (let Prettier handle it) npm install --save-dev eslint-config-prettier // eslint.config.js import prettier from 'eslint-config-prettier'; export default [ js.configs.recommended, prettier, // Disables conflicting rules { rules: { // Your custom rules here } } ];
Combined npm Scripts
{ "scripts": { "lint": "npm run lint:js && npm run lint:css", "lint:js": "eslint 'src/**/*.js'", "lint:css": "stylelint 'src/**/*.scss'", "lint:fix": "npm run lint:js:fix && npm run lint:css:fix", "lint:js:fix": "eslint 'src/**/*.js' --fix", "lint:css:fix": "stylelint 'src/**/*.scss' --fix", "format": "prettier --write .", "format:check": "prettier --check .", "validate": "npm run format:check && npm run lint" } }
Tool Responsibilities
Tool Handles
Prettier Formatting: indentation, line length, quotes, semicolons
ESLint JS quality: unused vars, == vs ===, best practices
Stylelint CSS/SCSS quality: invalid properties, duplicates, patterns
Execution Order
  1. Prettier - Format code first
  2. ESLint - Then check for JS issues
  3. Stylelint - Then check for CSS issues
  4. Pre-commit hook - Run all automatically
