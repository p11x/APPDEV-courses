Basic Build & Deploy
# .github/workflows/deploy.yml name: Build and Deploy # Trigger on push to main branch on: push: branches: [main] # Only one deploy at a time concurrency: group: "pages" cancel-in-progress: false jobs: build-and-deploy: runs-on: ubuntu-latest steps: # 1. Get the code - name: Checkout uses: actions/checkout@v4 # 2. Set up Node.js - name: Setup Node.js uses: actions/setup-node@v4 with: node-version: 20 cache: 'npm' # 3. Install dependencies (from lock file) - name: Install dependencies run: npm ci # 4. Run linting - name: Lint run: npm run lint # 5. Build for production - name: Build run: npm run build # 6. Deploy to GitHub Pages - name: Deploy to Pages uses: peaceiris/actions-gh-pages@v3 with: github_token: ${{ secrets.GITHUB_TOKEN }} publish_dir: ./dist
Environment Configuration
# .github/workflows/deploy-with-env.yml name: Deploy with Environment on: push: branches: [main] jobs: deploy: runs-on: ubuntu-latest # Environment secrets and variables environment: name: production url: ${{ steps.deploy.outputs.url }} steps: - uses: actions/checkout@v4 - uses: actions/setup-node@v4 with: node-version: 20 cache: 'npm' - run: npm ci - run: npm run build env: # Pass environment variables to build VITE_API_URL: ${{ vars.API_URL }} VITE_APP_VERSION: ${{ github.sha }} - name: Deploy id: deploy run: | echo "Deploying version ${{ github.sha }}" # Deploy commands here
Caching for Speed
# Cache node_modules between runs - uses: actions/setup-node@v4 with: node-version: 20 cache: 'npm' # Automatically caches ~/.npm # Or manual cache - uses: actions/cache@v4 with: path: node_modules key: ${{ runner.os }}-node-${{ hashFiles('package-lock.json') }} restore-keys: | ${{ runner.os }}-node-

2. Netlify Deployment

netlify.toml Configuration
# netlify.toml # Build settings [build] base = "/" publish = "dist" command = "npm run build" # Production context [build.environment] NODE_VERSION = "20" NPM_VERSION = "10" # Deploy previews (PRs) [context.deploy-preview] command = "npm run build" # Branch deploys [context.branch-deploy] command = "npm run build" # Redirects for SPA routing [[redirects]] from = "/*" to = "/index.html" status = 200 # Headers for security and caching [[headers]] for = "/*" [headers.values] X-Frame-Options = "DENY" X-Content-Type-Options = "nosniff" Referrer-Policy = "strict-origin-when-cross-origin" [[headers]] for = "/*.css" [headers.values] Cache-Control = "public, max-age=31536000, immutable" [[headers]] for = "/*.js" [headers.values] Cache-Control = "public, max-age=31536000, immutable"
Netlify CLI Deploy
# Install Netlify CLI npm install -g netlify-cli # Login netlify login # Initialize site netlify init # Deploy to preview netlify deploy # Deploy to production netlify deploy --prod # package.json scripts { "scripts": { "deploy:preview": "npm run build && netlify deploy", "deploy:prod": "npm run build && netlify deploy --prod" } }
Netlify Features
  • Automatic deploys from Git push
  • Deploy previews for PRs
  • Split testing (A/B)
  • Forms handling
  • Edge functions
  • Free SSL certificates

Vercel CLI
# Install Vercel CLI npm install -g vercel # Login vercel login # Deploy preview vercel # Deploy production vercel --prod # Pull environment variables vercel env pull .env.local
GitHub Integration
# .github/workflows/vercel.yml name: Vercel Deploy on: push: branches: [main] jobs: deploy: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - uses: amondnet/vercel-action@v25 with: vercel-token: ${{ secrets.VERCEL_TOKEN }} vercel-org-id: ${{ secrets.VERCEL_ORG_ID }} vercel-project-id: ${{ secrets.VERCEL_PROJECT_ID }} vercel-args: '--prod'

4. Complete CI/CD Pipeline

# .github/workflows/ci-cd.yml name: CI/CD Pipeline on: push: branches: [main] pull_request: branches: [main] jobs: # Job 1: Lint and format check lint: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - uses: actions/setup-node@v4 with: node-version: 20 cache: 'npm' - run: npm ci - run: npm run format:check - run: npm run lint # Job 2: Run tests test: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - uses: actions/setup-node@v4 with: node-version: 20 cache: 'npm' - run: npm ci - run: npm run test:coverage # Job 3: Security audit security: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - uses: actions/setup-node@v4 with: node-version: 20 cache: 'npm' - run: npm ci - run: npm audit --audit-level=high # Job 4: Build build: runs-on: ubuntu-latest needs: [lint, test, security] steps: - uses: actions/checkout@v4 - uses: actions/setup-node@v4 with: node-version: 20 cache: 'npm' - run: npm ci - run: npm run build - uses: actions/upload-artifact@v4 with: name: dist path: dist/ # Job 5: Deploy (only on main branch) deploy: runs-on: ubuntu-latest needs: [build] if: github.ref == 'refs/heads/main' environment: name: production url: https://your-site.com steps: - uses: actions/download-artifact@v4 with: name: dist path: dist/ - name: Deploy run: | echo "Deploying to production..." # Deploy commands (Netlify, Vercel, etc.)
Deployment Best Practices
  • Always use npm ci in CI (not npm install)
  • Cache node_modules between pipeline runs
  • Run lint/test/security before build
  • Deploy only from main branch
  • Use environment protection rules
  • Set up deploy previews for PRs
  • Use immutable deployment artifacts
