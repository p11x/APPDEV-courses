2. Unit & Component Testing

Jest Setup
# Install Jest npm install --save-dev jest @testing-library/dom npm install --save-dev @testing-library/jest-dom npm install --save-dev jest-environment-jsdom
jest.config.js
// jest.config.js module.exports = { testEnvironment: 'jsdom', setupFilesAfterSetup: ['<rootDir>/src/test/setup.js'], moduleNameMapper: { // Handle CSS imports in tests '\\.(css|scss)$': 'identity-obj-proxy', }, collectCoverageFrom: [ 'src/**/*.js', '!src/**/*.test.js', ], coverageThresholds: { global: { branches: 80, functions: 80, lines: 80, }, }, };
Testing Bootstrap Components
// src/__tests__/modal.test.js import { Modal } from 'bootstrap'; import '@testing-library/jest-dom'; describe('Modal Component', () => { beforeEach(() => { document.body.innerHTML = ` <button data-bs-toggle="modal" data-bs-target="#testModal"> Open </button> <div class="modal" id="testModal"> <div class="modal-dialog"> <div class="modal-content"> <div class="modal-body">Test Content</div> </div> </div> </div> `; }); test('should open modal on trigger click', () => { const trigger = document.querySelector('[data-bs-toggle]'); const modal = new Modal(document.getElementById('testModal')); trigger.click(); expect(document.getElementById('testModal')) .toHaveClass('show'); }); test('should close modal', () => { const modal = new Modal(document.getElementById('testModal')); modal.show(); modal.hide(); expect(document.getElementById('testModal')) .not.toHaveClass('show'); }); });

3. End-to-End Testing

Playwright Setup
# Install Playwright npm init playwright@latest
Bootstrap E2E Tests
// tests/navbar.spec.js import { test, expect } from '@playwright/test'; test('navbar collapse on mobile', async ({ page }) => { // Set mobile viewport await page.setViewportSize({ width: 375, height: 667 }); await page.goto('http://localhost:5173'); // Navbar should be collapsed const nav = page.locator('.navbar-collapse'); await expect(nav).not.toBeVisible(); // Click toggle button await page.locator('.navbar-toggler').click(); // Nav should be visible now await expect(nav).toBeVisible(); }); test('modal opens and closes', async ({ page }) => { await page.goto('http://localhost:5173'); // Open modal await page.locator('[data-bs-target="#myModal"]').click(); await expect(page.locator('#myModal')).toBeVisible(); // Close with Escape await page.keyboard.press('Escape'); await expect(page.locator('#myModal')).not.toBeVisible(); });
Visual Regression Testing
# Playwright visual comparisons test('card component renders correctly', async ({ page }) => { await page.goto('http://localhost:5173/components/cards'); // Compare screenshot with baseline await expect(page.locator('.card-container')) .toHaveScreenshot('card-component.png'); }); # Generate baseline: npx playwright test --update-snapshots
Playwright Config
// playwright.config.js import { defineConfig } from '@playwright/test'; export default defineConfig({ testDir: './tests', timeout: 30000, use: { baseURL: 'http://localhost:5173', screenshot: 'only-on-failure', trace: 'on-first-retry', }, projects: [ { name: 'chromium', use: { browserName: 'chromium' } }, { name: 'firefox', use: { browserName: 'firefox' } }, { name: 'webkit', use: { browserName: 'webkit' } }, { name: 'mobile', use: { ...devices['iPhone 13'] } }, ], });
