Modals: Testing Strategies

Modal testing patterns, lifecycle events, and assertion strategies

1. Modal Lifecycle Events

Understanding modal events is essential for writing reliable tests.

Event When Fired Use in Testing
show.bs.modal Immediately when show is called Assert initial state before animation
shown.bs.modal After modal is visible to user Assert focus, visibility, content
hide.bs.modal Immediately when hide is called Assert state before close animation
hidden.bs.modal After modal is completely hidden Assert cleanup, DOM state, focus return

2. Testing Patterns Reference

Pattern 1: Testing Modal Visibility
// Assert modal is visible after show
test('modal becomes visible', function(done) {
    const modal = document.getElementById('testModal');
    modal.addEventListener('shown.bs.modal', function() {
        assert.ok(modal.classList.contains('show'));
        assert.equal(modal.style.display, 'block');
        assert.equal(modal.getAttribute('aria-hidden'), 'false');
        done();
    });
    new bootstrap.Modal(modal).show();
});
Pattern 2: Testing Modal Close
// Assert modal is hidden after close
test('modal closes on button click', function(done) {
    const modal = document.getElementById('testModal');
    const bsModal = new bootstrap.Modal(modal);
    modal.addEventListener('hidden.bs.modal', function() {
        assert.ok(!modal.classList.contains('show'));
        assert.equal(modal.style.display, '');
        done();
    });
    bsModal.show();
    setTimeout(() => {
        modal.querySelector('.btn-close').click();
    }, 100);
});
Pattern 3: Testing Focus Management
// Assert focus moves into modal and returns on close
test('focus is trapped and restored', function(done) {
    const trigger = document.getElementById('openBtn');
    const modal = document.getElementById('testModal');
    const bsModal = new bootstrap.Modal(modal);

    modal.addEventListener('shown.bs.modal', function() {
        // Focus should be inside the modal
        assert.ok(modal.contains(document.activeElement));
        bsModal.hide();
    });

    modal.addEventListener('hidden.bs.modal', function() {
        // Focus should return to trigger
        assert.equal(document.activeElement, trigger);
        done();
    });

    trigger.focus();
    bsModal.show();
});
Pattern 4: Testing Form in Modal
// Assert form data is captured correctly
test('modal form submits with correct data', function(done) {
    const modal = document.getElementById('formModal');
    const bsModal = new bootstrap.Modal(modal);

    modal.addEventListener('shown.bs.modal', function() {
        const input = modal.querySelector('#emailInput');
        input.value = 'test@example.com';

        modal.querySelector('form').addEventListener('submit', function(e) {
            e.preventDefault();
            assert.equal(input.value, 'test@example.com');
            done();
        });

        modal.querySelector('[type="submit"]').click();
    });

    bsModal.show();
});
Pattern 5: Testing Keyboard (ESC) Close
// Assert ESC closes the modal
test('ESC key closes modal', function(done) {
    const modal = document.getElementById('testModal');
    const bsModal = new bootstrap.Modal(modal);

    modal.addEventListener('shown.bs.modal', function() {
        // Simulate ESC keypress
        const escEvent = new KeyboardEvent('keydown', {
            key: 'Escape',
            keyCode: 27,
            bubbles: true
        });
        document.dispatchEvent(escEvent);
    });

    modal.addEventListener('hidden.bs.modal', function() {
        assert.ok(true, 'Modal closed by ESC');
        done();
    });

    bsModal.show();
});
Pattern 6: Testing Backdrop Click
// Assert clicking backdrop closes modal
test('backdrop click closes modal', function(done) {
    const modal = document.getElementById('testModal');
    const bsModal = new bootstrap.Modal(modal);

    modal.addEventListener('shown.bs.modal', function() {
        const backdrop = document.querySelector('.modal-backdrop');
        backdrop.click();
    });

    modal.addEventListener('hidden.bs.modal', function() {
        assert.ok(true, 'Modal closed by backdrop click');
        done();
    });

    bsModal.show();
});
