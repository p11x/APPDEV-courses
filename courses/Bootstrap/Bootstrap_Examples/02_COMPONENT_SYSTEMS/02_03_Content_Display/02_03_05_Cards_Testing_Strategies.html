2. Unit Test Patterns (Reference)

Testing Pattern Reference
Jest + Testing Library
// Test: card renders product title
test('renders product card with title', () => {
  render(<ProductCard title="Test" />);
  expect(
    screen.getByTestId('product-title')
  ).toHaveTextContent('Test');
});

// Test: add to cart button click
test('calls onAddToCart when clicked', () => {
  const handler = jest.fn();
  render(<ProductCard onAddToCart={handler} />);
  fireEvent.click(
    screen.getByTestId('add-to-cart-btn')
  );
  expect(handler).toHaveBeenCalled();
});

// Test: loading state
test('shows skeleton when loading', () => {
  render(<ProductCard state="loading" />);
  expect(
    screen.getByTestId('loading-skeleton')
  ).toBeInTheDocument();
});
Cypress E2E
// Test: product card visibility
describe('Product Cards', () => {
  it('displays product cards', () => {
    cy.visit('/products');
    cy.get('[data-testid="product-card-001"]')
      .should('be.visible');
    cy.get('[data-testid="product-title"]')
      .should('contain', 'Test Product');
  });

  // Test: add to cart interaction
  it('adds product to cart', () => {
    cy.get('[data-testid="add-to-cart-btn"]')
      .click();
    cy.get('[data-testid="cart-count"]')
      .should('contain', '1');
  });

  // Test: responsive layout
  it('stacks cards on mobile', () => {
    cy.viewport('iphone-6');
    cy.get('.card').first()
      .should('have.css', 'width', '100%');
  });
});
