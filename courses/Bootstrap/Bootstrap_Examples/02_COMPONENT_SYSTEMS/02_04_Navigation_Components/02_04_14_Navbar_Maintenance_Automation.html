14.2 Navbar Renderer Class

/* ============================================================= */
/* NAVBAR RENDERER                                                  */
/* Takes config object and generates Bootstrap 5 navbar HTML        */
/* Supports dynamic updates, ARIA attributes, and validation        */
/* ============================================================= */

class NavbarRenderer {
    constructor(config) {
        this.config = this.validate(config);
    }
    
    validate(config) {
        // Validate required fields
        if (!config.brand || !config.brand.text) {
            throw new Error('NavbarRenderer: brand.text is required');
        }
        if (!Array.isArray(config.links) || config.links.length === 0) {
            throw new Error('NavbarRenderer: links array is required');
        }
        // Set defaults
        return {
            theme: 'dark',
            background: 'bg-dark',
            breakpoint: 'lg',
            sticky: false,
            ariaLabel: 'Main navigation',
            search: { enabled: false },
            actions: [],
            ...config
        };
    }
    
    render() {
        const c = this.config;
        const collapseId = 'navCollapse_' + Math.random().toString(36).substr(2, 9);
        
        let html = '';
        
        // Navbar element
        const classes = [
            'navbar',
            `navbar-expand-${c.breakpoint}`,
            `navbar-${c.theme}`,
            c.background,
            c.sticky ? 'sticky-top' : ''
        ].filter(Boolean).join(' ');
        
        html += `<nav class="${classes}" aria-label="${c.ariaLabel}">\n`;
        html += `  <div class="container-fluid">\n`;
        
        // Brand
        html += this.renderBrand();
        
        // Toggler
        html += `    <button class="navbar-toggler" type="button" ` +
                `data-bs-toggle="collapse" data-bs-target="#${collapseId}" ` +
                `aria-controls="${collapseId}" aria-expanded="false" ` +
                `aria-label="Toggle navigation">\n` +
                `      <span class="navbar-toggler-icon"></span>\n` +
                `    </button>\n`;
        
        // Collapse container
        html += `    <div class="collapse navbar-collapse" id="${collapseId}">\n`;
        
        // Nav links
        html += this.renderLinks();
        
        // Search
        if (c.search.enabled) {
            html += this.renderSearch();
        }
        
        // Action buttons
        if (c.actions.length > 0) {
            html += this.renderActions();
        }
        
        html += `    </div>\n`;
        html += `  </div>\n`;
        html += `</nav>`;
        
        return html;
    }
    
    renderBrand() {
        const b = this.config.brand;
        let content = b.text;
        if (b.logo) {
            content = `<img src="${b.logo.src}" alt="${b.logo.alt}" ` +
                     `width="${b.logo.width}" class="me-2">${b.text}`;
        }
        return `    <a class="navbar-brand" href="${b.href}">${content}</a>\n`;
    }
    
    renderLinks() {
        let html = `      <ul class="navbar-nav me-auto mb-2 mb-lg-0">\n`;
        
        this.config.links.forEach(link => {
            const activeClass = link.active ? ' active' : '';
            const ariaCurrent = link.active ? ' aria-current="page"' : '';
            
            if (link.dropdown) {
                html += `        <li class="nav-item dropdown">\n`;
                html += `          <a class="nav-link dropdown-toggle${activeClass}" ` +
                       `href="${link.href}" data-bs-toggle="dropdown">${link.text}</a>\n`;
                html += `          <ul class="dropdown-menu">\n`;
                link.dropdown.forEach(item => {
                    if (item.divider) {
                        html += `            <li><hr class="dropdown-divider"></li>\n`;
                    } else {
                        html += `            <li><a class="dropdown-item" ` +
                               `href="${item.href}">${item.text}</a></li>\n`;
                    }
                });
                html += `          </ul>\n`;
                html += `        </li>\n`;
            } else {
                html += `        <li class="nav-item">\n`;
                html += `          <a class="nav-link${activeClass}"${ariaCurrent} ` +
                       `href="${link.href}">${link.text}</a>\n`;
                html += `        </li>\n`;
            }
        });
        
        html += `      </ul>\n`;
        return html;
    }
    
    renderSearch() {
        return `      <form class="d-flex" role="search" ` +
               `action="${this.config.search.action || '#'}">\n` +
               `        <input class="form-control me-2" type="search" ` +
               `placeholder="${this.config.search.placeholder}" ` +
               `aria-label="Search">\n` +
               `        <button class="btn btn-outline-light" ` +
               `type="submit">Search</button>\n` +
               `      </form>\n`;
    }
    
    renderActions() {
        let html = `      <div class="d-flex">\n`;
        this.config.actions.forEach(action => {
            html += `        <a href="${action.href}" ` +
                   `class="btn ${action.class} ms-2">${action.text}</a>\n`;
        });
        html += `      </div>\n`;
        return html;
    }
}

14.3 Automated Link Validation

/* ============================================================= */
/* AUTOMATED LINK VALIDATION                                         */
/* Checks all navbar links for:                                     */
/* - Missing href attributes                                        */
/* - Empty href values                                              */
/* - Broken internal links                                          */
/* - Accessibility issues (missing aria-label on icon-only links)   */
/* ============================================================= */

function validateNavbarLinks(navbarSelector) {
    const nav = document.querySelector(navbarSelector);
    const links = nav.querySelectorAll('a, button[data-bs-toggle]');
    const issues = [];
    
    links.forEach((link, i) => {
        const tag = link.tagName;
        const text = link.textContent.trim();
        
        if (tag === 'A') {
            // Check href
            if (!link.hasAttribute('href')) {
                issues.push({ severity: 'error', element: link, message: `Link #${i}: Missing href attribute` });
            } else if (link.getAttribute('href') === '') {
                issues.push({ severity: 'warning', element: link, message: `Link #${i}: Empty href value` });
            }
            
            // Check accessible name
            if (!text && !link.getAttribute('aria-label') && !link.querySelector('img[alt]')) {
                issues.push({ severity: 'error', element: link, message: `Link #${i}: No accessible name (missing text, aria-label, or alt)` });
            }
            
            // Check target="_blank" has rel
            if (link.getAttribute('target') === '_blank' && !link.getAttribute('rel')?.includes('noopener')) {
                issues.push({ severity: 'warning', element: link, message: `Link #${i}: target="_blank" without rel="noopener"` });
            }
        }
    });
    
    return issues;
}

14.4 Navbar CSS Audit

/* ============================================================= */
/* CSS AUDIT FOR NAVBARS                                            */
/* Checks for:                                                      */
/* - Deprecated Bootstrap 4 classes                                 */
/* - Custom !important overrides                                    */
/* - Missing responsive breakpoint classes                          */
/* - Vendor prefix needs                                            */
/* ============================================================= */

function auditNavbarCSS(navbarSelector) {
    const nav = document.querySelector(navbarSelector);
    const results = { warnings: [], info: [] };
    
    // Check for deprecated BS4 classes
    const deprecated = {
        'navbar-toggler-right': 'Replaced by default behavior in BS5',
        'navbar-toggler-left': 'Use order utilities instead',
        'ml-auto': 'Use ms-auto in BS5',
        'mr-auto': 'Use me-auto in BS5',
        'pl-0': 'Use ps-0 in BS5',
        'pr-0': 'Use pe-0 in BS5'
    };
    
    nav.querySelectorAll('*').forEach(el => {
        el.classList.forEach(cls => {
            if (deprecated[cls]) {
                results.warnings.push({
                    element: el.outerHTML.substring(0, 50),
                    message: `Deprecated class "${cls}": ${deprecated[cls]}`
                });
            }
        });
    });
    
    // Check for !important in inline styles
    if (nav.style.cssText.includes('!important')) {
        results.warnings.push({
            element: 'navbar',
            message: 'Inline style uses !important - consider specificity fixes'
        });
    }
    
    return results;
}
