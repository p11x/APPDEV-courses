13.3 Performance Observer Implementation

/* ============================================================= */
/* NAVIGATION PERFORMANCE MONITORING                                */
/* Uses Performance Observer API to track navbar interactions       */
/* ============================================================= */

// 1. Track paint timing (FCP, LCP impact from navbar)
const paintObserver = new PerformanceObserver((list) => {
    for (const entry of list.getEntries()) {
        console.log(`[Paint] ${entry.name}: ${entry.startTime.toFixed(1)}ms`);
    }
});
paintObserver.observe({ type: 'paint', buffered: true });

// 2. Track navbar interaction timing
class NavbarPerformanceMonitor {
    constructor(navbarSelector) {
        this.navbar = document.querySelector(navbarSelector);
        this.metrics = {
            collapseToggleTimes: [],
            dropdownOpenTimes: [],
            linkClickTimes: []
        };
        this.setupListeners();
    }
    
    setupListeners() {
        // Monitor collapse toggle
        this.navbar.addEventListener('show.bs.collapse', (e) => {
            this._startTime = performance.now();
        });
        this.navbar.addEventListener('shown.bs.collapse', (e) => {
            const duration = performance.now() - this._startTime;
            this.metrics.collapseToggleTimes.push(duration);
            this.report('collapse-toggle', duration);
        });
        
        // Monitor dropdown open
        this.navbar.addEventListener('show.bs.dropdown', (e) => {
            this._dropdownStart = performance.now();
        });
        this.navbar.addEventListener('shown.bs.dropdown', (e) => {
            const duration = performance.now() - this._dropdownStart;
            this.metrics.dropdownOpenTimes.push(duration);
            this.report('dropdown-open', duration);
        });
        
        // Monitor link clicks (navigation timing)
        this.navbar.querySelectorAll('.nav-link').forEach(link => {
            link.addEventListener('click', (e) => {
                const clickTime = performance.now();
                this.report('link-click', clickTime);
            });
        });
    }
    
    report(event, value) {
        // Send to analytics
        if (window.gtag) {
            gtag('event', 'navbar_performance', {
                event_category: 'navigation',
                event_label: event,
                value: Math.round(value)
            });
        }
        console.log(`[Navbar Perf] ${event}: ${value.toFixed(2)}ms`);
    }
    
    getReport() {
        const avg = arr => arr.length ? arr.reduce((a, b) => a + b) / arr.length : 0;
        return {
            avgCollapseTime: avg(this.metrics.collapseToggleTimes),
            avgDropdownTime: avg(this.metrics.dropdownOpenTimes),
            totalInteractions: this.metrics.collapseToggleTimes.length + 
                              this.metrics.dropdownOpenTimes.length +
                              this.metrics.linkClickTimes.length
        };
    }
}

// Initialize
const navMonitor = new NavbarPerformanceMonitor('#monitoredNav');

13.4 Lazy Loading Navigation Content

/* ============================================================= */
/* LAZY LOAD MEGA MENU CONTENT                                       */
/* Only fetch mega menu HTML when user hovers/clicks the trigger   */
/* Reduces initial page weight and speeds up Time to Interactive    */
/* ============================================================= */

class LazyMegaMenu {
    constructor(triggerSelector, contentUrl) {
        this.trigger = document.querySelector(triggerSelector);
        this.contentUrl = contentUrl;
        this.loaded = false;
        this.setupTrigger();
    }
    
    setupTrigger() {
        // Load content on first interaction
        const events = ['mouseenter', 'focus'];
        events.forEach(event => {
            this.trigger.addEventListener(event, () => this.loadContent(), { once: true });
        });
        // Also load on click (for touch devices)
        this.trigger.addEventListener('click', () => this.loadContent(), { once: true });
    }
    
    async loadContent() {
        if (this.loaded) return;
        
        const start = performance.now();
        
        try {
            const response = await fetch(this.contentUrl);
            const html = await response.text();
            
            // Insert mega menu content
            const dropdown = this.trigger.nextElementSibling;
            dropdown.innerHTML = html;
            
            this.loaded = true;
            
            const loadTime = performance.now() - start;
            console.log(`[Lazy Menu] Loaded in ${loadTime.toFixed(1)}ms`);
            
        } catch (error) {
            console.error('[Lazy Menu] Failed to load:', error);
        }
    }
}

// Usage
const megaMenu = new LazyMegaMenu('#productsTrigger', '/api/mega-menu.html');
