// Dynamic RTL detection
function setDirection(lang) {
    const rtlLanguages = ['ar', 'he', 'fa', 'ur'];
    document.documentElement.dir = rtlLanguages.includes(lang) ? 'rtl' : 'ltr';
    document.documentElement.lang = lang;
}

7.4 Character Encoding & Font Considerations

  • Always use UTF-8: <meta charset="UTF-8">
  • Include language-specific fonts: Arabic, Chinese, Japanese require specialized web fonts
  • Text expansion: German/French translations can be 30-40% longer than English
  • Test with real content: Avoid Lorem Ipsum for i18n testing
  • Date/number formats: Use Intl API for locale-aware formatting
