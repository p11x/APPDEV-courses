/* Multi-level dropdown CSS */
.dropdown-submenu {
    position: relative;
}
.dropdown-submenu .dropdown-menu {
    top: 0;
    left: 100%;     /* Opens to the right */
    margin-top: -0.125rem;
}
/* On mobile, stack vertically instead */
@media (max-width: 991.98px) {
    .dropdown-submenu .dropdown-menu {
        left: 0;
        position: relative;
        border: none;
        box-shadow: none;
        padding-left: 1rem;
    }
}

    .dropdown-hover:hover > .dropdown-menu {
        display: block;

        margin-top: 0;
    }
}
