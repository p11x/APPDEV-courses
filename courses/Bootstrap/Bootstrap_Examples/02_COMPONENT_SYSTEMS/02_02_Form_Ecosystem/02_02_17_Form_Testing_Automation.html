2. E2E Test Patterns (Playwright/Cypress)

Playwright Example
const { test, expect } = require('@playwright/test');

test('form validation', async ({ page }) => {
    await page.goto('/form');

    // Submit empty form
    await page.click('[data-testid="submit-btn"]');
    await expect(page.locator('[data-testid="name-error"]')).toBeVisible();

    // Fill and submit
    await page.fill('[data-testid="name-input"]', 'John');
    await page.fill('[data-testid="email-input"]', 'john@example.com');
    await page.fill('[data-testid="message-input"]', 'Hello');
    await page.click('[data-testid="submit-btn"]');

    // No errors
    await expect(page.locator('.invalid-feedback:visible')).toHaveCount(0);
});
Cypress Example
describe('Contact Form', () => {
    beforeEach(() => { cy.visit('/form'); });

    it('shows errors for empty submission', () => {
        cy.get('[data-testid="submit-btn"]').click();
        cy.get('[data-testid="name-error"]').should('be.visible');
    });

    it('submits successfully with valid data', () => {
        cy.get('[data-testid="name-input"]').type('Jane');
        cy.get('[data-testid="email-input"]').type('jane@test.com');
        cy.get('[data-testid="message-input"]').type('Test message');
        cy.get('[data-testid="submit-btn"]').click();
        cy.get('.is-invalid').should('have.length', 0);
    });
});

3. CI/CD Configuration

GitHub Actions
name: Form Tests
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
      - run: npm ci
      - run: npm run test:forms
      - run: npm run test:e2e
        env:
          BASE_URL: ${{ secrets.DEPLOY_URL }}

4. Visual Regression Tests

// Visual regression with Playwright
test('form visual regression', async ({ page }) => {
    await page.goto('/form');
    await expect(page.locator('form')).toHaveScreenshot('form-initial.png');

    // Fill form and compare
    await page.fill('#name', 'Test');
    await expect(page.locator('form')).toHaveScreenshot('form-filled.png');

    // Error state
    await page.click('[type="submit"]');
    await expect(page.locator('form')).toHaveScreenshot('form-errors.png');
});

5. Automated Accessibility Tests

// Using axe-core with Playwright
const AxeBuilder = require('@axe-core/playwright').default;

test('form accessibility', async ({ page }) => {
    await page.goto('/form');
    const results = await new AxeBuilder({ page }).include('form').analyze();
    expect(results.violations).toEqual([]);
});

// Using cypress-axe
it('has no a11y violations', () => {
    cy.visit('/form');
    cy.injectAxe();
    cy.checkA11y('form');
});
