5. Test Pattern Examples

Pattern: Test Form Submission
// Test: Valid form submission
function testValidSubmission() {
    // 1. Fill in all required fields
    fillField('email', 'test@example.com');
    fillField('password', 'SecurePass123!');

    // 2. Click submit
    clickButton('submit');

    // 3. Assert no errors shown
    assertNoErrors();

    // 4. Assert form was submitted
    assertFormSubmitted();
}
Pattern: Test Validation Errors
// Test: Validation error display
function testValidationErrors() {
    // 1. Leave required fields empty
    clearAllFields();

    // 2. Click submit
    clickButton('submit');

    // 3. Assert error messages appear
    assertError('email', 'Email is required');
    assertError('password', 'Password is required');

    // 4. Assert error fields have is-invalid class
    assert.hasClass('email-input', 'is-invalid');

    // 5. Assert focus moved to first error
    assertFocused('email-input');
}
Pattern: Test Accessibility
// Test: Form accessibility
function testAccessibility() {
    // 1. All inputs have labels
    document.querySelectorAll('input').forEach(input => {
        const label = document.querySelector(`label[for="${input.id}"]`);
        assert.notNull(label, `Input ${input.id} has no label`);
    });

    // 2. All fieldsets have legends
    document.querySelectorAll('fieldset').forEach(fs => {
        assert.notNull(fs.querySelector('legend'));
    });

    // 3. Error messages use aria-describedby
    document.querySelectorAll('.invalid-feedback').forEach(fb => {
        const input = fb.previousElementSibling;
        assert.equals(input.getAttribute('aria-describedby'), fb.id);
    });

    // 4. Form is keyboard navigable
    const focusable = form.querySelectorAll('input, button, select');
    focusable.forEach(el => {
        assert.greaterThanOrEqual(el.tabIndex, 0);
    });
}

6. Test Data Fixtures

Valid Data
const VALID_DATA = {
    firstName: 'John',
    lastName: 'Doe',
    email: 'john@example.com',
    password: 'SecurePass123!',
    role: 'user',
    terms: true
};
Invalid Data
const INVALID_DATA = {
    firstName: '',
    lastName: '',
    email: 'not-an-email',
    password: '123',
    role: '',
    terms: false
};
Edge Cases
const EDGE_DATA = {
    firstName: 'A', // min length
    lastName: 'x'.repeat(100), // max length
    email: 'a@b.c', // minimal valid
    password: '!@#$%^&*()Ab1', // special chars
    role: 'guest',
    terms: true
};
