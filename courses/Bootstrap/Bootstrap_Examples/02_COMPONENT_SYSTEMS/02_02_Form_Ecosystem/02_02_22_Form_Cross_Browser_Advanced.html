2. Polyfill Patterns

// FormData polyfill check
if (!window.FormData) {
    // Fallback: serialize form manually
    function serializeForm(form) {
        var data = [];
        for (var i = 0; i < form.elements.length; i++) {
            var el = form.elements[i];
            if (el.name && !el.disabled) {
                data.push(encodeURIComponent(el.name) + '=' + encodeURIComponent(el.value));
            }
        }
        return data.join('&');
    }
}

// Element.closest() polyfill
if (!Element.prototype.closest) {
    Element.prototype.closest = function(s) {
        var el = this;
        do { if (el.matches(s)) return el; el = el.parentElement; } while (el);
        return null;
    };
}

// Custom validity polyfill
if (!HTMLInputElement.prototype.setCustomValidity) {
    HTMLInputElement.prototype.setCustomValidity = function(msg) {
        this.dataset.customError = msg;
    };
}
