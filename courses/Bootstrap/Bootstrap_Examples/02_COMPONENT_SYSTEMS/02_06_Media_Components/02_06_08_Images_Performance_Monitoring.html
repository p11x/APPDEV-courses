Images: Performance Monitoring

Core Web Vitals, LCP, CLS, image metrics tracking, and monitoring tools

1. Core Web Vitals & Images

LCP
Largest Contentful Paint

Target: < 2.5s
Often caused by hero/banner images

CLS
Cumulative Layout Shift

Target: < 0.1
Caused by images without dimensions

INP
Interaction to Next Paint

Target: < 200ms
Affected by heavy image processing

2. Monitoring with JavaScript

Track Image Load Times
// Track individual image load performance
function trackImageLoad(img) {
    const startTime = performance.now();
    if (img.complete) {
        logMetric(img.src, 0, 'cached');
    } else {
        img.addEventListener('load', () => {
            const loadTime = performance.now() - startTime;
            logMetric(img.src, loadTime, 'loaded');
        });
        img.addEventListener('error', () => {
            logMetric(img.src, -1, 'error');
        });
    }
}

function logMetric(src, time, status) {
    const name = src.split('/').pop();
    console.log(`Image ${name}: ${time.toFixed(0)}ms (${status})`);
}

// Track all images
document.querySelectorAll('img').forEach(trackImageLoad);
Measure LCP Element
// Detect if LCP element is an image
new PerformanceObserver((list) => {
    const entries = list.getEntries();
    const lastEntry = entries[entries.length - 1];
    if (lastEntry.element?.tagName === 'IMG') {
        console.log('LCP Image:', lastEntry.element.src);
        console.log('LCP Time:', lastEntry.startTime.toFixed(0), 'ms');
        console.log('Image Size:', lastEntry.size, 'bytes');
    }
}).observe({ type: 'largest-contentful-paint', buffered: true });
