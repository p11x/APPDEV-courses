1. Build Pipeline Automation

Sharp (Node.js) - Automated Resize & Convert
// build-images.js - Run as part of build process
const sharp = require('sharp');
const glob = require('glob');
const path = require('path');

const sizes = [400, 800, 1200];
const formats = ['webp', 'avif', 'jpg'];

glob.sync('src/images/**/*.{jpg,png}').forEach(file => {
    sizes.forEach(width => {
        formats.forEach(format => {
            sharp(file)
                .resize(width)
                .toFormat(format, { quality: 80 })
                .toFile(`dist/images/${path.basename(file, path.extname(file))}-${width}.${format}`);
        });
    });
});
Webpack - image-webpack-loader
// webpack.config.js
module.exports = {
    module: {
        rules: [{
            test: /\.(jpe?g|png|gif|webp|avif)$/i,
            use: [
                { loader: 'file-loader', options: { name: '[name]-[hash].[ext]' } },
                { loader: 'image-webpack-loader', options: {
                    mozjpeg: { progressive: true, quality: 80 },
                    optipng: { enabled: true },
                    webp: { quality: 80 },
                    avif: { quality: 60 }
                }}
            ]
        }]
    }
};

3. Automated Alt Text Patterns

Build-time Alt Text Extraction
// alt-text-manifest.json - Generated at build time
{
    "hero-banner.jpg": {

        "generated": true,
        "source": "ai-vision-api",
        "confidence": 0.92
    },
    "team-sarah.jpg": {
        "alt": "Sarah Chen - CEO",
        "generated": false,
        "source": "manual"
    }

// Build script validates all images have alt text
const manifest = require('./alt-text-manifest.json');
const images = glob.sync('src/images/**/*.{jpg,png,webp}');

images.forEach(img => {
    if (!manifest[path.basename(img)]) {
        console.warn(`Missing alt text: ${img}`);
        // Flag for manual review or AI generation
    }
});

4. CI/CD Image Checks

GitHub Actions - Image Check
# .github/workflows/image-check.yml
name: Image Quality Check
on: [push, pull_request]
jobs:
  check-images:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Check image sizes
        run: |
          # Fail if any image exceeds 500KB
          find src/images -type f \( -name "*.jpg" -o -name "*.png" \) \
            -exec stat --format="%s %n" {} \; | \
            awk '$1 > 512000 { print "TOO LARGE: " $2; exit 1 }'

      - name: Check alt text in HTML
        run: |
          # Fail if any img tag lacks alt attribute
          ! grep -rn ']*>' src/ | grep -v 'alt='

      - name: Check for WebP alternatives
        run: |
          # Warn if JPEG exists without WebP version
          for jpg in $(find src/images -name "*.jpg"); do
            webp="${jpg%.jpg}.webp"
            [ ! -f "$webp" ] && echo "Missing WebP: $webp"
          done
