7. Programmatic Testing Pattern

// Example: Programmatic display utility tests
function testDisplayUtility(elementId, expectedDisplay) {
    const el = document.getElementById(elementId);
    const computed = window.getComputedStyle(el).display;
    const pass = computed === expectedDisplay;
    console[pass ? 'log' : 'error'](
        `Test ${elementId}: ${pass ? 'PASS' : 'FAIL'}`,
        `Expected: ${expectedDisplay}, Got: ${computed}`
    );
    return pass;
}

// Usage:
testDisplayUtility('test-none', 'none');
testDisplayUtility('test-block', 'block');
testDisplayUtility('test-flex', 'flex');
