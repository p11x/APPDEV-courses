  <button>Trigger</button>
  <div class="position-absolute bottom-100 start-50
              translate-middle-x mb-2 bg-dark text-white
              px-3 py-2 rounded">
    Tooltip text
    <!-- Arrow -->

                translate-middle-x"

                border-top: 6px solid #212529;">
    </div>
  </div>
</div>
