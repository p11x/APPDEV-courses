Global Defaults Reference

// Override global defaults for all components of a type
bootstrap.Modal.Default.backdrop = 'static';
bootstrap.Modal.Default.keyboard = false;
bootstrap.Tooltip.Default.delay = { show: 500, hide: 100 };
bootstrap.Toast.Default.autohide = false;

// Access component version
console.log(bootstrap.Modal.VERSION); // "5.3.3"
