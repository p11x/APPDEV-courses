Maintaining JavaScript Code

Patterns for organizing, documenting, and maintaining Bootstrap JavaScript in production.

1. Module Pattern

Organize Bootstrap component code into reusable modules.

Module Pattern Example
const ModalManager = (() => {
    const _instances = new Map();

    function create(id, options) {
        const el = document.getElementById(id);
        if (!el) return null;
        const modal = new bootstrap.Modal(el, options);
        _instances.set(id, modal);
        return modal;
    }

    function get(id) {
        return _instances.get(id) || null;
    }

    function destroy(id) {
        const modal = _instances.get(id);
        if (modal) { modal.dispose(); _instances.delete(id); }
    }

    return { create, get, destroy };
})();

3. Event Cleanup Registry

Track all event listeners for proper cleanup on component destruction.

class EventRegistry {
    constructor() { this._handlers = []; }

    on(element, event, handler) {
        element.addEventListener(event, handler);
        this._handlers.push({ element, event, handler });
    }

    cleanup() {
        this._handlers.forEach(({ element, event, handler }) => {
            element.removeEventListener(event, handler);
        });
        this._handlers = [];
    }
}
