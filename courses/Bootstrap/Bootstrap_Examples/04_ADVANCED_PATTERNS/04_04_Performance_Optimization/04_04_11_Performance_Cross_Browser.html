Feature Detection

CSS @supports
/* Progressive enhancement with @supports */
.container {
  display: flex; /* Fallback */
}

@supports (display: grid) {
  .container {
    display: grid;
    grid-template-columns: repeat(auto-fit,
      minmax(300px, 1fr));
  }
}

@supports (aspect-ratio: 16/9) {
  .video-wrapper {
    aspect-ratio: 16/9;
  }
}
JavaScript Feature Detection
// Check for IntersectionObserver
if ('IntersectionObserver' in window) {
  // Use native lazy loading observer
  const obs = new IntersectionObserver(...);
} else {
  // Fallback: load all images immediately
  loadAllImages();
}

// Check for fetch API
if ('fetch' in window) {
  fetch('/api/data').then(...);
} else {
  // Fallback to XMLHttpRequest
  const xhr = new XMLHttpRequest();
}

Conditional Polyfill Loading

// Load polyfills only for browsers that need them
(function() {
  const polyfills = [];

  if (!('IntersectionObserver' in window)) {
    polyfills.push('intersection-observer');
  }
  if (!('fetch' in window)) {
    polyfills.push('fetch');
  }
  if (!Element.prototype.closest) {
    polyfills.push('element-closest');
  }

  if (polyfills.length) {
    // Dynamic import of polyfill bundle
    const script = document.createElement('script');
    script.src = '/polyfills/' + polyfills.join(',') + '.js';
    document.head.appendChild(script);
  }
})();
Tip: Use polyfill.io or core-js for on-demand polyfill serving based on the User-Agent header.
