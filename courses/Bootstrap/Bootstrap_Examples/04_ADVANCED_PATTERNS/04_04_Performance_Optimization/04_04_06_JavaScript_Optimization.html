Event Delegation

Instead of attaching listeners to every button, attach one listener to a parent element and use event bubbling.

Without Delegation
// 100 buttons = 100 listeners
document.querySelectorAll('.btn')
  .forEach(btn => {
    btn.addEventListener('click',
      handleClick);
  });
// Memory: O(n) listeners

Each button gets its own listener. More memory, slower initialization.

With Delegation
// 100 buttons = 1 listener
document.getElementById('container')
  .addEventListener('click', e => {
    if (e.target.matches('.btn')) {
      handleClick(e);
    }
  });
// Memory: O(1) listeners

Single listener on parent. Handles dynamically added elements too.

Efficient DOM Manipulation

Build DOM nodes in a DocumentFragment (off-DOM), then append once. This triggers only one reflow.

// Efficient: single reflow
const fragment = document.createDocumentFragment();
for (let i = 0; i < 1000; i++) {
  const li = document.createElement('li');
  li.textContent = `Item ${i}`;
  fragment.appendChild(li);
}
list.appendChild(fragment); // 1 reflow

// Inefficient: 1000 reflows
for (let i = 0; i < 1000; i++) {
  list.innerHTML += `<li>Item ${i}</li>`; // reflow each time
}

Debounce (wait until pause)
function debounce(fn, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() =>
      fn(...args), delay);
  };
}

input.addEventListener('input',
  debounce(search, 300));
Throttle (max once per interval)
function throttle(fn, limit) {
  let last = 0;
  return (...args) => {
    const now = Date.now();
    if (now - last >= limit) {
      last = now;
      fn(...args);
    }
  };
}

window.addEventListener('scroll',
  throttle(onScroll, 100));

Schedule non-essential tasks during browser idle time to avoid blocking user interactions.

// Run analytics during idle time
requestIdleCallback(() => {
  trackUserBehavior();
  warmCache();
});

// Fallback for browsers without support
if (!('requestIdleCallback' in window)) {
  setTimeout(fn, 1);
}
