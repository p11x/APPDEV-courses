12. Web Animations API Code Examples

Basic WAAPI Usage
const anim = el.animate([
    { transform: 'translateX(0)', opacity: 1 },
    { transform: 'translateX(100px)', opacity: 0.5 },
    { transform: 'translateX(0)', opacity: 1 }
], {
    duration: 1000,
    easing: 'ease-in-out',
    iterations: Infinity,
    direction: 'alternate'
});

// Control playback
anim.pause();
anim.play();
anim.reverse();
anim.playbackRate = 2;
anim.cancel();
Choreography with WAAPI
const elements = document.querySelectorAll('.item');

elements.forEach((el, i) => {
    const anim = el.animate([
        { opacity: 0, transform: 'translateY(20px)' },
        { opacity: 1, transform: 'translateY(0)' }
    ], {
        duration: 400,
        delay: i * 100,
        easing: 'ease-out',
        fill: 'forwards'
    });

    // Wait for completion
    anim.onfinish = () => {
        console.log(`Item ${i} done`);
    };
});
