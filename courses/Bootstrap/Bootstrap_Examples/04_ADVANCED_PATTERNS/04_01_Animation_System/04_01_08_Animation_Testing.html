4. Animation Testing Toolkit

5. Test Code Examples

Jest Unit Test
test('element has animation class', () => {
  const el = document.createElement('div');
  el.classList.add('animated-card');

  expect(el.classList.contains('animated-card'))
    .toBe(true);

  const style = getComputedStyle(el);
  expect(style.animationName).toBe('fadeIn');
});

test('animation completes', async () => {
  const el = document.querySelector('.card');
  el.classList.add('animate');

  await new Promise(resolve => {
    el.addEventListener('animationend', resolve);
  });

  expect(el.style.opacity).toBe('1');
});
Cypress E2E Test
describe('Card Animation', () => {
  it('animates on scroll', () => {
    cy.visit('/page');
    cy.get('.card').should('have.css',
      'opacity', '0');

    cy.scrollTo('bottom');
    cy.get('.card').should('have.css',
      'opacity', '1');

    cy.get('.card').should('have.css',
      'transform', 'none');
  });

  it('respects reduced motion', () => {
    cy.visit('/page', {
      onBeforeLoad(win) {
        cy.stub(win, 'matchMedia')
          .withArgs('(prefers-reduced-motion: reduce)')
          .returns({ matches: true });
      }
    });
    cy.get('.card').should('have.css',
      'opacity', '1');
  });
});
Playwright Visual Test
test('card animation snapshot', async ({ page }) => {
  await page.goto('/page');

  // Wait for animation to start
  await page.waitForTimeout(100);

  // Screenshot during animation
  await expect(page).toHaveScreenshot(
    'card-animating.png',
    { maxDiffPixels: 100 }
  );

  // Wait for animation to end
  await page.waitForTimeout(500);

  // Screenshot after animation
  await expect(page).toHaveScreenshot(
    'card-complete.png'
  );
});
Accessibility Test
test('no flashing content', async ({ page }) => {
  await page.goto('/page');

  // Check no element flashes >3 times/sec
  const elements = await page.$$('*');
  for (const el of elements) {
    const style = await el.evaluate(e =>
      getComputedStyle(e)
    );
    const duration = parseFloat(
      style.animationDuration
    );
    // If duration < 0.33s and infinite,
    // it flashes >3 times/sec
    if (duration < 0.33) {
      const count = style.animationIterationCount;
      expect(count).not.toBe('infinite');
    }
  }
});

6. CI/CD Animation Testing Pipeline

# .github/workflows/animation-tests.yml
name: Animation Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4

      - name: Install dependencies
        run: npm ci

      - name: Unit tests (Jest)
        run: npm run test:unit -- --coverage

      - name: E2E tests (Playwright)
        run: npx playwright test

      - name: Visual regression
        run: npx playwright test --grep "@visual"
        # Compares screenshots against baselines

      - name: Accessibility audit
        run: npx pa11y-ci --config .pa11yci.json
        # Checks WCAG compliance including motion

      - name: Performance audit
        run: npx lighthouse-ci autorun
        # Lighthouse flags animations causing jank
