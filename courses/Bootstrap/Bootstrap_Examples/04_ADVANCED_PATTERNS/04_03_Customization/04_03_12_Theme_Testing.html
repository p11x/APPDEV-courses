Contrast Test (JavaScript)
// Test custom colors meet WCAG AA
function getLuminance(r, g, b) {
  const [rs, gs, bs] = [r, g, b]
    .map(c => c / 255 <= 0.03928
      ? c / 255 / 12.92
      : Math.pow((c / 255 + 0.055)
          / 1.055, 2.4));
  return 0.2126 * rs + 0.7152 * gs
       + 0.0722 * bs;
}

function contrastRatio(l1, l2) {
  const lighter = Math.max(l1, l2);
  const darker = Math.min(l1, l2);
  return (lighter + 0.05)
       / (darker + 0.05);
}

// Assert primary meets AA
expect(contrastRatio(
  getLuminance(255,255,255),
  getLuminance(79, 70, 229)
)).toBeGreaterThan(4.5);
CSS Variable Test (JavaScript)
// Test all theme variables are defined
const requiredVars = [
  '--bs-primary',
  '--bs-secondary',
  '--bs-success',
  '--bs-body-bg',
  '--bs-body-color',
  '--bs-font-sans-serif',
  '--bs-border-radius',
];

const styles = getComputedStyle(
  document.documentElement
);

requiredVars.forEach(varName => {
  const value = styles
    .getPropertyValue(varName)
    .trim();
  expect(value).not.toBe('');
});

// Test variable is valid color
const primary = styles
  .getPropertyValue('--bs-primary')
  .trim();
expect(primary).toMatch(
  /^#[0-9a-f]{6}$/i
);
