Minimal Theme Switcher
// Set theme document.documentElement.setAttribute('data-bs-theme', 'dark'); // Toggle theme const current = document.documentElement.getAttribute('data-bs-theme'); document.documentElement.setAttribute('data-bs-theme', current === 'dark' ? 'light' : 'dark'); // Persist to localStorage localStorage.setItem('bs-theme', 'dark'); // Load saved theme const saved = localStorage.getItem('bs-theme') || 'light'; document.documentElement.setAttribute('data-bs-theme', saved);
