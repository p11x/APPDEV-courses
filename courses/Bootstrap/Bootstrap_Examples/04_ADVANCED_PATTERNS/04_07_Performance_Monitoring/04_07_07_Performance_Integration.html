Integration Code Example
// RUM data collection with web-vitals
import { onLCP, onINP, onCLS } from 'web-vitals';

function sendToAnalytics(metric) {
  const body = JSON.stringify({
    name: metric.name,
    value: metric.value,
    delta: metric.delta,
    id: metric.id,
    navigationType: metric.navigationType,
    // Add context about the user session
    url: window.location.href,
    userAgent: navigator.userAgent,
    connectionType: navigator.connection?.effectiveType,
    timestamp: Date.now()
  });

  // Use Beacon API for reliable delivery
  if (navigator.sendBeacon) {
    navigator.sendBeacon('/api/vitals', body);
  } else {
    fetch('/api/vitals', { method: 'POST', body });
  }
}

onLCP(sendToAnalytics);
onINP(sendToAnalytics);
onCLS(sendToAnalytics);

// Server-side alert handler
app.post('/api/vitals', (req, res) => {
  const { name, value, url } = req.body;
  
  // Define thresholds
  const thresholds = {
    LCP: { warning: 2500, critical: 4000 },
    INP: { warning: 200, critical: 500 },
    CLS: { warning: 0.1, critical: 0.25 }
  };
  
  const threshold = thresholds[name];
  if (!threshold) return res.sendStatus(200);
  
  // Check if metric exceeds threshold
  if (value > threshold.critical) {
    sendAlert({
      severity: 'critical',
      metric: name,
      value,
      url,
      message: `${name} is ${value} on ${url}`
    });
  } else if (value > threshold.warning) {
    sendAlert({
      severity: 'warning',
      metric: name,
      value,
      url
    });
  }
  
  // Store in time-series DB
  storeMetric(req.body);
  res.sendStatus(204);
});
