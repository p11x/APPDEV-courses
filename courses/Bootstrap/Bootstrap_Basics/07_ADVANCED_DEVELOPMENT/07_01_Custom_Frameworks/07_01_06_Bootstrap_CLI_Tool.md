## Overview

Building a CLI tool for scaffolding Bootstrap projects automates the repetitive setup tasks involved in starting new Bootstrap-based applications. The CLI generates project structures, installs dependencies, creates component boilerplate, applies themes, and configures build tools through interactive prompts or command-line arguments.

A well-designed Bootstrap CLI follows conventions established by tools like Create React App and Vue CLI. It provides sensible defaults while allowing customization, supports both interactive and non-interactive modes, and generates code that follows Bootstrap best practices. The CLI can scaffold entire projects, individual components, or theme configurations depending on the command invoked.

The tool should be distributed as an npm package with a binary entry point, support plugin extensions for custom generators, and maintain a template system that makes it easy to update generated code when Bootstrap releases new versions.

## Basic Implementation

A CLI tool uses Commander.js for command parsing and Inquirer.js for interactive prompts.

```js
#!/usr/bin/env node
// bin/bootstrap-cli.js
const { program } = require('commander');
const inquirer = require('inquirer');
const fs = require('fs-extra');
const path = require('path');
const chalk = require('chalk');

program
  .name('bootstrap-cli')
  .description('CLI for scaffolding Bootstrap 5 projects')
  .version('1.0.0');

program
  .command('create <project-name>')
  .description('Create a new Bootstrap project')
  .option('-t, --template <type>', 'Project template', 'default')
  .option('--no-install', 'Skip dependency installation')
  .action(async (projectName, options) => {
    const answers = await inquirer.prompt([
      {
        type: 'list',
        name: 'template',
        message: 'Select a project template:',
        choices: ['default', 'admin-dashboard', 'landing-page', 'blog'],
        when: !options.template
      },
      {
        type: 'checkbox',
        name: 'features',
        message: 'Select features to include:',
        choices: [
          { name: 'SASS/SCSS support', value: 'sass', checked: true },
          { name: 'Icon library (Bootstrap Icons)', value: 'icons' },
          { name: 'Dark mode toggle', value: 'darkmode' },
          { name: 'RTL support', value: 'rtl' }
        ]
      },
      {
        type: 'confirm',
        name: 'typescript',
        message: 'Use TypeScript?',
        default: false
      }
    ]);

    const config = { projectName, ...options, ...answers };
    await scaffoldProject(config);
  });

program
  .command('generate <type> <name>')
  .alias('g')
  .description('Generate a component')
  .action(async (type, name) => {
    const generators = {
      component: generateComponent,
      page: generatePage,
      layout: generateLayout
    };

    if (!generators[type]) {
      console.error(chalk.red(`Unknown type: ${type}`));
      console.log(chalk.yellow('Available types: component, page, layout'));
      process.exit(1);
    }

    await generators[type](name);
  });

async function scaffoldProject(config) {
  const { projectName, template, features, typescript } = config;
  const projectDir = path.resolve(projectName);

  console.log(chalk.blue(`Creating ${projectName}...`));

  // Create directory structure
  const dirs = [
    'src/scss',
    'src/js',
    'src/images',
    'public'
  ];

  for (const dir of dirs) {
    await fs.ensureDir(path.join(projectDir, dir));
  }

  // Generate package.json
  const packageJson = {
    name: projectName,
    version: '1.0.0',
    scripts: {
      'build:css': 'sass src/scss/main.scss dist/css/main.css',
      'watch:css': 'sass --watch src/scss/main.scss dist/css/main.css',
      build: 'npm run build:css',
      start: 'npm run watch:css'
    },
    dependencies: {
      bootstrap: '^5.3.0'
    },
    devDependencies: {
      sass: '^1.69.0'
    }
  };

  await fs.writeJson(path.join(projectDir, 'package.json'), packageJson, { spaces: 2 });

  // Generate SCSS entry point
  const scssContent = generateScssEntryPoint(features);
  await fs.writeFile(path.join(projectDir, 'src/scss/main.scss'), scssContent);

  // Generate index.html
  const htmlContent = generateIndexHtml(projectName);
  await fs.writeFile(path.join(projectDir, 'public/index.html'), htmlContent);

  console.log(chalk.green(`Project ${projectName} created successfully!`));
}

program.parse();
```

## Advanced Variations

```js
// Advanced: Component generator with templates
async function generateComponent(name) {
  const pascalName = name.charAt(0).toUpperCase() + name.slice(1);
  const kebabName = name.replace(/([A-Z])/g, '-$1').toLowerCase().replace(/^-/, '');

  const componentDir = path.join(process.cwd(), 'src/components', kebabName);
  await fs.ensureDir(componentDir);

  // Component HTML template
  const htmlTemplate = `<!-- ${pascalName} Component -->
<div class="${kebabName}" data-component="${kebabName}">
  <div class="${kebabName}__header">
    <h3 class="${kebabName}__title">{{title}}</h3>
  </div>
  <div class="${kebabName}__body">
    {{content}}
  </div>
</div>
`;

  // Component SCSS
  const scssTemplate = `// ${pascalName} Component
.${kebabName} {
  @extend .card;

  &__header {
    @extend .card-header;
  }

  &__title {
    margin-bottom: 0;
    font-size: $h5-font-size;
  }

  &__body {
    @extend .card-body;
  }
}
`;

  // Component JS
  const jsTemplate = `// ${pascalName} Component
export class ${pascalName} {
  static selector = '[data-component="${kebabName}"]';

  constructor(element) {
    this.element = element;
    this.init();
  }

  init() {
    // Component initialization logic
  }

  static initAll() {
    document.querySelectorAll(${pascalName}.selector)
      .forEach(el => new ${pascalName}(el));
  }
}
`;

  await fs.writeFile(path.join(componentDir, `${kebabName}.html`), htmlTemplate);
  await fs.writeFile(path.join(componentDir, `_${kebabName}.scss`), scssTemplate);
  await fs.writeFile(path.join(componentDir, `${kebabName}.js`), jsTemplate);

  console.log(chalk.green(`Component ${pascalName} created at src/components/${kebabName}/`));
}

// Theme generator command
program
  .command('theme <name>')
  .description('Generate a custom Bootstrap theme')
  .option('-b, --base <color>', 'Primary color', '#0d6efd')
  .action(async (name, options) => {
    const themeDir = path.join(process.cwd(), 'src/scss/themes', name);
    await fs.ensureDir(themeDir);

    const themeScss = `// Theme: ${name}
$theme-${name}-primary: ${options.base};
$theme-${name}-secondary: mix(black, ${options.base}, 30%);

[data-bs-theme="${name}"] {
  --bs-primary: #{$theme-${name}-primary};
  --bs-secondary: #{$theme-${name}-secondary};
  --bs-body-bg: #{lighten(${options.base}, 45%)};
  --bs-body-color: #212529;
}
`;

    await fs.writeFile(path.join(themeDir, `_${name}.scss`), themeScss);
    console.log(chalk.green(`Theme "${name}" created`));
  });
```

## Best Practices

1. **Use Commander.js for argument parsing** - It provides subcommands, options, help generation, and validation out of the box.
2. **Provide interactive and non-interactive modes** - Support both `bootstrap-cli create myapp` and `bootstrap-cli create myapp --template admin` for CI/CD.
3. **Template with handlebars or ejs** - Use a templating engine for generated files so placeholders are clearly defined and maintainable.
4. **Validate project names** - Ensure project names are valid directory names (no spaces, special characters) before attempting to create directories.
5. **Include a --dry-run flag** - Allow users to preview what files will be generated without actually creating them.
6. **Log clear next steps** - After scaffolding, print the commands the user should run (e.g., `cd myapp && npm install && npm start`).
7. **Version templates separately** - Decouple template versions from CLI versions so templates can be updated without forcing a CLI upgrade.
8. **Support custom templates** - Allow users to provide their own template directories or registries for team-specific scaffolding.
9. **Handle errors gracefully** - Catch filesystem errors, network failures, and invalid inputs with helpful error messages.
10. **Include a doctor command** - Add `bootstrap-cli doctor` that checks for Node.js version, npm availability, and write permissions.
11. **Cache templates locally** - Download and cache remote templates to avoid network calls on every project creation.
12. **Respect existing files** - Never overwrite existing files without explicit confirmation or a `--force` flag.

## Common Pitfalls

1. **Hardcoded paths** - Using absolute paths in templates breaks portability; always use relative paths from the project root.
2. **Missing executable permissions** - The bin file needs `#!/usr/bin/env node` shebang and must be marked executable in package.json.
3. **Not handling existing directories** - Failing to check if the target directory exists before writing causes silent failures.
4. **Template drift** - Generated code becomes stale if templates aren't updated with each Bootstrap release.
5. **Ignoring .gitignore** - Not generating a proper `.gitignore` leads to `node_modules` being committed to version control.
6. **Blocking the event loop** - Using synchronous filesystem operations in large scaffolding scripts freezes the CLI during generation.

## Accessibility Considerations

The CLI should generate accessible templates by default. All scaffolded HTML must include proper semantic structure, ARIA landmarks, and skip navigation links.

```html
<!-- CLI-generated accessible template -->
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>{{PROJECT_NAME}}</title>
</head>
<body>
  <a href="#main-content" class="visually-hidden-focusable">Skip to main content</a>

  <header role="banner">
    <nav class="navbar" aria-label="Main navigation">
      <div class="container">
        <a class="navbar-brand" href="/" aria-label="{{PROJECT_NAME}} home">{{PROJECT_NAME}}</a>
      </div>
    </nav>
  </header>

  <main id="main-content" role="main" tabindex="-1">
    <div class="container my-5">
      <h1>Welcome</h1>
    </div>
  </main>

  <footer role="contentinfo" class="mt-auto py-3 bg-body-tertiary">
    <div class="container text-center">
      <span class="text-muted">&copy; {{YEAR}} {{PROJECT_NAME}}</span>
    </div>
  </footer>
</body>
</html>
```

## Responsive Behavior

Generated templates should include Bootstrap's responsive grid by default. The CLI can prompt for responsive preferences and generate appropriate column configurations.
