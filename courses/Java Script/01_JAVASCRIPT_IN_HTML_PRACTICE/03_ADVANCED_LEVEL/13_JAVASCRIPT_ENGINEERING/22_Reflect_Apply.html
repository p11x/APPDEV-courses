Reflect.apply

Reflect.apply Method

Reflect.apply(target, thisArg, argumentsList) calls a function with a given this value and arguments.

function greet(message, punctuation) {
    return message + ' ' + this.name + punctuation;
}

const person = { name: 'Alice' };

// Using Reflect.apply
const result = Reflect.apply(greet, person, ['Hello', '!']);
// Equivalent to: greet.call(person, 'Hello', '!')
console.log(result); // "Hello Alice!"
