Making Objects Iterable

Plain objects are not iterable by default. You can make them iterable by implementing Symbol.iterator.

const obj = {
    a: 1, b: 2, c: 3,
    [Symbol.iterator]() {
        const keys = Object.keys(this);
        let index = 0;
        return {
            next: () => ({
                value: [keys[index], this[keys[index++]]],
                done: index > keys.length
            })
        };
    }
};

for (const [key, value] of obj) {
    console.log(key, value);
}
