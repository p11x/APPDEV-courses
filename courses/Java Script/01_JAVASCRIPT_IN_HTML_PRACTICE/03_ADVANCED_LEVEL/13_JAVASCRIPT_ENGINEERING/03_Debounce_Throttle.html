Debounce & Throttle Techniques

1. Debounce

Delays function execution until after a pause in events. Great for search inputs.

// Debounce function
function debounce(func, wait) {
    let timeout;
    return function executedFunction(...args) {
        clearTimeout(timeout);
        timeout = setTimeout(() => {
            func.apply(this, args);
        }, wait);
    };
}

// Usage - search only fires after user stops typing
const searchInput = document.getElementById('search');
const handleSearch = debounce((query) => {
    console.log('Searching for:', query);
}, 500);

searchInput.addEventListener('input', (e) => {
    handleSearch(e.target.value);
});

2. Throttle

Ensures function executes at most once per specified interval. Great for scroll/resize.

// Throttle function
function throttle(func, limit) {
    let inThrottle;
    return function executedFunction(...args) {
        if (!inThrottle) {
            func.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}

// Usage - scroll event limited to once per 100ms
const handleScroll = throttle(() => {
    console.log('Scroll position:', window.scrollY);
}, 100);

window.addEventListener('scroll', handleScroll);

3. Advanced: Leading & Trailing Options

Execute immediately on first call (leading) or wait until quiet (trailing).

// Debounce with options
function debounceAdvanced(func, wait, options = {}) {
    let timeout;
    const { leading = false, trailing = true } = options;
    
    return function executedFunction(...args) {
        const callNow = leading && !timeout;
        
        clearTimeout(timeout);
        timeout = setTimeout(() => {
            if (trailing) func.apply(this, args);
            timeout = null;
        }, wait);
        
        if (callNow) func.apply(this, args);
    };
}
