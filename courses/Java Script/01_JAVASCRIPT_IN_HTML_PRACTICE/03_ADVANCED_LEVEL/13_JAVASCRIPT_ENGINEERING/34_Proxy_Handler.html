34. Proxy Handler

Handler API

The handler object defines custom behavior for proxy operations.

const handler = {
    get(target, property) { },
    set(target, property, value) { },
    has(target, property) { },
    deleteProperty(target, property) { },
    apply(target, thisArg, args) { },
    construct(target, args) { }
};

const proxy = new Proxy(target, handler);
