Custom Event Emitter Pattern

1. Basic Event Emitter

Implement publish-subscribe pattern for custom events.

// Event Emitter class
class EventEmitter {
    constructor() {
        this.events = {};
    }
    
    on(event, listener) {
        if (!this.events[event]) {
            this.events[event] = [];
        }
        this.events[event].push(listener);
    }
    
    emit(event, ...args) {
        if (this.events[event]) {
            this.events[event].forEach(listener => 
                listener(...args)
            );
        }
    }
    
    off(event, listenerToRemove) {
        if (!this.events[event]) return;
        this.events[event] = this.events[event].filter(
            listener => listener !== listenerToRemove
        );
    }
}

// Usage
const emitter = new EventEmitter();
emitter.on('message', (msg) => console.log('A:', msg));
emitter.on('message', (msg) => console.log('B:', msg));
emitter.emit('message', 'Hello!');

2. Once - Single Execution

Listen for event only one time, then auto-remove.

// Once method - fires only once
class EventEmitterWithOnce extends EventEmitter {
    once(event, listener) {
        const wrappedListener = (...args) => {
            listener(...args);
            this.off(event, wrappedListener);
        };
        this.on(event, wrappedListener);
    }
}

// Usage
emitter.once('connect', () => console.log('Connected!'));
emitter.emit('connect'); // fires
emitter.emit('connect'); // won't fire

3. Wildcard & Namespaces

Handle multiple events with wildcards or namespaced events.

// Wildcard support
class WildcardEmitter extends EventEmitter {
    on(event, listener, wildcard = false) {
        const key = wildcard ? '*' : event;
        if (!this.events[key]) {
            this.events[key] = [];
        }
        this.events[key].push({ listener, event });
    }
    
    emit(event, ...args) {
        // Specific event
        if (this.events[event]) {
            this.events[event].forEach(e => e.listener(...args));
        }
        // Wildcard listeners
        if (this.events['*']) {
            this.events['*']
                .filter(e => e.event === event || e.event === '*')
                .forEach(e => e.listener(...args));
        }
    }
}
