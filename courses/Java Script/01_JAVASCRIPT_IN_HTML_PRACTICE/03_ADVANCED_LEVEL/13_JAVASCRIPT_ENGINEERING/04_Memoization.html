Memoization Pattern

1. Basic Memoization

Cache expensive function results to avoid recomputation.

// Basic memoization function
function memoize(fn) {
    const cache = {};
    return function(...args) {
        const key = JSON.stringify(args);
        if (cache[key]) {
            return cache[key];
        }
        const result = fn.apply(this, args);
        cache[key] = result;
        return result;
    };
}

// Example: expensive Fibonacci
function fib(n) {
    if (n <= 1) return n;
    return fib(n - 1) + fib(n - 2);
}

const memoizedFib = memoize(fib);

2. Performance Comparison

Compare execution time with and without memoization.

// Without memoization - exponential time
function slowFib(n) {
    if (n <= 1) return n;
    return slowFib(n - 1) + slowFib(n - 2);
}

// With memoization - linear time
function fastFib() {
    const cache = { 0: 0, 1: 1 };
    return function fib(n) {
        if (cache[n] !== undefined) return cache[n];
        return cache[n] = fib(n - 1) + fib(n - 2);
    };
}

3. Cache Management

Advanced memoize with clear, has, and size methods.

// Memoize with cache management
function memoizeWithCache(fn) {
    const cache = {};
    const fn2 = function(...args) {
        const key = JSON.stringify(args);
        if (cache[key] !== undefined) {
            return cache[key];
        }
        const result = fn.apply(this, args);
        cache[key] = result;
        return result;
    };
    
    fn2.cache = cache;
    fn2.clear = () => { 
        Object.keys(cache).forEach(k => delete cache[k]); 
    };
    fn2.has = (key) => JSON.stringify(key) in cache;
    fn2.size = () => Object.keys(cache).length;
    
    return fn2;
}
