Pub/Sub Pattern

Publish-Subscribe Pattern

The Pub/Sub pattern decouples publishers from subscribers through an event/message broker. Publishers emit events without knowing subscribers.

// PubSub Event Bus
class PubSub {
    constructor() {
        this.events = {};
    }
    subscribe(event, callback) {
        if (!this.events[event]) {
            this.events[event] = [];
        }
        this.events[event].push(callback);
    }
    publish(event, data) {
        if (this.events[event]) {
            this.events[event].forEach(cb => cb(data));
        }
    }
    unsubscribe(event, callback) {
        if (this.events[event]) {
            this.events[event] = this.events[event].filter(cb => cb !== callback);
        }
    }
}

// Usage
const bus = new PubSub();
bus.subscribe('news', data => console.log('Subscriber 1:', data));
bus.subscribe('news', data => console.log('Subscriber 2:', data));
bus.publish('news', 'Breaking News!');
