30 - Redux Reducer

What is a Reducer?

A reducer is a pure function that takes the current state and an action, and returns a new state. Reducers must be pure functions with no side effects.

Reducer Pattern

function reducer(state, action) {
    // Return current state for unknown actions
    switch (action.type) {
        case 'INCREMENT':
            return { ...state, count: state.count + 1 };
        case 'DECREMENT':
            return { ...state, count: state.count - 1 };
        case 'SET':
            return { ...state, count: action.payload };
        default:
            return state;
    }
}
