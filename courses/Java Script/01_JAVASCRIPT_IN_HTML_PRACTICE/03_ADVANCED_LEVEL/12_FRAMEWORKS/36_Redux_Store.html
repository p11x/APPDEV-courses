36. Redux Store

Creating a Redux Store

Redux store holds the application state and provides methods to update it.

const createStore = (reducer) => { let state; let listeners = []; return { getState: () => state, dispatch: (action) => { state = reducer(state, action); listeners.forEach(listener => listener()); }, subscribe: (listener) => { listeners.push(listener); return () => { listeners = listeners.filter(l => l !== listener); }; } }; };
