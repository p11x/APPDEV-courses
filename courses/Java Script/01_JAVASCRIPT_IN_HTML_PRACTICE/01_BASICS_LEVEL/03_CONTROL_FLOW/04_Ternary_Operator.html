4. Ternary Operator

Example 1: Basic Ternary

// Syntax: condition ? valueIfTrue : valueIfFalse let age = 20; let status = age >= 18 ? "Adult" : "Minor"; console.log(status);

Example 2: Assignment with Ternary

// Assign values based on condition let score = 75; let grade = score >= 60 ? "Pass" : "Fail"; let percentage = score >= 50 ? 100 : 50; console.log("Grade: " + grade); console.log("Percentage: " + percentage + "%");

// Multiple conditions in ternary let marks = 85; let result = marks >= 90 ? "A" : marks >= 80 ? "B" : marks >= 70 ? "C" : marks >= 60 ? "D" : "F"; console.log("Your grade: " + result);

Example 4: Ternary in Function Return

// Use ternary for function return function getDiscount(isMember, purchase) { return isMember && purchase > 100 ? 20 : 10; } console.log("Member discount: " + getDiscount(true, 150) + "%"); console.log("Non-member discount: " + getDiscount(false, 150) + "%");

Example 5: Default Value with Ternary

// Set default value using ternary let userInput = ""; let name = userInput ? userInput : "Guest"; let count = null; let quantity = count !== null ? count : 0; console.log("Name: " + name); console.log("Quantity: " + quantity);
