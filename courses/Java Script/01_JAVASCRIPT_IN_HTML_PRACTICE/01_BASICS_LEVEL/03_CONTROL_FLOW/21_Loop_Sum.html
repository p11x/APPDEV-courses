21. Loop Sum

Example 1: Sum 1 to N

// Sum 1 to 10 let sum = 0; for (let i = 1; i <= 10; i++) { sum += i; } console.log("Sum: " + sum);

Example 2: Sum Array

// Sum array elements let nums = [1, 2, 3, 4, 5]; let total = 0; for (let i = 0; i < nums.length; i++) { total += nums[i]; } console.log("Total: " + total);
Output:

Example 3: Using While

// While loop sum let s = 0; let count = 1; while (count <= 5) { s += count; count++; } console.log("Sum: " + s);
Output:

Example 4: ForEach Sum

// forEach sum let arr = [10, 20, 30]; let sum = 0; arr.forEach(n => sum += n); console.log("Sum: " + sum);

Example 5: Even Numbers Sum

// Sum numbers > 3 let data = [1, 4, 2, 5, 3]; let result = 0; for (let i = 0; i < data.length; i++) { if (data[i] > 3) { result += data[i]; } } console.log("Sum > 3: " + result);
