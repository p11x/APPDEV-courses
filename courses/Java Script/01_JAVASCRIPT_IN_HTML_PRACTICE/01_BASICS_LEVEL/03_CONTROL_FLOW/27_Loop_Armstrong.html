27. Loop Armstrong

Example 1: Check Armstrong

// Check 153 is Armstrong // 1^3 + 5^3 + 3^3 = 153 let n = 153; let temp = n; let sum = 0; while (temp > 0) { let digit = temp % 10; sum += digit * digit * digit; temp = Math.floor(temp / 10); } console.log(n === sum);

Example 3: Find Armstrong (1-500)

// Find all Armstrong numbers let armstrongs = []; for (let n = 1; n <= 500; n++) { let temp = n; let sum = 0; let digits = 0; let t = n; while (t > 0) { digits++; t = Math.floor(t / 10); } t = n; while (t > 0) { let d = t % 10; let p = 1; for (let i = 0; i < digits; i++) p *= d; sum += p; t = Math.floor(t / 10); } if (sum === n) armstrongs.push(n); }
