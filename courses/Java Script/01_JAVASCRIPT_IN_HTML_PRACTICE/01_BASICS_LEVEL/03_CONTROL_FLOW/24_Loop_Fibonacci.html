24. Loop Fibonacci

Example 1: First 10 Numbers

// Generate Fibonacci let a = 0, b = 1; console.log(a); console.log(b); for (let i = 0; i < 8; i++) { let c = a + b; console.log(c); a = b; b = c; }

Example 2: Generate N Terms

// Generate N terms let n = 10; let fib = [0, 1]; for (let i = 2; i < n; i++) { fib[i] = fib[i-1] + fib[i-2]; }

Example 3: Sum of First 10

// Sum first 10 fib let a = 0, b = 1, sum = 1; for (let i = 0; i < 9; i++) { let c = a + b; sum += c; a = b; b = c; } console.log("Sum: " + sum);

Example 4: While Loop

// While fibonacci let a = 0, b = 1; let count = 0; while (count < 8) { let c = a + b; console.log(c); a = b; b = c; count++; }

Example 5: Nth Term

// Get Nth term let term = 10; let a = 0, b = 1; for (let i = 2; i <= term; i++) { let c = a + b; a = b; b = c; } console.log("Term " + term + ": " + b);
