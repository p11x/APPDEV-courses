Example 2: Loop Through Array

// Iterate over array elements let fruits = ["Apple", "Banana", "Orange"]; for (let i = 0; i < fruits.length; i++) { console.log(fruits[i]); }

Example 5: Calculate Sum

// Calculate sum of 1 to 10 let sum = 0; for (let i = 1; i <= 10; i++) { sum += i; } console.log("Sum: " + sum);

// Create a multiplication table for (let i = 1; i <= 3; i++) { let row = ""; for (let j = 1; j <= 3; j++) { row += (i * j) + " "; } console.log(row); }
