10. For Each Loop

Example 1: Basic forEach

// Iterate array elements let fruits = ["Apple", "Banana", "Orange"]; fruits.forEach(function(fruit) { console.log(fruit); });
Output:

Example 2: With Index

// Access index and array let colors = ["Red", "Green", "Blue"]; colors.forEach(function(color, index) { console.log(index + ": " + color); });
Output:

Example 3: Arrow Function

// Using arrow function let numbers = [1, 2, 3, 4, 5]; numbers.forEach(n => console.log(n * 2));
Output:

Example 4: Calculate Sum

// Sum all elements let nums = [1, 2, 3, 4, 5]; let sum = 0; nums.forEach(n => sum += n); console.log("Sum: " + sum);
Output:

Example 5: Transform Elements

// Double each element let arr = [1, 2, 3, 4]; let doubled = []; arr.forEach(n => doubled.push(n * 2)); console.log(doubled.join(", "));
Output:

// Iterate objects let users = [ {name: "Alice", age: 25}, {name: "Bob", age: 30} ]; users.forEach(user => { console.log(user.name + ": " + user.age); });
