25. Loop Prime

Example 1: Check Prime Number

// Check if number is prime let n = 17; let isPrime = true; for (let i = 2; i < n; i++) { if (n % i === 0) { isPrime = false; break; } } console.log(n + " is " + (isPrime ? "prime" : "not prime"));

Example 2: Find Primes 1 to 20

// Find all primes 1-20 let primes = []; for (let n = 2; n <= 20; n++) { let p = true; for (let i = 2; i < n; i++) { if (n % i === 0) { p = false; break; } } if (p) primes.push(n); }
