Example 4: 2D Array

// Iterate 2D array let matrix = [[1,2], [3,4], [5,6]]; for (let i = 0; i < matrix.length; i++) { for (let j = 0; j < matrix[i].length; j++) { console.log("[" + i + "][" + j + "] = " + matrix[i][j]); } }
