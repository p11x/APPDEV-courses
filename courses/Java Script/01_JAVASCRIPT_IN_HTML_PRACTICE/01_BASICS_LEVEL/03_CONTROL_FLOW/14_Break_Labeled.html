14. Break Labeled

Example 1: Basic Labeled Break

// Break outer loop with label outer: for (let i = 1; i <= 3; i++) { for (let j = 1; j <= 3; j++) { if (j === 2) break outer; console.log(i + "," + j); } }

// Find in 2D array let grid = [[1,2], [3,4], [5,6]]; outer: for (let i = 0; i < grid.length; i++) { for (let j = 0; j < grid[i].length; j++) { if (grid[i][j] === 4) { console.log("Found at " + i + "," + j); break outer; } } }
