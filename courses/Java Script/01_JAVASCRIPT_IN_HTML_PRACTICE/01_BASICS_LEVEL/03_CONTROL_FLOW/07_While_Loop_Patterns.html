7. While Loop Patterns

Example 1: Basic While Loop

// Count from 1 to 5 using while let count = 1; while (count <= 5) { console.log(count); count++; }
Output:

Example 2: While with Array

// Process array elements let items = ["A", "B", "C", "D"]; let index = 0; while (index < items.length) { console.log("Item: " + items[index]); index++; }
Output:

Example 3: Do-While Loop

// Execute at least once let num = 0; do { console.log(num); num++; } while (num < 3);
Output:

Example 4: While with Condition

// Find first number divisible by 7 let number = 1; while (number % 7 !== 0) { number++; } console.log("First divisible by 7: " + number);
Output:

Example 5: While with Break

// Stop when condition is met let sum = 0; let i = 1; while (true) { sum += i; if (sum > 20) { break; } i++; } console.log("Sum exceeded 20 at i=" + i + ", sum=" + sum);
Output:

Example 6: While with Continue

// Skip specific iterations let j = 0; while (j < 5) { j++; if (j === 3) { continue; } console.log(j); }
Output:

Example 7: While with String

// Process string character by character let text = "Hello"; let pos = 0; let result = ""; while (pos < text.length) { result += text[pos].toUpperCase(); pos++; } console.log(result);
