9. Loop Control - Break

Example 1: Break on Condition

// Stop when value is 5 for (let i = 1; i <= 10; i++) { if (i === 5) { break; } console.log(i); }

Example 2: Break in While Loop

// Stop at specific value let num = 1; while (num <= 10) { if (num === 7) { break; } console.log(num); num++; }

// Find value and break let arr = [3, 7, 2, 9, 5]; for (let j = 0; j < arr.length; j++) { if (arr[j] === 9) { console.log("Found at index: " + j); break; } }
