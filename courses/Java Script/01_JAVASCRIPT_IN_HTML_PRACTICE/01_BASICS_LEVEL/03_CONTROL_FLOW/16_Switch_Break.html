16. Switch Break

Example 1: Basic Switch

// Switch with break let day = 3; switch (day) { case 1: console.log("Mon"); break; case 2: console.log("Tue"); break; case 3: console.log("Wed"); break; case 4: console.log("Thu"); break; case 5: console.log("Fri"); break; default: console.log("Weekend"); }
