Constructor Functions in JavaScript

Constructor functions are used to create multiple objects of the same type.

1. Basic Constructor Function

// Constructor function naming convention: Capital first letter function Person(name, age) { this.name = name; this.age = age; } // Create instances using 'new' keyword const person1 = new Person("John", 30); const person2 = new Person("Alice", 25); console.log(person1); console.log(person2);

2. Constructor with Methods

// Adding methods inside constructor function BankAccount(owner, initialBalance) { this.owner = owner; this.balance = initialBalance; this.deposit = function(amount) { this.balance += amount; return this.balance; }; this.withdraw = function(amount) { if (amount <= this.balance) { this.balance -= amount; return this.balance; } return "Insufficient funds"; }; } const account = new BankAccount("Mike", 1000); account.deposit(500); account.withdraw(200); console.log(account);

3. Using Prototype for Methods (Memory Efficient)

// Add methods to prototype instead of inside constructor function Vehicle(type, brand) { this.type = type; this.brand = brand; } // Methods on prototype are shared across all instances Vehicle.prototype.start = function() { return `${this.brand} ${this.type} is starting...`; }; Vehicle.prototype.stop = function() { return `${this.brand} ${this.type} is stopping...`; }; const car1 = new Vehicle("Car", "Toyota"); const car2 = new Vehicle("Car", "Honda"); console.log(car1.start()); console.log(car2.start());

4. Constructor with Default Values

// Using default parameters in constructor function Book(title, author, pages = 0) { this.title = title || "Untitled"; this.author = author || "Unknown"; this.pages = pages; this.isRead = false; } const book1 = new Book("JavaScript Guide", "John Doe", 350); const book2 = new Book("Python Basics"); console.log(book1); console.log(book2);

5. Checking Instance Type

// Using instanceof to check object type function Animal(name, species) { this.name = name; this.species = species; } const dog = new Animal("Buddy", "Dog"); const cat = new Animal("Whiskers", "Cat"); console.log(dog instanceof Animal); // true console.log(cat instanceof Animal); // true console.log(dog instanceof Object); // true // Using constructor property console.log(dog.constructor === Animal);
