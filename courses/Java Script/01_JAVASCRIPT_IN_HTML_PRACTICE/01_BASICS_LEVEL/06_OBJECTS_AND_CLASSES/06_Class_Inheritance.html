Class Inheritance in JavaScript

Inheritance allows a class to inherit properties and methods from another class using 'extends' and 'super'.

1. Basic Inheritance with extends

// Parent class class Animal { constructor(name) { this.name = name; } speak() { return `${this.name} makes a sound`; } } // Child class inherits from Animal class Dog extends Animal { constructor(name, breed) { super(name); // Call parent constructor this.breed = breed; } } const dog = new Dog("Buddy", "Golden Retriever"); console.log(dog.name); console.log(dog.breed); console.log(dog.speak());

2. Method Overriding

// Parent class with greet method class Person { constructor(name) { this.name = name; } greet() { return `Hello, I am ${this.name}`; } } // Student overrides greet method class Student extends Person { constructor(name, grade) { super(name); this.grade = grade; } // Override the greet method greet() { return `Hi, I'm ${this.name} and I'm in grade ${this.grade}`; } } const student = new Student("Alice", 10); console.log(student.greet());

3. Using super to Call Parent Methods

// Parent class class Shape { constructor(color) { this.color = color; } describe() { return `A ${this.color} shape`; } } // Child class class Rectangle extends Shape { constructor(color, width, height) { super(color); this.width = width; this.height = height; } describe() { // Call parent method and extend it return `${super.describe()} - ${this.width}x${this.height} rectangle`; } getArea() { return this.width * this.height; } } const rect = new Rectangle("red", 10, 5); console.log(rect.describe()); console.log("Area:", rect.getArea());

4. Multi-level Inheritance

// Grandparent class class Vehicle { constructor(brand) { this.brand = brand; } getBrand() { return this.brand; } } // Parent class extends Vehicle class Car extends Vehicle { constructor(brand, model) { super(brand); this.model = model; } } // Child class extends Car class SportsCar extends Car { constructor(brand, model, topSpeed) { super(brand, model); this.topSpeed = topSpeed; } getDetails() { return `${this.getBrand()} ${this.model} - Top Speed: ${this.topSpeed} mph`; } } const sportsCar = new SportsCar("Ferrari", "488 GTB", 205); console.log(sportsCar.getDetails());

5. Checking Inheritance Relationships

// Parent class class Device { constructor(brand) { this.brand = brand; } } // Child class class Phone extends Device { constructor(brand, model) { super(brand); this.model = model; } } const myPhone = new Phone("Apple", "iPhone 14"); // instanceof checks inheritance chain console.log("myPhone instanceof Phone:", myPhone instanceof Phone); console.log("myPhone instanceof Device:", myPhone instanceof Device); console.log("myPhone instanceof Object:", myPhone instanceof Object); // prototype chain check console.log("Phone.prototype:", Object.getPrototypeOf(myPhone));
