This Keyword

'this' refers to the current execution context in JavaScript.

1. This in Object Method

const person = { name: "John", greet() { return `Hello, I'm ${this.name}`; } }; console.log(person.greet());

2. This in Constructor

function Car(brand, model) { this.brand = brand; this.model = model; this.getInfo = function() { return `${this.brand} ${this.model}`; }; } const car = new Car("Toyota", "Camry"); console.log(car.getInfo());

3. This in Standalone Function

function showThis() { console.log(this); } showThis();

4. This in Event Handler

const button = { label: "Click Me", click() { console.log(this.label); } }; const handler = button.click; handler();
