Prototype Inheritance

JavaScript uses prototypal inheritance where objects can inherit properties from other objects.

1. Prototypal Inheritance with Constructor

function Animal(name) { this.name = name; } Animal.prototype.speak = function() { return `${this.name} makes a sound`; }; function Dog(name, breed) { Animal.call(this, name); this.breed = breed; } Dog.prototype = Object.create(Animal.prototype); Dog.prototype.constructor = Dog; Dog.prototype.bark = function() { return `${this.name} barks!`; }; const dog = new Dog("Buddy", "Labrador"); console.log(dog.speak()); console.log(dog.bark());

2. Inheriting Prototype Methods

function Person(name) { this.name = name; } Person.prototype.greet = function() { return `Hello, I'm ${this.name}`; }; function Student(name, grade) { Person.call(this, name); this.grade = grade; } Student.prototype = Object.create(Person.prototype); Student.prototype.constructor = Student; Student.prototype.study = function() { return `${this.name} is studying`; }; const student = new Student("Alice", "A"); console.log(student.greet()); console.log(student.study()); console.log(student instanceof Person);

3. Multi-level Inheritance

function Vehicle(type) { this.type = type; this.wheels = 0; } Vehicle.prototype.getInfo = function() { return `Type: ${this.type}, Wheels: ${this.wheels}`; }; function Car(brand, model) { Vehicle.call(this, "Car"); this.brand = brand; this.model = model; this.wheels = 4; } Car.prototype = Object.create(Vehicle.prototype); Car.prototype.constructor = Car; Car.prototype.drive = function() { return `Driving ${this.brand} ${this.model}`; }; function SportsCar(brand, model, topSpeed) { Car.call(this, brand, model); this.topSpeed = topSpeed; } SportsCar.prototype = Object.create(Car.prototype); SportsCar.prototype.constructor = SportsCar; SportsCar.prototype.drift = function() { return `Drifting at ${this.topSpeed} mph!`; }; const sportsCar = new SportsCar("Ferrari", "488 GTB", 200); console.log(sportsCar.getInfo()); console.log(sportsCar.drive()); console.log(sportsCar.drift());
