Static Methods in JavaScript

Static methods belong to the class itself, not to instances. They are called on the class, not on objects.

1. Basic Static Method

// Static methods are called on the class, not instances class Calculator { // Static method - called on class directly static add(a, b) { return a + b; } static subtract(a, b) { return a - b; } } // Call using class name, not instance console.log(Calculator.add(10, 5)); // 15 console.log(Calculator.subtract(10, 5)); // 5 // const calc = new Calculator(); // calc.add(10, 5); // Error - cannot call static on instance

2. Static Utility Methods

// Common use case: utility functions class StringUtils { static capitalize(str) { return str.charAt(0).toUpperCase() + str.slice(1); } static reverse(str) { return str.split('').reverse().join(''); } static isPalindrome(str) { const reversed = str.split('').reverse().join(''); return str === reversed; } } console.log(StringUtils.capitalize("hello")); console.log(StringUtils.reverse("JavaScript")); console.log(StringUtils.isPalindrome("radar")); console.log(StringUtils.isPalindrome("hello"));

3. Static Factory Method

// Static methods as factory methods class User { constructor(name, email) { this.name = name; this.email = email; this.createdAt = new Date(); } // Factory method to create admin static createAdmin(name, email) { const user = new User(name, email); user.role = "admin"; return user; } // Factory method from JSON static fromJSON(json) { const data = JSON.parse(json); return new User(data.name, data.email); } } const admin = User.createAdmin("John", "john@admin.com"); const userFromJson = User.fromJSON('{"name": "Alice", "email": "alice@example.com"}'); console.log(admin); console.log(userFromJson);

4. Static vs Instance Methods

// Show difference between static and instance methods class MathHelper { // Static - belongs to class static square(n) { return n * n; } // Instance - belongs to each object constructor(value) { this.value = value; } // Instance method getSquare() { return this.value * this.value; } } // Static call - no object needed console.log("Static square(5):", MathHelper.square(5)); // Instance call - needs object const helper = new MathHelper(7); console.log("Instance getSquare():", helper.getSquare()); // Cannot use static on instance // helper.square(5) // Error

5. Static Methods in Inheritance

// Static methods are inherited but called on child class class Animal { static create(name) { return new this(name); // Polymorphic - uses child class } constructor(name) { this.name = name; } } class Dog extends Animal { static create(name, breed) { const dog = super.create(name); dog.breed = breed; return dog; } bark() { return `${this.name} says Woof!`; } } // Using inherited static method const dog1 = Animal.create("Max"); const dog2 = Dog.create("Buddy", "Labrador"); console.log(dog1); console.log(dog2); console.log(dog2.bark());
