Getters and Setters in JavaScript

Getters and setters allow you to define computed properties with custom getter and setter logic.

1. Basic Getter

// Getter to compute property values class Circle { constructor(radius) { this.radius = radius; } // Getter for computed diameter get diameter() { return this.radius * 2; } // Getter for computed area get area() { return Math.PI * this.radius * this.radius; } } const circle = new Circle(5); console.log("Radius:", circle.radius); console.log("Diameter:", circle.diameter); // Calls getter console.log("Area:", circle.area.toFixed(2)); // Calls getter

2. Basic Setter with Validation

// Setter to add validation logic class Temperature { constructor(celsius) { this._celsius = celsius; // Private convention } // Getter for celsius get celsius() { return this._celsius; } // Setter with validation set celsius(value) { if (typeof value !== 'number') { throw new Error("Temperature must be a number"); } this._celsius = value; } // Getter for fahrenheit get fahrenheit() { return (this._celsius * 9/5) + 32; } // Setter for fahrenheit set fahrenheit(value) { this._celsius = (value - 32) * 5/9; } } const temp = new Temperature(25); console.log("Celsius:", temp.celsius); console.log("Fahrenheit:", temp.fahrenheit); temp.celsius = 30; console.log("After setting celsius to 30:"); console.log("Celsius:", temp.celsius); console.log("Fahrenheit:", temp.fahrenheit);

3. Getter and Setter in Object Literals

// Using get/set in object literals const person = { _name: "John", _age: 25, // Getter get name() { return this._name.toUpperCase(); }, // Setter with validation set name(value) { if (value.length < 2) { console.log("Name too short"); return; } this._name = value; }, get age() { return this._age; }, set age(value) { if (value < 0 || value > 150) { console.log("Invalid age"); return; } this._age = value; } }; console.log(person.name); // Calls getter person.name = "Alice"; // Calls setter console.log(person.name); // Calls getter again person.age = -5; // Invalid - won't set console.log("Age:", person.age);

4. Getter for Read-only Properties

// Using getters to create read-only properties class User { constructor(username, email) { this.username = username; this.email = email; this._createdAt = new Date(); } // Read-only createdAt (no setter) get createdAt() { return this._createdAt.toLocaleDateString(); } // Computed property get profile() { return `@${this.username} - ${this.email}`; } get isNewUser() { const now = new Date(); const daysDiff = (now - this._createdAt) / (1000 * 60 * 60 * 24); return daysDiff < 7; } } const user = new User("johndoe", "john@example.com"); console.log("Username:", user.username); console.log("Created At:", user.createdAt); // Read-only console.log("Profile:", user.profile); console.log("Is New User:", user.isNewUser); // user.createdAt = "2024-01-01"; // Error - no setter

5. Getter/Setter for Clean API

// Using getters/setters to provide clean API class BankAccount { constructor(initialBalance) { this._balance = initialBalance; this._currency = "USD"; } // Getter - returns formatted balance get balance() { return `${this._currency} ${this._balance.toFixed(2)}`; } // Setter - accepts plain number set balance(amount) { if (amount < 0) { throw new Error("Balance cannot be negative"); } this._balance = amount; } get availableFunds() { return this._balance; } } const account = new BankAccount(1000); console.log("Balance:", account.balance); // "USD 1000.00" account.balance = 1500; // Set plain number console.log("New Balance:", account.balance); // "USD 1500.00" account.balance = -100; // Throws error console.log("Available:", account.availableFunds);
