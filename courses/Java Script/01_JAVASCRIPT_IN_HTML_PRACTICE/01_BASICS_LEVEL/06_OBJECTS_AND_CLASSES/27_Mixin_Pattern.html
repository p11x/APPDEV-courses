Mixin Pattern

Mixins allow sharing behavior between classes without inheritance.

1. Basic Mixin

const canWalk = { walk() { return "Walking..."; } }; const canSwim = { swim() { return "Swimming..."; } }; class Duck { constructor(name) { this.name = name; } } Object.assign(Duck.prototype, canWalk, canSwim); const duck = new Duck("Donald"); console.log(duck.walk(), duck.swim());

2. Function Mixin

function withLogging(Base) { return class extends Base { log() { console.log(`[${this.constructor.name}]: called`); } }; } class Simple {} const LoggedSimple = withLogging(Simple); const obj = new LoggedSimple(); obj.log();

3. Multiple Mixins

const canEat = { eat() { return "eating"; } }; const canSleep = { sleep() { return "sleeping"; } }; const canPlay = { play() { return "playing"; } }; class Person {} Object.assign(Person.prototype, canEat, canSleep, canPlay); const p = new Person(); console.log(p.eat(), p.sleep(), p.play());
