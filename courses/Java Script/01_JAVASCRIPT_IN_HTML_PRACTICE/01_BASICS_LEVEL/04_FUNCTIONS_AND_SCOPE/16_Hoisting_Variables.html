Hoisting - Variables

Variable declarations are hoisted to the top of their scope, but not their assignments.

var Hoisting

console.log(x); // undefined (not error)
var x = 5;
console.log(x); // 5

Examples

Example 1: Basic var Hoisting

document.getElementById("output1").innerHTML = "Before: " + typeof myVar;
var myVar = 5;
document.getElementById("output1").innerHTML += "<br>After: " + myVar;

Example 2: Hoisting with Initialization

var x = 10;

function test() {
    document.getElementById("output2").innerHTML = "Inside before: " + x;
    var x = 20;
    document.getElementById("output2").innerHTML += "<br>Inside after: " + x;
}

test();
document.getElementById("output2").innerHTML += "<br>Outside: " + x;

Example 3: Temporal Dead Zone (let/const)

// This causes ReferenceError with let
try {
    console.log(y);
    let y = 10;
} catch(e) {
    document.getElementById("output3").innerHTML = "Error: " + e.message;
}

let z = 20;
document.getElementById("output3").innerHTML += "<br>After declaration: " + z;

Example 4: let vs var in Loop

// var - shared across iterations
var result1 = "";
for (var i = 0; i < 3; i++) {
    setTimeout(function() { result1 += i + " "; }, 100);
}

// let - block scope
var result2 = "";
for (let j = 0; j < 3; j++) {
    setTimeout(function() { result2 += j + " "; }, 100);
}

setTimeout(function() {
    document.getElementById("output4").innerHTML = "var: " + result1 + "<br>let: " + result2;
}, 200);

Example 5: const Hoisting

// const also has TDZ
try {
    const PI;
    PI = 3.14;
} catch(e) {
    document.getElementById("output5").innerHTML = "Error: " + e.message;
}

const GRAVITY = 9.8;
document.getElementById("output5").innerHTML += "<br>Gravity: " + GRAVITY;

Key Points

  • var declarations are hoisted, initialization is not
  • let and const have Temporal Dead Zone - can't access before declaration
  • let has block scope, var has function scope
  • Use let/const to avoid hoisting issues
