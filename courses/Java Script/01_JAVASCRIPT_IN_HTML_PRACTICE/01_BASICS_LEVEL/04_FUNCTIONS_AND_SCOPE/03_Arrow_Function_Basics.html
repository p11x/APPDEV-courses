Arrow Function Basics

Arrow functions provide a shorter syntax for writing functions. They also handle this differently.

Syntax Comparison

// Traditional function
var add = function(a, b) {
    return a + b;
};

// Arrow function
var add = (a, b) => a + b;

Examples

Example 1: Basic Arrow Function

// Arrow function with body
var greet = (name) => {
    return "Hello, " + name + "!";
};

var message = greet("Arrow");
document.getElementById("output1").innerHTML = message;

Example 2: Single Expression (Implicit Return)

var square = (x) => x * x;

var result = square(6);
document.getElementById("output2").innerHTML = "6² = " + result;

Example 3: Single Parameter

// With parentheses - var double = (x) => x * 2;
// Without parentheses (single parameter)
var double = x => x * 2;

var result = double(8);
document.getElementById("output3").innerHTML = "Double of 8 = " + result;

Example 4: Multiple Parameters

var add = (a, b) => a + b;
var subtract = (a, b) => a - b;

document.getElementById("output4").innerHTML = 
    "10 + 3 = " + add(10, 3) + "
" +
    "10 - 3 = " + subtract(10, 3);

Example 5: Block Body

var calculate = (a, b) => {
    var sum = a + b;
    var product = a * b;
    return sum + " | " + product;
};

document.getElementById("output5").innerHTML = calculate(4, 5);

Example 6: With Array Methods

var numbers = [1, 2, 3, 4, 5];

var evens = numbers.filter(n => n % 2 === 0);
var tripled = numbers.map(n => n * 3);
var sum = numbers.reduce((acc, n) => acc + n, 0);

document.getElementById("output6").innerHTML = 
    "Original: " + numbers.join(", ") + "
" +
    "Evens: " + evens.join(", ") + "
" +
    "Tripled: " + tripled.join(", ") + "
" +
    "Sum: " + sum;

Example 7: Returning Object Literal

// Must wrap object in parentheses for implicit return
var createPerson = (name, age) => ({ name: name, age: age });

var person = createPerson("Alice", 25);
document.getElementById("output7").innerHTML = 
    "Name: " + person.name + ", Age: " + person.age;

Key Points

  • Syntax: (params) => expression or (params) => { statements; }
  • Single parameter can omit parentheses
  • Single expression has implicit return
  • Does not have its own this - inherits from parent scope
  • Cannot be used as constructors (no new)
