Side Effects

A side effect is any observable change to the state outside the function (modifying global variables, DOM, I/O, etc.)

Common Side Effects

Modifying global variables, DOM manipulation, console.log(), 
network requests, reading/writing files, throwing errors

Examples

Example 1: Global Variable Modification

var counter = 0;

function increment() {
    counter++; // Side effect: modifies global variable
    return counter;
}

increment();
increment();
document.getElementById("output1").innerHTML = "Counter: " + counter;

Example 2: Console Output

function logMessage(message) {
    console.log(message); // Side effect: I/O operation
    return "Done";
}

document.getElementById("output2").innerHTML = logMessage("Hello!");

Example 3: DOM Manipulation

function updateDOM(elementId, value) {
    document.getElementById(elementId).innerHTML = value; // Side effect
    return "Updated";
}

document.getElementById("output3").innerHTML = updateDOM("output3", "Modified DOM!");

Example 4: Separating Pure and Impure

// Pure function - no side effects
function calculateTotal(items) {
    return items.reduce((sum, item) => sum + item.price, 0);
}

// Impure function - side effect
function displayTotal(total) {
    document.getElementById("output4").innerHTML = "Total: $" + total;
}

var items = [{price: 10}, {price: 20}, {price: 30}];
var total = calculateTotal(items);
displayTotal(total);

document.getElementById("output4").innerHTML = "Total: $" + total + " (computed purely, displayed impurely)";

Example 5: Pure vs Impure Functions

// PURE - same input, same output, no side effects
function pureAdd(a, b) { return a + b; }

// IMPURE - depends on external state
var x = 5;
function impureAdd(a) { return a + x; }

// PURE - creates new array
function pureFilter(arr, min) { return arr.filter(n => n > min); }

// IMPURE - modifies original
function impureFilter(arr, min) {
    for (var i = 0; i < arr.length; i++) {
        if (arr[i] <= min) arr.splice(i, 1);
    }
    return arr;
}

document.getElementById("output5").innerHTML = 
    "Pure add(2,3) = " + pureAdd(2, 3) + "<br>" +
    "Pure filter [1,2,3,4] > 2: " + pureFilter([1,2,3,4], 2);

Key Points

  • Side effects change state outside the function
  • Common side effects: global vars, DOM, console, network
  • Separate pure logic from impure actions
  • Minimize side effects for better code quality
