Function Practice 1

Practice exercises combining functions, callbacks, and closures.

Exercises

Exercise 1: Using Callbacks with Arrays

var numbers = [1, 2, 3, 4, 5];

function processArray(arr, callback) {
    return arr.map(callback);
}

var squared = processArray(numbers, function(n) { return n * n; });
document.getElementById("output1").innerHTML = "Squared: " + squared.join(", ");

Exercise 2: Closure for Private State

function createStore(initial) {
    var balance = initial;
    
    return {
        getBalance: function() { return balance; },
        deposit: function(amount) { balance += amount; return balance; },
        withdraw: function(amount) { 
            if (amount <= balance) { balance -= amount; return balance; }
            return "Insufficient";
        }
    };
}

var store = createStore(100);
store.deposit(50);
store.withdraw(30);
document.getElementById("output2").innerHTML = "Balance: $" + store.getBalance();

Exercise 3: Higher-Order Function

function createMultiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

var double = createMultiplier(2);
var triple = createMultiplier(3);
var quadruple = createMultiplier(4);

document.getElementById("output3").innerHTML = 
    "double(5) = " + double(5) + "<br>" +
    "triple(5) = " + triple(5) + "<br>" +
    "quadruple(5) = " + quadruple(5);

Exercise 4: Callback with Filter

var products = [
    { name: "Laptop", price: 1000 },
    { name: "Mouse", price: 25 },
    { name: "Keyboard", price: 75 },
    { name: "Monitor", price: 300 }
];

function filterByPrice(items, minPrice) {
    return items.filter(function(item) {
        return item.price >= minPrice;
    });
}

var affordable = filterByPrice(products, 50);
var names = affordable.map(function(p) { return p.name; }).join(", ");
document.getElementById("output4").innerHTML = "Affordable (>$50): " + names;

Exercise 5: Multiple Closures

function createCounter() {
    var count = 0;
    return {
        increment: function() { count++; return count; },
        decrement: function() { count--; return count; },
        reset: function() { count = 0; return count; }
    };
}

var counter1 = createCounter();
var counter2 = createCounter();

counter1.increment();
counter1.increment();
counter2.increment();
counter2.increment();
counter2.increment();

document.getElementById("output5").innerHTML = 
    "Counter1: " + counter1.increment() + "<br>" +
    "Counter2: " + counter2.increment();

Key Points

  • Callbacks allow passing behavior to functions
  • Closures create private variables
  • Higher-order functions return functions
  • Combine these patterns for flexible code
