Recursion - Factorial

Factorial is the product of all positive integers up to a given number. Written as n! = n × (n-1) × ... × 1

Factorial Formula

n! = n × (n-1)!
Base case: 0! = 1

Examples

Example 1: Basic Factorial

function factorial(n) {
    if (n === 0 || n === 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

document.getElementById("output1").innerHTML = "5! = " + factorial(5);

Example 2: Calculate Multiple Factorials

function factorial(n) {
    if (n === 0 || n === 1) return 1;
    return n * factorial(n - 1);
}

document.getElementById("output2").innerHTML = 
    "0! = " + factorial(0) + "<br>" +
    "1! = " + factorial(1) + "<br>" +
    "2! = " + factorial(2) + "<br>" +
    "3! = " + factorial(3) + "<br>" +
    "4! = " + factorial(4);

Example 3: Step by Step Calculation

function factorialWithSteps(n) {
    if (n === 0 || n === 1) {
        return { result: 1, steps: "Base case: 1" };
    }
    var sub = factorialWithSteps(n - 1);
    return {
        result: n * sub.result,
        steps: n + " × (" + sub.steps + ")"
    };
}

var result = factorialWithSteps(5);
document.getElementById("output3").innerHTML = "5! = " + result.result + "<br>Steps: " + result.steps;

Example 5: Iterative Comparison

function factorialRecursive(n) {
    if (n === 0 || n === 1) return 1;
    return n * factorialRecursive(n - 1);
}

function factorialIterative(n) {
    var result = 1;
    for (var i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

document.getElementById("output5").innerHTML = 
    "Recursive 5! = " + factorialRecursive(5) + "<br>" +
    "Iterative 5! = " + factorialIterative(5);

Key Points

  • Factorial: n! = n × (n-1) × ... × 1
  • Base case: 0! = 1 and 1! = 1
  • Recursive approach calls itself with n-1
  • Can also be solved iteratively
