Default Parameters

Default parameters allow specifying default values for parameters when no argument is provided.

Syntax

function functionName(param = defaultValue) {
    // function body
}

Examples

Example 1: Simple Default Parameter

function greet(name = "Guest") {
    return "Hello, " + name + "!";
}

document.getElementById("output1").innerHTML = 
    greet("Alice") + "
" +
    greet();

Example 2: Multiple Default Parameters

function createUser(name = "Anonymous", age = 0, country = "Unknown") {
    return {
        name: name,
        age: age,
        country: country
    };
}

var user1 = createUser();
var user2 = createUser("Bob", 25);

document.getElementById("output2").innerHTML = 
    "User 1: " + JSON.stringify(user1) + "
" +
    "User 2: " + JSON.stringify(user2);

Example 3: Default with Expression

function calculatePrice(price, taxRate = 0.1, discount = price * 0) {
    return price + (price * taxRate) - discount;
}

var price1 = calculatePrice(100);
var price2 = calculatePrice(100, 0.2);
var price3 = calculatePrice(100, 0.2, 10);

document.getElementById("output3").innerHTML = 
    "Price 1: $" + price1 + "
" +
    "Price 2: $" + price2 + "
" +
    "Price 3: $" + price3;

Example 4: Using undefined

function setConfig(config = "default") {
    return "Config: " + config;
}

document.getElementById("output4").innerHTML = 
    setConfig("custom") + "
" +
    setConfig(undefined) + "
" +
    setConfig();

Example 5: Arrow Function with Defaults

var multiply = (a, b = 1) => a * b;

document.getElementById("output5").innerHTML = 
    "multiply(5): " + multiply(5) + "
" +
    "multiply(5, 3): " + multiply(5, 3);

Example 6: Parameter Order

// Default parameters should come after required ones
function createPost(title, content = "", category = "General") {
    return { title: title, content: content, category: category };
}

var post = createPost("My Post");

document.getElementById("output6").innerHTML = JSON.stringify(post);

Key Points

  • Syntax: param = defaultValue
  • Used when argument is undefined or not provided
  • Expressions in defaults are evaluated left to right
  • Works with all function types (declarations, expressions, arrow)
