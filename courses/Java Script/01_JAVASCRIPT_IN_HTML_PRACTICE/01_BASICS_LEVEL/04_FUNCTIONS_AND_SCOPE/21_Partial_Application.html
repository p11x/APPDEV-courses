Partial Application

Partial application fixes some arguments of a function, returning a new function that takes the remaining arguments.

Partial Application Syntax

function partial(fn, ...fixedArgs) {
    return function(...remainingArgs) {
        return fn(...fixedArgs, ...remainingArgs);
    };
}

Examples

Example 1: Using bind()

function multiply(a, b, c) {
    return a * b * c;
}

var partiallyApplied = multiply.bind(null, 2);
var result = partiallyApplied(3, 4);

document.getElementById("output1").innerHTML = "2 * 3 * 4 = " + result;

Example 2: Manual Partial Function

function partial(fn, fixedArgs) {
    return function(remainingArgs) {
        return fn(fixedArgs, remainingArgs);
    };
}

function add(a, b) { return a + b; }

var add5 = partial(add, 5);
document.getElementById("output2").innerHTML = "5 + 3 = " + add5(3);

Example 3: Partial with Multiple Fixed Args

function createMessage(greeting, name, punctuation) {
    return greeting + ", " + name + punctuation;
}

var sayHello = createMessage.bind(null, "Hello", "John");
var sayGoodbye = createMessage.bind(null, "Goodbye", "Jane");

document.getElementById("output3").innerHTML = 
    sayHello("!") + "<br>" +
    sayGoodbye("...");

Example 4: Currying vs Partial Application

// Currying - each arg gets its own function
var curriedAdd = a => b => a + b;
var add5 = curriedAdd(5);

// Partial - fix multiple args at once
function subtract(a, b, c) { return a - b - c; }
var subFrom100 = subtract.bind(null, 100);

document.getElementById("output4").innerHTML = 
    "Curried: 5 + 3 = " + add5(3) + "<br>" +
    "Partial: 100 - 20 - 5 = " + subFrom100(20, 5);

Example 5: Practical Example

function formatCurrency(symbol, amount) {
    return symbol + amount.toFixed(2);
}

var formatUSD = formatCurrency.bind(null, "$");
var formatEUR = formatCurrency.bind(null, "€");

document.getElementById("output5").innerHTML = 
    formatUSD(100.5) + "<br>" +
    formatEUR(250.75);

Key Points

  • Partial application fixes some arguments of a function
  • bind() is commonly used for partial application
  • Currying creates nested single-arg functions
  • Partial can fix multiple args at once
