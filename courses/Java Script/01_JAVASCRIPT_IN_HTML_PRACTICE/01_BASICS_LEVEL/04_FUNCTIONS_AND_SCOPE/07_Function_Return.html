Function Return

The return statement specifies the value a function outputs.

Syntax

function functionName() {
    return value;
}

Examples

Example 1: Basic Return

function add(a, b) {
    return a + b;
}

var result = add(3, 4);
document.getElementById("output1").innerHTML = "3 + 4 = " + result;

Example 2: Return Object

function createUser(name, age) {
    return {
        name: name,
        age: age,
        created: new Date()
    };
}

var user = createUser("Alice", 25);
document.getElementById("output2").innerHTML = JSON.stringify(user);

Example 3: Return Undefined

function greet(name) {
    if (!name) {
        return; // returns undefined
    }
    return "Hello, " + name + "!";
}

document.getElementById("output3").innerHTML = 
    greet("Bob") + "
" +
    greet() + "
" +
    greet(null);

Example 4: Return Ends Execution

function findFirstEven(numbers) {
    for (var i = 0; i < numbers.length; i++) {
        if (numbers[i] % 2 === 0) {
            return "First even: " + numbers[i];
        }
    }
    return "No even found";
}

document.getElementById("output4").innerHTML = 
    findFirstEven([1, 3, 5, 6, 7]) + "
" +
    findFirstEven([1, 3, 5, 7]);

Example 5: Early Return Pattern

function divide(a, b) {
    if (b === 0) {
        return "Error: Cannot divide by zero";
    }
    return a / b;
}

document.getElementById("output5").innerHTML = 
    "10 / 2 = " + divide(10, 2) + "
" +
    "10 / 0 = " + divide(10, 0);

Example 6: Return Function

function createMultiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

var double = createMultiplier(2);
var triple = createMultiplier(3);

document.getElementById("output6").innerHTML = 
    "Double 5: " + double(5) + "
" +
    "Triple 5: " + triple(5);

Example 7: Conditional Return

function getDiscount(student, senior) {
    if (student) return 0.5;
    if (senior) return 0.3;
    return 0;
}

document.getElementById("output7").innerHTML = 
    "Student: " + getDiscount(true, false) * 100 + "%
" +
    "Senior: " + getDiscount(false, true) * 100 + "%
" +
    "Regular: " + getDiscount(false, false) * 100 + "%";

Key Points

  • return specifies the function's output value
  • Functions without return return undefined
  • return ends function execution immediately
  • The expression after return is evaluated
