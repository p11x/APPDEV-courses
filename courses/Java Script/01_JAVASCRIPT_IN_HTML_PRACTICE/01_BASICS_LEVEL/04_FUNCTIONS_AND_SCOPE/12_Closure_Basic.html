Closure - Basic

A closure is a function that has access to variables from its outer (enclosing) scope, even after the outer function has returned.

What is a Closure?

function outerFunction() {
    var outerVariable = "I am from outer";
    
    return function innerFunction() {
        // This inner function "closes over" outerVariable
        return outerVariable;
    };
}

Examples

Example 1: Basic Closure

function greet(name) {
    var message = "Hello, " + name;
    return function() {
        return message;
    };
}

var greeting = greet("World");
document.getElementById("output1").innerHTML = greeting();

Example 2: Lexical Scope

function outer() {
    var x = 10;
    
    function middle() {
        var y = 20;
        
        function inner() {
            return x + y; // Can access both x and y
        }
        
        return inner();
    }
    
    return middle();
}

document.getElementById("output2").innerHTML = "Result: " + outer();

Example 3: Returning Function that Retains Scope

function multiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

var double = multiplier(2);
var triple = multiplier(3);

document.getElementById("output3").innerHTML = "Double 5: " + double(5) + "<br>Triple 5: " + triple(5);

Example 4: Simple Closure

function createAdder(x) {
    return function(y) {
        return x + y;
    };
}

var add5 = createAdder(5);
var add10 = createAdder(10);

document.getElementById("output4").innerHTML = "5 + 3 = " + add5(3) + "<br>10 + 7 = " + add10(7);

Example 5: Closure with Multiple Variables

function person(name, age) {
    return function() {
        return name + " is " + age + " years old";
    };
}

var info = person("John", 25);
document.getElementById("output5").innerHTML = info();

Key Points

  • A closure gives access to an outer function's scope
  • Lexical scope means the function's environment where it was defined
  • Returned functions retain access to their outer variables
  • Closures are created every time a function is defined inside another function
