Recursion - Fibonacci

The Fibonacci sequence: 0, 1, 1, 2, 3, 5, 8, 13, 21... Each number is the sum of the two preceding ones.

Fibonacci Formula

fib(n) = fib(n-1) + fib(n-2)
Base cases: fib(0) = 0, fib(1) = 1

Examples

Example 1: Basic Fibonacci

function fibonacci(n) {
    if (n === 0) return 0;
    if (n === 1) return 1;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

document.getElementById("output1").innerHTML = "fib(7) = " + fibonacci(7);

Example 2: Generate Fibonacci Sequence

function fibonacci(n) {
    if (n === 0) return 0;
    if (n === 1) return 1;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

var sequence = "";
for (var i = 0; i < 10; i++) {
    sequence += fibonacci(i) + (i < 9 ? ", " : "");
}

document.getElementById("output2").innerHTML = "First 10: " + sequence;

Example 3: Memoization (Optimization)

function fibonacciMemo(n, memo) {
    memo = memo || {};
    if (memo[n]) return memo[n];
    if (n === 0) return 0;
    if (n === 1) return 1;
    memo[n] = fibonacciMemo(n - 1, memo) + fibonacciMemo(n - 2, memo);
    return memo[n];
}

document.getElementById("output3").innerHTML = 
    "fib(20) = " + fibonacciMemo(20) + "<br>fib(30) = " + fibonacciMemo(30);

Example 4: Iterative Fibonacci

function fibonacciIterative(n) {
    if (n === 0) return 0;
    if (n === 1) return 1;
    
    var a = 0, b = 1;
    for (var i = 2; i <= n; i++) {
        var temp = a + b;
        a = b;
        b = temp;
    }
    return b;
}

document.getElementById("output4").innerHTML = 
    "Recursive fib(10) = " + fibonacci(10) + "<br>" +
    "Iterative fib(10) = " + fibonacciIterative(10);

Key Points

  • Fibonacci: each number is sum of previous two
  • Base cases: fib(0)=0, fib(1)=1
  • Simple recursion is inefficient for large n
  • Memoization or iteration can optimize
