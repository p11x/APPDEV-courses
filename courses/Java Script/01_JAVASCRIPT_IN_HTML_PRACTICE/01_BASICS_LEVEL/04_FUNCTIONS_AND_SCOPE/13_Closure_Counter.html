Closure - Counter

Closures can create private variables and maintain state. This is useful for creating counters.

Creating Counter with Closures

function createCounter() {
    var count = 0; // private variable
    
    return {
        increment: function() { count++; },
        decrement: function() { count--; },
        getCount: function() { return count; }
    };
}

Examples

Example 1: Basic Counter

function createCounter() {
    var count = 0;
    
    return {
        getCount: function() {
            return count;
        },
        increment: function() {
            count++;
        }
    };
}

var counter = createCounter();
counter.increment();
counter.increment();
counter.increment();

document.getElementById("output1").innerHTML = "Count: " + counter.getCount();

Example 2: Increment and Decrement

function createCounter() {
    var count = 0;
    
    return {
        increment: function() {
            count++;
            return count;
        },
        decrement: function() {
            count--;
            return count;
        },
        getCount: function() {
            return count;
        }
    };
}

var counter = createCounter();
counter.increment();
counter.increment();
counter.decrement();
counter.increment();

document.getElementById("output2").innerHTML = "Final count: " + counter.getCount();

Example 3: Multiple Independent Counters

function createCounter() {
    var count = 0;
    return {
        increment: function() { count++; },
        getCount: function() { return count; }
    };
}

var counter1 = createCounter();
var counter2 = createCounter();

counter1.increment();
counter1.increment();
counter1.increment();
counter2.increment();

document.getElementById("output3").innerHTML = "Counter1: " + counter1.getCount() + "<br>Counter2: " + counter2.getCount();

Example 4: Counter with Reset

function createCounter(initial) {
    var count = initial || 0;
    
    return {
        increment: function() { count++; },
        decrement: function() { count--; },
        reset: function() { count = 0; },
        getCount: function() { return count; }
    };
}

var counter = createCounter(10);
counter.increment();
counter.increment();

document.getElementById("output4").innerHTML = "After 2 increments from 10: " + counter.getCount() + "<br>After reset: " + (counter.reset() || counter.getCount());

Example 5: Private Variables

function createBankAccount() {
    var balance = 100; // private
    
    return {
        deposit: function(amount) {
            balance += amount;
            return balance;
        },
        withdraw: function(amount) {
            if (amount <= balance) {
                balance -= amount;
                return balance;
            }
            return "Insufficient funds";
        },
        getBalance: function() {
            return balance;
        }
    };
}

var account = createBankAccount();
account.deposit(50);
account.withdraw(30);

document.getElementById("output5").innerHTML = "Balance: $" + account.getBalance();

Key Points

  • Closures create private variables that can't be accessed directly
  • Each counter has its own independent count variable
  • Only the returned functions have access to the private variable
  • Useful for encapsulation and data privacy
