Factory Pattern

Factory functions create and return objects without using the 'new' keyword or 'this' keyword.

Factory Function Syntax

function createObject(param) {
    return {
        property: param,
        method: function() { ... }
    };
}

Examples

Example 1: Basic Factory Function

function createPerson(name, age) {
    return {
        name: name,
        age: age,
        greet: function() {
            return "Hello, I'm " + this.name;
        }
    };
}

var person1 = createPerson("John", 25);
var person2 = createPerson("Jane", 30);

document.getElementById("output1").innerHTML = 
    person1.greet() + "<br>" +
    person2.greet();

Example 2: Factory with Multiple Methods

function createBankAccount(initialBalance) {
    var balance = initialBalance;
    
    return {
        deposit: function(amount) {
            balance += amount;
            return balance;
        },
        withdraw: function(amount) {
            if (amount <= balance) {
                balance -= amount;
                return balance;
            }
            return "Insufficient funds";
        },
        getBalance: function() {
            return balance;
        }
    };
}

var account = createBankAccount(100);
account.deposit(50);
account.withdraw(30);
document.getElementById("output2").innerHTML = "Balance: $" + account.getBalance();

Example 3: Factory vs Constructor

// Factory (no 'new' needed)
function createPoint(x, y) {
    return { x: x, y: y };
}

// Constructor (requires 'new')
function Point(x, y) {
    this.x = x;
    this.y = y;
}

var p1 = createPoint(1, 2);
var p2 = new Point(3, 4);

document.getElementById("output3").innerHTML = 
    "Factory: (" + p1.x + ", " + p1.y + ")<br>" +
    "Constructor: (" + p2.x + ", " + p2.y + ")";

Example 4: Factory with Validation

function createProduct(name, price) {
    if (!name || price < 0) {
        return null;
    }
    
    return {
        name: name,
        price: price,
        getInfo: function() {
            return this.name + ": $" + this.price.toFixed(2);
        },
        applyDiscount: function(discount) {
            return this.price * (1 - discount);
        }
    };
}

var product = createProduct("Laptop", 999.99);
var invalid = createProduct("", -5);

document.getElementById("output4").innerHTML = 
    "Valid: " + product.getInfo() + "<br>" +
    "Invalid product: " + invalid;

Example 5: Creating Multiple Objects

function createCar(brand, model, year) {
    return {
        brand: brand,
        model: model,
        year: year,
        getDescription: function() {
            return this.year + " " + this.brand + " " + this.model;
        },
        age: function(currentYear) {
            return currentYear - this.year;
        }
    };
}

var cars = [
    createCar("Toyota", "Camry", 2020),
    createCar("Honda", "Civic", 2018),
    createCar("Tesla", "Model 3", 2022)
];

var result = "";
cars.forEach(function(car) {
    result += car.getDescription() + " (Age: " + car.age(2024) + ")<br>";
});

document.getElementById("output5").innerHTML = result;

Key Points

  • Factory functions create objects without 'new' keyword
  • No 'this' keyword needed
  • Can encapsulate private variables with closures
  • Simpler than constructor pattern for simple objects
