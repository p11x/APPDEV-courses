31. Function Currying Advanced

Exercise 1: Curried Math Operations

const add = a => b => a + b;
const multiply = a => b => a * b;
const divide = a => b => a / b;

Exercise 2: Curried Comparison

const greaterThan = a => b => b > a;
const lessThan = a => b => b < a;
const equals = a => b => b === a;

Exercise 3: Curried Array Filter

const filterBy = condition => arr => arr.filter(condition);
const greaterThan10 = filterBy(greaterThan(10));
const lessThan5 = filterBy(lessThan(5));

Exercise 4: Curried Object Mapper

const mapBy = prop => arr => arr.map(item => item[prop]);
const getNames = mapBy('name');
const getAges = mapBy('age');

Exercise 5: Curried Function Composition

const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x);
const compose = (...fns) => x => fns.reduceRight((v, f) => f(v), x);
