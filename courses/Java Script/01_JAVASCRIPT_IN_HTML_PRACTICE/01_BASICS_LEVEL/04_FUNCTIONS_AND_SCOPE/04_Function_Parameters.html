Function Parameters

Parameters are variables in function definitions that accept input values.

Syntax

function functionName(param1, param2) {
    // use param1 and param2
}

Examples

Example 1: Single Parameter

function square(num) {
    return num * num;
}

var result = square(7);
document.getElementById("output1").innerHTML = "7² = " + result;

Example 2: Multiple Parameters

function fullName(firstName, lastName) {
    return firstName + " " + lastName;
}

var name = fullName("John", "Doe");
document.getElementById("output2").innerHTML = name;

Example 3: No Parameters

function getRandomNumber() {
    return Math.floor(Math.random() * 100);
}

var random = getRandomNumber();
document.getElementById("output3").innerHTML = "Random number: " + random;

Example 4: Parameter Type Flexibility

function greet(person) {
    if (typeof person === "string") {
        return "Hello, " + person + "!";
    } else if (typeof person === "object") {
        return "Hello, " + person.name + "!";
    }
    return "Hello!";
}

var result1 = greet("World");
var result2 = greet({ name: "Alice" });
var result3 = greet();

document.getElementById("output4").innerHTML = 
    result1 + "
" + result2 + "
" + result3;

Example 5: Arrays as Parameters

function sumArray(numbers) {
    var total = 0;
    for (var i = 0; i < numbers.length; i++) {
        total += numbers[i];
    }
    return total;
}

var numbers = [1, 2, 3, 4, 5];
var result = sumArray(numbers);
document.getElementById("output5").innerHTML = "Sum: " + result;

Example 6: Functions as Parameters

function applyOperation(a, b, operation) {
    return operation(a, b);
}

var add = (a, b) => a + b;
var multiply = (a, b) => a * b;

document.getElementById("output6").innerHTML = 
    "Apply add: " + applyOperation(5, 3, add) + "
" +
    "Apply multiply: " + applyOperation(5, 3, multiply);

Example 7: Undefined Parameters

function describe(name, age, city) {
    var description = name;
    if (age !== undefined) {
        description += " is " + age;
    }
    if (city !== undefined) {
        description += " from " + city;
    }
    return description;
}

document.getElementById("output7").innerHTML = 
    describe("Alice", 25, "NYC") + "
" +
    describe("Bob") + "
" +
    describe("Charlie", 30);

Key Points

  • Parameters are defined in function parentheses
  • Can have zero or more parameters
  • Can accept any data type as argument
  • Arguments can be values, variables, or expressions
