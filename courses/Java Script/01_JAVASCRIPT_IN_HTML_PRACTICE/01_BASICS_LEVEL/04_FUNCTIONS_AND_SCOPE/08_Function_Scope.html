Function Scope

Scope determines where variables are accessible. Functions create their own scope.

Scope Levels

var global = "I am global";        // Global scope

function myFunction() {
    var functionScope = "I am in function"; // Function scope
    
    if (true) {
        var blockScope = "I am in block"; // Block scope (var)
    }
}

Examples

Example 1: Global vs Local Scope

var globalVar = "I am global";

function testScope() {
    var localVar = "I am local";
    return "Global: " + globalVar + ", Local: " + localVar;
}

document.getElementById("output1").innerHTML = testScope() + "
" +
    "Global only: " + globalVar + "
" +
    "Local only: " + (typeof localVar === "undefined" ? "undefined" : localVar);

Example 2: Inner Function Scope

var outerVar = "Outer";

function outerFunction() {
    var outerVar = "Outer (function)";
    
    function innerFunction() {
        var innerVar = "Inner";
        return innerVar + " - " + outerVar;
    }
    
    return innerFunction();
}

document.getElementById("output2").innerHTML = outerFunction();

Example 3: Scope Chain

var a = 1;

function level1() {
    var a = 2;
    
    function level2() {
        var a = 3;
        return "a = " + a;
    }
    
    return level2();
}

document.getElementById("output3").innerHTML = 
    "Global a: " + a + "
" +
    level1();

Example 4: Block Scope (var)

function testBlock() {
    if (true) {
        var blockVar = "I am in block";
    }
    
    return "Accessible: " + blockVar;
}

document.getElementById("output4").innerHTML = testBlock();

Example 5: Modifying Global Variable

var counter = 0;

function increment() {
    counter++;
    return counter;
}

function incrementLocal() {
    var counter = 0;
    counter++;
    return counter;
}

document.getElementById("output5").innerHTML = 
    "Global counter: " + increment() + "
" +
    "Global counter: " + increment() + "
" +
    "Local counter: " + incrementLocal() + "
" +
    "Global counter: " + counter;

Example 6: Closure - Private Variable

function createCounter() {
    var count = 0;
    
    return function() {
        count++;
        return count;
    };
}

var counter1 = createCounter();
var counter2 = createCounter();

document.getElementById("output6").innerHTML = 
    "Counter1: " + counter1() + "
" +
    "Counter1: " + counter1() + "
" +
    "Counter1: " + counter1() + "
" +
    "Counter2: " + counter2() + "
" +
    "Counter2: " + counter2();

Example 7: Variable Hoisting

function testHoisting() {
    console.log("Before: " + myVar);
    
    var myVar = "Defined";
    
    console.log("After: " + myVar);
    return "Done";
}

testHoisting();

document.getElementById("output7").innerHTML = "Check console for hoisting demo";

Key Points

  • Variables declared with var are function-scoped
  • Variables declared with let/const are block-scoped
  • Inner functions have access to outer scope variables
  • Closures allow functions to retain access to outer scope
