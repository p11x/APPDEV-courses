Exercises

Exercise 1: Recursive Sum of Digits

function sumDigits(n) {
    n = Math.abs(n);
    if (n < 10) return n;
    return (n % 10) + sumDigits(Math.floor(n / 10));
}

document.getElementById("output1").innerHTML = 
    "sumDigits(123) = " + sumDigits(123) + "<br>" +
    "sumDigits(9999) = " + sumDigits(9999);

Exercise 2: Curried Power Function

function power(base) {
    return function(exp) {
        return Math.pow(base, exp);
    };
}

var powerOf2 = power(2);
var powerOf3 = power(3);

document.getElementById("output2").innerHTML = 
    "2^5 = " + powerOf2(5) + "<br>" +
    "3^4 = " + powerOf3(4) + "<br>" +
    "2^10 = " + power(2)(10);

Exercise 3: Function Composition

function compose(f, g) {
    return function(x) { return f(g(x)); };
}

function trim(s) { return s.trim(); }
function upper(s) { return s.toUpperCase(); }
function exclaim(s) { return s + "!"; }

var processString = compose(exclaim, compose(upper, trim));

document.getElementById("output3").innerHTML = processString("  hello world  ");

Exercise 4: Recursive Countdown

function countdown(n) {
    if (n < 0) return [];
    return [n].concat(countdown(n - 1));
}

document.getElementById("output4").innerHTML = 
    "countdown(5): " + countdown(5).join(", ");

Exercise 5: Curried String Formatter

function format(prefix) {
    return function(suffix) {
        return function(text) {
            return prefix + text + suffix;
        };
    };
}

var quote = format('"')('"');
var parentheses = format('(')(')');
var brackets = format('[')(']');

document.getElementById("output5").innerHTML = 
    quote("Hello") + "<br>" +
    parentheses("World") + "<br>" +
    brackets("Test");

Key Points

  • Recursion breaks problems into smaller subproblems
  • Currying creates specialized functions from general ones
  • Composition combines simple functions
  • Practice combining these patterns
