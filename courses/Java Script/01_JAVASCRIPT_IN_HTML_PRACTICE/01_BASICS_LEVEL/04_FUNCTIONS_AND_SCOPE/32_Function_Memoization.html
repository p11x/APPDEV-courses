32. Function Memoization

Exercise 1: Basic Memoization

function memoize(fn) {
    const cache = {};
    return function(...args) {
        const key = JSON.stringify(args);
        if (cache[key]) return cache[key];
        cache[key] = fn.apply(this, args);
        return cache[key];
    };
}

Exercise 2: Fibonacci with Memoization

function fibMemo(n, cache = {}) {
    if (n in cache) return cache[n];
    if (n <= 1) return n;
    cache[n] = fibMemo(n - 1, cache) + fibMemo(n - 2, cache);
    return cache[n];
}

Exercise 3: Factorial with Memoization

const memoize = fn => {
    const cache = {};
    return (...args) => {
        const key = args.join(',');
        if (key in cache) return cache[key];
        return cache[key] = fn(...args);
    };
};

const fact = memoize(n => n <= 1 ? 1 : n * fact(n - 1));

Exercise 5: Recursive DP Pattern

function countPaths(m, n, memo = {}) {
    const key = m + ',' + n;
    if (key in memo) return memo[key];
    if (m === 0 || n === 0) return 0;
    if (m === 1 && n === 1) return 1;
    memo[key] = countPaths(m - 1, n, memo) + countPaths(m, n - 1, memo);
    return memo[key];
}
