Function Composition

Function composition combines simple functions to build more complex operations.

Composition Syntax

// Compose: right-to-left execution
compose(f, g)(x) = f(g(x))

// Pipe: left-to-right execution
pipe(f, g)(x) = g(f(x))

Examples

Example 1: Basic Compose

function compose(f, g) {
    return function(x) {
        return f(g(x));
    };
}

function addOne(x) { return x + 1; }
function double(x) { return x * 2; }

var addOneThenDouble = compose(double, addOne);
document.getElementById("output1").innerHTML = "(5 + 1) * 2 = " + addOneThenDouble(5);

Example 2: Pipe (Left to Right)

function pipe(...fns) {
    return function(x) {
        return fns.reduce((acc, fn) => fn(acc), x);
    };
}

function addOne(x) { return x + 1; }
function square(x) { return x * x; }
function subtractTen(x) { return x - 10; }

var process = pipe(addOne, square, subtractTen);
document.getElementById("output2").innerHTML = "((3 + 1)² - 10) = " + process(3);

Example 3: Real World Example

function trim(str) { return str.trim(); }
function toUpper(str) { return str.toUpperCase(); }
function addExclamation(str) { return str + "!"; }

var formatText = pipe(trim, toUpper, addExclamation);
document.getElementById("output3").innerHTML = formatText("  hello world  ");

Example 4: Compose with Array Methods

function compose(...fns) {
    return function(x) {
        return fns.reduceRight((acc, fn) => fn(acc), x);
    };
}

var numbers = [1, 2, 3, 4, 5];

var processNumbers = compose(
    arr => arr.filter(n => n > 2),
    arr => arr.map(n => n * 2),
    arr => arr.reduce((a, b) => a + b, 0)
);

document.getElementById("output4").innerHTML = "Sum of doubled >2: " + processNumbers(numbers);

Example 5: Chaining Multiple Functions

function pipe(...fns) {
    return x => fns.reduce((v, f) => f(v), x);
}

function multiplyBy2(x) { return x * 2; }
function add10(x) { return x + 10; }
function subtract5(x) { return x - 5; }
function square(x) { return x * x; }

var complexCalc = pipe(multiplyBy2, add10, subtract5, square);
document.getElementById("output5").innerHTML = "((3 * 2 + 10 - 5)²) = " + complexCalc(3);

Key Points

  • Composition combines simple functions to build complex logic
  • Compose executes right-to-left, pipe executes left-to-right
  • Each function should do one thing well
  • Makes code more readable and reusable
