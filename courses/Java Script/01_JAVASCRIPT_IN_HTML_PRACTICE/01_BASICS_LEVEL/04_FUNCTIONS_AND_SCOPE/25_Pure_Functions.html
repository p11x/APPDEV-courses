Pure Functions

A pure function always returns the same output for the same input and has no side effects.

Pure Function Rules

1. Same input → Same output
2. No side effects (no modifying external state)

Examples

Example 1: Pure Function

function add(a, b) {
    return a + b;
}

document.getElementById("output1").innerHTML = 
    "add(2, 3) = " + add(2, 3) + "<br>" +
    "add(2, 3) = " + add(2, 3) + " (always same!)";

Example 2: Impure Function

var total = 0;

function addToTotal(value) {
    total += value; // Modifies external variable!
    return total;
}

addToTotal(5);
addToTotal(10);
document.getElementById("output2").innerHTML = "Total: " + total;

Example 3: Same Input, Different Output

function getRandom(max) {
    return Math.floor(Math.random() * max) + 1;
}

document.getElementById("output3").innerHTML = 
    "getRandom(10): " + getRandom(10) + "<br>" +
    "getRandom(10): " + getRandom(10) + "<br>" +
    "(Not pure - different outputs!)";

Example 4: Pure Array Function

function doubleArray(arr) {
    return arr.map(function(n) {
        return n * 2;
    });
}

var original = [1, 2, 3];
var doubled = doubleArray(original);

document.getElementById("output4").innerHTML = 
    "Original: " + original.join(", ") + "<br>" +
    "Doubled: " + doubled.join(", ") + "<br>" +
    "(Original unchanged - pure!)";

Example 5: Impure Array Function

var numbers = [1, 2, 3];

function impureDouble() {
    for (var i = 0; i < numbers.length; i++) {
        numbers[i] *= 2;
    }
}

impureDouble();
document.getElementById("output5").innerHTML = 
    "Modified original: " + numbers.join(", ") + "<br>" +
    "(Not pure - changed original array!)";

Key Points

  • Pure functions: same input → same output
  • Pure functions don't modify external state
  • Easier to test and debug
  • Enable better code maintainability
