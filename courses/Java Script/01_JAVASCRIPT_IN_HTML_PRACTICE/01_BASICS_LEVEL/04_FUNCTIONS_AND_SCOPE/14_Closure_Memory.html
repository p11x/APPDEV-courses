Closure - Memory

Understanding closures in loops, setTimeout, and common pitfalls.

Closures in Loops

// Common pitfall with var in loops
for (var i = 1; i <= 3; i++) {
    setTimeout(function() {
        console.log(i); // Prints 4, 4, 4 instead of 1, 2, 3
    }, 1000);
}

Examples

Example 1: Loop Problem with var

var funcs = [];
for (var i = 1; i <= 3; i++) {
    funcs.push(function() {
        return i;
    });
}

document.getElementById("output1").innerHTML = 
    "funcs[0](): " + funcs[0]() + "<br>" +
    "funcs[1](): " + funcs[1]() + "<br>" +
    "funcs[2](): " + funcs[2]();

Example 2: Using let (Block Scope)

var funcs = [];
for (let i = 1; i <= 3; i++) {
    funcs.push(function() {
        return i;
    });
}

document.getElementById("output2").innerHTML = 
    "funcs[0](): " + funcs[0]() + "<br>" +
    "funcs[1](): " + funcs[1]() + "<br>" +
    "funcs[2](): " + funcs[2]();

Example 3: setTimeout with Closures

function printDelayed(i) {
    setTimeout(function() {
        document.getElementById("output3").innerHTML += "Number: " + i + "<br>";
    }, i * 500);
}

for (var i = 1; i <= 3; i++) {
    printDelayed(i);
}

Example 4: Closure Retaining Variables

function createFunctions() {
    var funcs = [];
    for (var i = 0; i < 3; i++) {
        funcs.push((function(index) {
            return function() {
                return index;
            };
        })(i));
    }
    return funcs;
}

var functions = createFunctions();
document.getElementById("output4").innerHTML = 
    functions[0]() + ", " + functions[1]() + ", " + functions[2]();

Example 5: IIFE Pattern to Capture Value

var funcs = [];
for (var i = 1; i <= 3; i++) {
    (function(index) {
        funcs.push(function() {
            return index;
        });
    })(i);
}

document.getElementById("output5").innerHTML = 
    funcs[0]() + ", " + funcs[1]() + ", " + funcs[2]();

Key Points

  • Use let instead of var in loops to get block scope
  • IIFE can capture loop variable values
  • setTimeout creates closure over the variable
  • Closures can cause memory if not handled properly
