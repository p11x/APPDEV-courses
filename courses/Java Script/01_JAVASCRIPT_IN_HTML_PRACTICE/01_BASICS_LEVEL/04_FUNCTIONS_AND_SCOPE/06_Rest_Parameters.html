Rest Parameters

Rest parameters allow functions to accept an indefinite number of arguments as an array.

Syntax

function functionName(...args) {
    // args is an array
}

Examples

Example 1: Basic Rest Parameter

function sum(...numbers) {
    var total = 0;
    for (var i = 0; i < numbers.length; i++) {
        total += numbers[i];
    }
    return total;
}

document.getElementById("output1").innerHTML = 
    "sum(1, 2): " + sum(1, 2) + "
" +
    "sum(1, 2, 3, 4): " + sum(1, 2, 3, 4);

Example 2: With Regular Parameters

function greet(message, ...names) {
    var greetings = [];
    for (var i = 0; i < names.length; i++) {
        greetings.push(message + ", " + names[i] + "!");
    }
    return greetings.join("
");
}

document.getElementById("output2").innerHTML = 
    greet("Hello", "Alice", "Bob", "Charlie");

Example 3: Rest Parameter is an Array

function getTypes(...values) {
    return values.map(function(val) {
        return typeof val;
    });
}

document.getElementById("output3").innerHTML = 
    "Types: " + getTypes(1, "hello", true, 3.14);

Example 4: Array Methods

function findMax(...numbers) {
    return Math.max(...numbers);
}

function findMin(...numbers) {
    return Math.min(...numbers);
}

document.getElementById("output4").innerHTML = 
    "Max: " + findMax(3, 1, 9, 5, 7) + "
" +
    "Min: " + findMin(3, 1, 9, 5, 7);

Example 5: Arrow Function with Rest

var average = (...numbers) => {
    var sum = numbers.reduce((a, b) => a + b, 0);
    return numbers.length > 0 ? sum / numbers.length : 0;
};

document.getElementById("output5").innerHTML = 
    "Average [2,4,6]: " + average(2, 4, 6);

Example 6: Rest vs Arguments

// Rest parameter (modern way)
function showArgs(...args) {
    return "Rest array length: " + args.length;
}

// Arguments object (older way)
function showArgsOld() {
    return "Arguments length: " + arguments.length;
}

document.getElementById("output6").innerHTML = 
    showArgs(1, 2, 3) + "
" +
    showArgsOld(1, 2, 3, 4, 5);

Key Points

  • Syntax: ...args - collects remaining arguments
  • args is a real Array (not array-like)
  • Must be last parameter in function
  • Replaces the old arguments object
