# C and C++ Integration with Rcpp

## Learning Objectives

- Interface R with C++ using Rcpp
- Create functions callable from R
- Use .Call for C-level interfaces
- Write inline C++ code with Rcpp::cppFunction
- Pass R objects between R and C++

## Theoretical Background

### Rcpp Package

Rcpp enables seamless integration of C++ code in R packages. It provides:

1. **cppFunction()** - Write inline C++ code
2. **sourceCpp()** - Compile and load C++ from file
3. **Rcpp::export()** - Export C++ functions to R
4. **Rcpp attributes** - Automatic binding generation

### Benefits of C++ Integration

- **Speed**: C++ executes much faster for loops and heavy computation
- **Memory**: More efficient memory handling
- **Libraries**: Access to existing C/C++ libraries

### .Call Interface

.Call is the R API for calling compiled C code:
- Takes SEXP (S-expression) arguments
- Returns SEXP result
- Requires careful memory management
- Must use PROTECT/un PROTECT for garbage collection

## Code Examples

### Standard Example: Basic Rcpp

```r
# ===== RCPP BASICS =====

# Install Rcpp if needed
# install.packages("Rcpp")

cat("===== SIMPLE C++ FUNCTION =====\n")

# Inline C++ function - add numbers
library(Rcpp)

# Create and compile C++ function
cppFunction(
  includes = '
    #include <numeric>
  ',
  code = '
  int add_two(int a, int b) {
    return a + b;
  }
  ',
  depends = NA_character_
)

# Test the function
result <- add_two(5, 3)
cat("add_two(5, 3) =", result, "\n")

# More complex example - compute sum
cat("\n===== SUM VECTOR =====\n")

cppFunction(
  includes = '#include <Rcpp.h>',
  code = '
  // Sum elements of a numeric vector
  double vector_sum(NumericVector x) {
    double sum = 0;
    for (int i = 0; i < x.size(); i++) {
      sum += x[i];
    }
    return sum;
  }
  '
)

# Test
x <- c(1, 2, 3, 4, 5)
cat("sum of 1:5 =", vector_sum(x), "\n")
```

### Standard Example: Data Frame Operations

```r
# ===== DATA FRAME OPERATIONS =====
cat("\n===== DATA FRAME IN C++ =====\n")

cppFunction(
  includes = '
    #include <Rcpp.h>
    using namespace Rcpp;
  ',
  code = '
  // Calculate row sums of a DataFrame
  NumericVector row_sums(DataFrame df) {
    int ncol = df.size();
    int nrow = df.nrows();
    NumericVector result(nrow);
    
    for (int j = 0; j < ncol; j++) {
      NumericVector col = df[j];
      for (int i = 0; i < nrow; i++) {
        result[i] += col[i];
      }
    }
    return result;
  }
  
  // Extract numeric columns
  DataFrame select_numeric(DataFrame df) {
    int ncol = df.size();
    std::vector<std::string> numeric_cols;
    
    for (int j = 0; j < ncol; j++) {
      if (Rf_isReal(df[j])) {
        numeric_cols.push_back(df.names()[j]);
      }
    }
    
    DataFrame result = DataFrame::create(
      Named(numeric_cols[0]) = df[0],
      Named(numeric_cols[1]) = df[1]
    );
    return result;
  }
  '
)

# Create test data frame
df <- data.frame(
  a = c(1, 2, 3),
  b = c(4, 5, 6),
  c = c("x", "y", "z"),
  stringsAsFactors = FALSE
)

cat("Original data frame:\n")
print(df)

cat("\nRow sums:\n")
print(row_sums(df))
```

### Standard Example: Using sourceCpp

```r
# ===== SOURCECPP FOR FILES =====
cat("\n===== SOURCE FROM FILE =====\n")

# C++ source file content example
cpp_source <- '
// File: cpp_code.cpp

#include <Rcpp.h>
using namespace Rcpp;

// [[Rcpp::export]]
NumericMatrix invert_matrix(NumericMatrix input) {
  int n = input.nrow();
  NumericMatrix result(n, n);
  
  // Simple diagonal matrix creation
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++) {
      result(i, j) = (i == j) ? 1.0/input(i,i) : 0.0;
    }
  }
  return result;
}

// [[Rcpp::export]]
List fast_regression(NumericVector y, NumericMatrix X) {
  int n = X.nrow();
  int p = X.ncol();
  
  // Compute XtX
  NumericMatrix XtX(p, p);
  for (int i = 0; i < p; i++) {
    for (int j = 0; j < p; j++) {
      double sum = 0;
      for (int k = 0; k < n; k++) {
        sum += X(k, i) * X(k, j);
      }
      XtX(i, j) = sum;
    }
  }
  
  // Compute Xty
  NumericVector Xty(p);
  for (int i = 0; i < p; i++) {
    double sum = 0;
    for (int j = 0; j < n; j++) {
      sum += X(j, i) * y[j];
    }
    Xty[i] = sum;
  }
  
  return List::create(
    Named("XtX") = XtX,
    Named("Xty") = Xty
  );
}
'

cat("Example C++ file structure:\n")
cat(cpp_source)

# Note: In practice, save to .cpp file and run:

 regression_results <- fast_regression(y, X)
"
cat(usage_notes)
```

### Standard Example: Using .Call Directly

```r
# ===== DIRECT .CALL INTERFACE =====
cat("\n===== .CALL C INTERFACE =====\n")

c_code <- '
/* File: my_c_code.c */
/* Compile: R CMD SHLIB my_c_code.c */

#include <R.h>
#include <Rinternals.h>

/* Function to add two numbers */
SEXP add_ints(SEXP a, SEXP b) {
  int x = INTEGER(a)[0];
  int y = INTEGER(b)[0];
  SEXP result;
  
  PROTECT(result = allocVector(INTSXP, 1));
  INTEGER(result)[0] = x + y;
  
  UNPROTECT(1);
  return result;
}

/* Function to compute sum */
SEXP sum_vector(SEXP x) {
  int n = LENGTH(x);
  double *p = REAL(x);
  double total = 0.0;
  SEXP result;
  
  for (int i = 0; i < n; i++) {
    total += p[i];
  }
  
  PROTECT(result = allocVector(REALSXP, 1));
  REAL(result)[0] = total;
  
  UNPROTECT(1);
  return result;
}
'

cat("Example C code for .Call:\n")
cat(c_code)

cat("\n\n===== LOADING C CODE =====\n")
c_loading <- "
# Load compiled C code
dyn.load(\"my_c_code.so\")  # or .dll on Windows

# Call the C function
result <- .Call(\"add_ints\", as.integer(5), as.integer(3))

# Unload when done
dyn.unload(\"my_c_code.so\")
"
cat(c_loading)
```
